29  	protected CaptureSearchResults getSampleIndex() throws IOException, CDXFormatException {
30  		CaptureSearchResults results = new CaptureSearchResults();
31  		// XXX CDXFormat has a constant for "n", but does #getField() does not recognize it.
32  		// It recognizes "S" instead. Field names doesn't matter anyway..
33  		CDXFormat format = new CDXFlexFormat(" CDX A b a m s k r M V S g");
34  		CDXFormatToSearchResultAdapter parser = new CDXFormatToSearchResultAdapter(format);
35  		InputStream is = getClass().getResourceAsStream("test-cdx.txt");
36  		assertNotNull("classpath resource not found", is);
37  		BufferedReader br = new BufferedReader(new InputStreamReader(is, "UTF-8"));
38  		String line;
39  		while ((line = br.readLine()) != null) {
40  			CaptureSearchResult csr = parser.adapt(line);
41  			results.addSearchResult(csr);
42  		}
43  		return results;
44  	}

46  	public void testGetCaptureCalendar() throws Exception {
47  		WaybackRequest wbRequest = WaybackRequest.createReplayRequest(
48  			"http://archive.org", "20130801000000", "20130101000000", "20131231115959");
49  		UIResults uiResults = new UIResults(wbRequest, null, getSampleIndex());
50  		BubbleCalendarData data = new BubbleCalendarData(uiResults);
51  
52  		SimpleDateFormat df = new SimpleDateFormat("yyyyMMddHHmmss");
53  		df.setTimeZone(TimeZone.getTimeZone("UTC"));
54  
55  		data.setMonth(0);
56  		List<List<Partition<CaptureSearchResult>>> cal = data.getCaptureCalendar();
57  		assertEquals("Jan 2013 has five week rows", 5, cal.size());
58  
59  		// Jan 1, 2013 is Tuesday.
60  		List<Partition<CaptureSearchResult>> w1 = cal.get(0);
61  		assertNull(w1.get(0)); // Sunday
62  		assertNull(w1.get(1)); // Monday
63  		Partition<CaptureSearchResult> d1 = w1.get(2);
64  		assertEquals("20130101000000", df.format(d1.getStart()));
65  
66  		List<Partition<CaptureSearchResult>> w5 = cal.get(4);
67  		assertNull(w5.get(6)); // Saturday
68  		assertNull(w5.get(5)); // Friday
69  		Partition<CaptureSearchResult> d31 = w5.get(4);
70  		assertEquals("20130131000000", df.format(d31.getStart()));
71  	}
