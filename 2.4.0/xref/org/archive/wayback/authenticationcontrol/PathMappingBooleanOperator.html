1   package org.archive.wayback.authenticationcontrol;
2   
3   import java.util.Map;
4   
5   import org.archive.wayback.core.WaybackRequest;
6   import org.archive.wayback.util.operator.BooleanOperator;
7   import org.archive.wayback.webapp.AccessPoint;
8   
9   public class PathMappingBooleanOperator implements BooleanOperator<WaybackRequest> {
10  	
11  	protected Map<String, BooleanOperator<WaybackRequest>> pathMap;
12  	
13  	protected boolean noMatchesDefault = true;
14  
15  	@Override
16  	public boolean isTrue(WaybackRequest value) {
17  		AccessPoint ap = value.getAccessPoint();
18  		
19  		String path = ap.getAccessPointPath();
20  		
21  		if (path == null) {
22  			return noMatchesDefault;
23  		}
24  		
25  		BooleanOperator<WaybackRequest> op = pathMap.get(path);
26  		
27  		if (op == null) {
28  			return noMatchesDefault;
29  		}
30  		
31  		return op.isTrue(value);
32  	}
33  
34  	public Map<String, BooleanOperator<WaybackRequest>> getPathMap() {
35  		return pathMap;
36  	}
37  
38  	public void setPathMap(Map<String, BooleanOperator<WaybackRequest>> pathMap) {
39  		this.pathMap = pathMap;
40  	}
41  
42  	public boolean isNoMatchesDefault() {
43  		return noMatchesDefault;
44  	}
45  
46  	public void setNoMatchesDefault(boolean noMatchesDefault) {
47  		this.noMatchesDefault = noMatchesDefault;
48  	}
49  }
