9   /**
10   * Variant of {@link AuthToken} used to pass {@link AccessPoint} reference to
11   * {@link AuthChecker} implementations. Primary purpose is to configure
12   * {@link CDXAccessFilter} with AccessPoint-dependent information,
13   * such as {@link AccessPoint#getFileIncludePrefixes()}.
14   * <p>
15   * TODO: Considering AuthToken represents a user, it sounds like a bad design to have
16   * it carry target context information. It is in fact resulting in counter-intuitive data
17   * flow. Refactor the {@link CDXServer} interface.
18   * </p>
19   * @see EmbeddedCDXServerIndex#createAuthToken(org.archive.wayback.core.WaybackRequest, String)
20   * @see WaybackAPAuthChecker
21   *
22   */
23  public class APContextAuthToken extends AuthToken {
24  	final AccessPoint ap;
25  
26  	public APContextAuthToken(AccessPoint ap) {
27  		this.ap = ap;
28  	}
29  
30  	public AccessPoint getAccessPoint() {
31  		return ap;
32  	}
33  }
