54  	private final static Pattern WB_DATE_PATTERN = Pattern
55  		.compile("^(\\d{0,14})$");
56  
57  	@Override
58  	public WaybackRequest parse(HttpServletRequest httpRequest,
59  			AccessPoint accessPoint) throws BadQueryException,
60  			BetterRequestException {
61  
62  		if (!accessPoint.isEnableMemento()) {
63  			return super.parse(httpRequest, accessPoint);
64  		}
65  
66  		String acceptDateTime = httpRequest
67  			.getHeader(MementoUtils.ACCEPT_DATETIME);
68  
69  		// Memento TimeGate
70  		Date date = null;
71  
72  		String requestPath = accessPoint.translateRequestPathQuery(httpRequest);
73  
74  		// FIXME: this flag indicates if Accept-Datetime header had a non-null value,
75  		// not it's value is valid or not. equivalent of acceptDateTime != null.
76  		boolean invalidAcceptDateTime = false;
77  
78  		if (acceptDateTime != null) {
79  			date = MementoUtils.parseAcceptDateTimeHeader(acceptDateTime);
80  			invalidAcceptDateTime = true;
81  		}
82  
83  		if (date == null) {
84  			//TODO: Integrate with Accept-Datetime?
85  			String acceptTimestamp = httpRequest.getHeader("Accept-Timestamp");
86  
87  			if ((acceptTimestamp != null) &&
88  					WB_DATE_PATTERN.matcher(acceptTimestamp).matches()) {
89  				String timestamp = Timestamp.padEndDateStr(acceptTimestamp);
90  				date = ArchiveUtils.getDate(timestamp, null);
91  			}
92  		}
93  
94  		// Accept-Datetime specified but is invalid and timestamp not specified, must return a 400
95  		if (invalidAcceptDateTime && (date == null)) {
96  			throw new TimeGateBadQueryException(
97  				"Invald Memento TimeGate datetime request, Accept-Datetime: " +
98  						acceptDateTime, requestPath);
99  		}
100 
101 		WaybackRequest wbRequest = this.parse(requestPath, accessPoint, date);
102 
103 		if (wbRequest != null) {
104 			wbRequest.setResultsPerPage(getMaxRecords());
105 			wbRequest.setMementoTimegate();
106 		}
107 
108 		return wbRequest;
109 	}

111 	public WaybackRequest parse(String requestPath, AccessPoint accessPoint)
112 			throws BetterRequestException, BadQueryException {
113 
114 		return parse(requestPath, accessPoint, null);
115 	}

117 	public WaybackRequest parse(String requestPath, AccessPoint accessPoint,
118 			Date mementoDate) throws BetterRequestException, BadQueryException {
119 		/*
120 		 * 
121 		 * We're trying to catch requests without a datespec, in which case, we
122 		 * just redirect to the same request, inserting today's datespec, and
123 		 * then we'll let the normal redirection occur.
124 		 * 
125 		 * The one tricky point is that we don't want to defeat the
126 		 * server-relative redirection handling, so we want to do some
127 		 * inspection to make sure it actually looks like an URL, and not like:
128 		 * 
129 		 * images/foo.gif redirect.php?blargity=blargblarg
130 		 * 
131 		 * What would be perfect is if the user supplied http:// at the front.
132 		 * 
133 		 * So, we'll assume that if we see that, we either match, or throw a
134 		 * BadQueryException.
135 		 */
136 
137 		String scheme = UrlOperations.urlToScheme(requestPath);
138 		if (scheme == null) {
139 			// if it has "http:/" instead of "http://", repair it.
140 			// (some client canonicalizes "//" in path into "/".)
141 			if (requestPath.startsWith("http:/")) {
142 				requestPath = "http://" + requestPath.substring(6);
143 				scheme = "http://";
144 			}
145 		}
146 		if (scheme == null) {
147 			try {
148 				URL u = new URL(UrlOperations.HTTP_SCHEME + requestPath);
149 				// does the authority look legit?
150 				if (u.getUserInfo() != null) {
151 					throw new BadQueryException(
152 						"Unable to handle URLs with user information");
153 				}
154 
155 				if (UrlOperations.isAuthority(u.getAuthority())) {
156 					// ok, we're going to assume this is good:
157 					return handleDatelessRequest(accessPoint, requestPath,
158 						mementoDate);
159 				}
160 
161 			} catch (MalformedURLException e) {
162 				// eat it silently
163 			}
164 		} else {
165 			// OK, we're going to assume this is a replay request, sans timestamp,
166 			// ALWAYS redirect:
167 			return handleDatelessRequest(accessPoint, requestPath, mementoDate);
168 		}
169 		return null;
170 	}

172 	protected WaybackRequest handleDatelessRequest(AccessPoint accessPoint,
173 			String requestPath, Date mementoDate) throws BetterRequestException {
174 //		String nowTS = Timestamp.currentTimestamp().getDateStr();
175 //		String newUrl = accessPoint.getUriConverter().makeReplayURI(nowTS, requestPath);
176 //		throw new BetterRequestException(newUrl);
177 
178 		WaybackRequest wbRequest = new WaybackRequest();
179 
180 		if (wbRequest.getStartTimestamp() == null) {
181 			wbRequest.setStartTimestamp(getEarliestTimestamp());
182 		}
183 
184 		if (wbRequest.getEndTimestamp() == null) {
185 			wbRequest.setEndTimestamp(getLatestTimestamp());
186 		}
187 
188 		if (mementoDate == null) {
189 			mementoDate = new Date();
190 			wbRequest.setBestLatestReplayRequest();
191 		}
192 
193 		wbRequest.setReplayDate(mementoDate);
194 		wbRequest.setAnchorDate(mementoDate);
195 		wbRequest.setReplayRequest();
196 		wbRequest.setRequestUrl(requestPath);
197 
198 		return wbRequest;
199 	}
