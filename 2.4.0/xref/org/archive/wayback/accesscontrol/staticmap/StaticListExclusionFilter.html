3   import java.util.TreeSet;
4   import java.util.logging.Logger;
5   
6   import org.apache.commons.httpclient.URIException;
7   import org.archive.util.SURT;
8   import org.archive.wayback.UrlCanonicalizer;
9   import org.archive.wayback.core.CaptureSearchResult;
10  import org.archive.wayback.resourceindex.filters.ExclusionFilter;
11  import org.archive.wayback.util.ObjectFilter;
12  import org.archive.wayback.util.url.AggressiveUrlCanonicalizer;
13  
14  public class StaticListExclusionFilter extends ExclusionFilter {
15  	private static final Logger LOGGER = Logger.getLogger(
16  			StaticMapExclusionFilter.class.getName());
17  
18  	private String lastChecked = null;
19  	private boolean lastCheckedExcluded = false;
20  	private boolean notifiedSeen = false;
21  	private boolean notifiedPassed = false;
22  	TreeSet<String> exclusions = null;
23  	UrlCanonicalizer canonicalizer = new AggressiveUrlCanonicalizer();
24          
25  	/**
26           * @param exclusions
27           * @param canonicalizer
28  	 */
29  	public StaticListExclusionFilter(TreeSet<String> exclusions, UrlCanonicalizer canonicalizer) {
30  		this.exclusions = exclusions;
31  		this.canonicalizer = canonicalizer;
32  	}
33  	
34  	protected boolean isExcluded(String surt) {
35  	    String possiblePrefix = exclusions.floor(surt);
36  	    return (possiblePrefix != null && surt.startsWith(possiblePrefix));
37  	}
38  
39  	/* (non-Javadoc)
40  	 * @see org.archive.wayback.resourceindex.SearchResultFilter#filterSearchResult(org.archive.wayback.core.SearchResult)
41  	 */
42          @Override
43  	public int filterObject(CaptureSearchResult r) {
44  		if(!notifiedSeen) { 
45  			if(filterGroup != null) {
46  				filterGroup.setSawAdministrative();
47  			}
48  			notifiedSeen = true;
49  		}
50  		String surt;
51  		try {
52  			String url = canonicalizer.urlStringToKey(r.getOriginalUrl());
53  			surt = SURT.fromPlain(url);
54  //			surt = SURTTokenizer.prefixKey(url);
55  		} catch (URIException e) {
56  			
57  			//e.printStackTrace();
58  			return FILTER_EXCLUDE;
59  		}
60  		if(lastChecked != null) {
61  			if(lastChecked.equals(surt)) {
62  				if(lastCheckedExcluded) { 
63  					return ObjectFilter.FILTER_EXCLUDE;
64  				} else {
65  					// don't need to: already did last time...
66  					//filterGroup.setPassedAdministrative();
67  					return ObjectFilter.FILTER_INCLUDE;
68  				}
69  			}
70  		}
71  		lastChecked = surt;
72  		lastCheckedExcluded = isExcluded(surt);
73  		if(lastCheckedExcluded) {
74  			return ObjectFilter.FILTER_EXCLUDE;
75  		} else {
76  			if(!notifiedPassed) {
77  				if(filterGroup != null) {
78  					filterGroup.setPassedAdministrative();
79  				}
80  				notifiedPassed = true;
81  			}
82  			return ObjectFilter.FILTER_INCLUDE;
83  		}
84  			
85  	}
86  	
87  }
