44  public class WarcResource extends Resource {
45  	private WARCRecord rec = null;
46  	private ArchiveReader reader = null;
47  	private Map<String, String> headers = null;
48  	private long length = 0;
49  	private int status = 0;
50  	private boolean parsedHeaders = false;
51  	public WarcResource(WARCRecord rec, ArchiveReader reader) {
52  		this.rec = rec;
53  		this.reader = reader;
54  	}

56  	/**
57  	 * @param bytes Array of bytes to examine for an EOL.
58  	 * @return Count of end-of-line characters or zero if none.
59  	 */
60  	private int getEolCharsCount(byte [] bytes) {
61  		int count = 0;
62  		if (bytes != null && bytes.length >=1 &&
63  				bytes[bytes.length - 1] == '\n') {
64  			count++;
65  			if (bytes.length >=2 && bytes[bytes.length -2] == '\r') {
66  				count++;
67  			}
68  		}
69  		return count;
70  	}

72  	public void parseHeaders() throws IOException {
73  		if(parsedHeaders) {
74  			return;
75  		}
76  
77  		// If warc or arc record is 0 length, don't do any more parsing!
78  		// Hopefully caller code will check this before proceeding as well
79  		if (getRecordLength() <= 0) {
80  			parsedHeaders = true;
81  			return;
82  		}
83  
84  		// WARCRecord should have getRecordType() method returning WARCRecordType.
85  		String rectypeStr = (String)rec.getHeader().getHeaderValue("WARC-Type");
86  		WARCRecordType rectype;
87  		try {
88  		    rectype = WARCRecordType.valueOf(rectypeStr);
89  		} catch (IllegalArgumentException ex) {
90  		    throw new RecoverableIOException("unrecognized WARC-Type \"" + rectypeStr + "\"");
91  		}
92  
93  		if (rectype == WARCRecordType.response || rectype == WARCRecordType.revisit) {
94  		    byte [] statusBytes = LaxHttpParser.readRawLine(rec);
95  		    int eolCharCount = getEolCharsCount(statusBytes);
96  		    if (eolCharCount <= 0) {
97  		        throw new RecoverableIOException("Failed to read http status where one " +
98  		                " was expected: " + new String(statusBytes));
99  		    }
100 		    String statusLineStr = EncodingUtil.getString(statusBytes, 0,
101 		            statusBytes.length - eolCharCount, ARCConstants.DEFAULT_ENCODING);
102 		    if ((statusLineStr == null) ||
103 		            !StatusLine.startsWithHTTP(statusLineStr)) {
104 		        throw new RecoverableIOException("Failed parse of http status line.");
105 		    }
106 		    StatusLine statusLine = new StatusLine(statusLineStr);
107 
108 		    this.status = statusLine.getStatusCode();
109 
110 		    Header[] tmpHeaders = LaxHttpParser.parseHeaders(rec,
111 		            ARCConstants.DEFAULT_ENCODING);
112 		    headers = new Hashtable<String,String>();
113 		    this.setInputStream(rec);
114 		    for(Header header: tmpHeaders) {
115 		        headers.put(header.getName(), header.getValue());
116 		        if(header.getName().toUpperCase().contains(
117 		                HttpHeaderOperation.HTTP_TRANSFER_ENC_HEADER)) {
118 		            if(header.getValue().toUpperCase().contains(
119 		                    HttpHeaderOperation.HTTP_CHUNKED_ENCODING_HEADER)) {
120 		                setChunkedEncoding();
121 		            }
122 		        }
123 		    }
124 		} else if (rectype == WARCRecordType.metadata || rectype == WARCRecordType.resource) {
125 		    status = 200;
126 		    headers = new HashMap<String, String>();
127 		    String ct = (String)rec.getHeader().getHeaderValue("Content-Type");
128 		    if (ct != null) {
129 		        headers.put("Content-Type", ct);
130 		    }
131 		    // necessary?
132 		    String date = rec.getHeader().getDate();
133 		    if (date != null) {
134 		        try {
135 		            Date d = org.apache.commons.lang.time.DateUtils.parseDate(date, new String[] { "yyyy-MM-dd'T'HH:mm:ss'Z'"});
136 		            String httpDate = DateUtils.getRFC1123Date(d);
137 		            headers.put("Date", httpDate);
138 		        } catch (ParseException ex) {
139 		            //
140 		        }
141 		    }
142 		    setInputStream(rec);
143 		}
144 		parsedHeaders = true;
145 	}
146 
147 	@Override
148 	public Map<String, String> getHttpHeaders() {
149 		return headers;
150 	}
151 
152 	public ArchiveRecordHeader getWarcHeaders() {
153 		return rec.getHeader();
154 	}
155 
156 	@Override
157 	public long getRecordLength() {
158 		if ((length == 0) && (rec.getHeader() != null)) {
159 			length = rec.getHeader().getContentLength();
160 		}
161 		return length;
162 	}
163 
164 	@Override
165 	public int getStatusCode() {
166 		return status;
167 	}
168 
169 	@Override
170 	public void close() throws IOException {
171 		rec.close();
172 		reader.close();
173 	}
174 
175 	public String getRefersToTargetURI() {
176 		return (String)getWarcHeaders().getHeaderFields().get(
177 			"WARC-Refers-To-Target-URI");
178 	}
179 
180 	public String getRefersToDate() {
181 		String dateString = (String)getWarcHeaders().getHeaderFields().get(
182 			"WARC-Refers-To-Date");
183 		if (dateString != null) {
184 			Date date = ArchiveUtils.parse14DigitISODate(dateString, null);
185 			if (date != null) {
186 				return ArchiveUtils.get14DigitDate(date);
187 			}
188 		}
189 		return null;
190 	}
191 
192 	public static final String PROFILE_REVISIT_SERVER_NOT_MODIFIED =
193 			"http://netpreserve.org/warc/1.0/revisit/server-not-modified";
194 
195 	/**
196 	 * whether this Resource is {@code server-not-modified} revisit.
197 	 * (this method used to be {@code AccessPoint#isWarcRevisitNotModified(Resource)}.
198 	 * Not made a part of {@code Resource} interface because it was unused.)
199 	 * @return {@code true} if it is
200 	 */
201 	public boolean isRevisitNotModified() {
202 		Map<String, Object> warcHeaders = getWarcHeaders().getHeaderFields();
203 		String warcProfile = (String)warcHeaders.get("WARC-Profile");
204 		return PROFILE_REVISIT_SERVER_NOT_MODIFIED.equals(warcProfile);
205 	}
206 }
