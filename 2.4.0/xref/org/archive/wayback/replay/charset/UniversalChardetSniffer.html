1   package org.archive.wayback.replay.charset;
2   
3   import java.io.IOException;
4   
5   import org.archive.wayback.core.Resource;
6   import org.mozilla.universalchardet.UniversalDetector;
7   
8   /**
9    * {@link EncodingSniffer} that runs {@link UniversalDetector} on
10   * the content.
11   * <p>Note: as of version 1.0.3, UniversalDetector returns {@code null}
12   * for ASCII-only text encoded as either {@code UTF-8} or {@code UTF-16}.</p>
13   */
14  public class UniversalChardetSniffer extends BaseEncodingSniffer {
15  	// hand off this many bytes to the chardet library
16  	protected final static int MAX_CHARSET_READAHEAD = 65536;
17  
18  	@Override
19  	public String sniff(Resource resource) {
20  		String charsetName = null;
21  
22  		byte[] bbuffer = new byte[MAX_CHARSET_READAHEAD];
23  		// (1)
24  		UniversalDetector detector = new UniversalDetector(null);
25  
26  		// (2)
27  		resource.mark(MAX_CHARSET_READAHEAD);
28  		try {
29  			int len = resource.read(bbuffer, 0, MAX_CHARSET_READAHEAD);
30  			resource.reset();
31  			detector.handleData(bbuffer, 0, len);
32  		} catch (IOException ex) {
33  			//
34  		}
35  		// (3)
36  		detector.dataEnd();
37  		// (4)
38  		charsetName = detector.getDetectedCharset();
39  
40  		// (5)
41  		detector.reset();
42  		if (isCharsetSupported(charsetName)) {
43  			return charsetName;
44  		}
45  		return null;
46  	}
47  }
