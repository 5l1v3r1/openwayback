42  /**
43   * {@link ReplayRenderer} for rewriting textual resource with {@link TextDocument}.
44   * <p>
45   * {@link Resource} is first decoded (if {@code Transfer-Encoding} is applied),
46   * and fully loaded into {@code TextDocument} object. Then {@link #updatePage}
47   * method is called for actual rewrite. Then updated content is sent to the response.
48   * </p>
49   * <p>Customization Properties:
50   * <ul>
51   * <li>{@code jspInserts}: a list of Servlets for annotation inserts. It is
52   * {@code updatePage}'s responsibility to perform actual insertion.</li>
53   * <li>{@code charsetDetector}: {@link CharsetDetector} for detecting resource's
54   * charset. Default is {@link StandardCharsetDetector}.</li>
55   * <li>{@code httpHeaderProcessor} (constructor arg): {@link HttpHeaderProcessor}
56   * for rewriting resource's HTTP headers.</li>
57   * <li>{@code guessedCharsetHeader}: name of response HTTP header for sending
58   * out resource's charset detected by {@code charsetDetector}. Default is
59   * {@link #GUESSED_CHARSET_HEADER}.</li>
60   * <li>{@code pageURIConverterFactory}: You can use different URI conversion
61   * just for {@code TextDocument} and {#updatePage} by setting this property 
62   * to non-{@code null} {@code ContextResultURIConverterFactory}. It does not
63   * affect URI conversion for HTTP headers.</li>
64   * </ul>
65   * </p>
66   * @author brad
67   */
68  public abstract class TextReplayRenderer implements ReplayRenderer {

101 	@Override
102 	public void renderResource(HttpServletRequest httpRequest,
103 			HttpServletResponse httpResponse, WaybackRequest wbRequest,
104 			CaptureSearchResult result, Resource httpHeadersResource,
105 			Resource payloadResource, ResultURIConverter uriConverter,
106 			CaptureSearchResults results) throws ServletException,
107 			IOException, BadContentException {
108 
109 		// Decode resource (such as if gzip encoded)
110 		Resource decodedResource = decodeResource(httpHeadersResource, payloadResource);
111 		
112 		HttpHeaderOperation.copyHTTPMessageHeader(httpHeadersResource, httpResponse);
113 
114 		Map<String,String> headers = HttpHeaderOperation.processHeaders(
115 				httpHeadersResource, result, uriConverter, httpHeaderProcessor);
116 
117 		String charSet = charsetDetector.getCharset(httpHeadersResource,
118 				decodedResource, wbRequest);
119 
120 		ResultURIConverter pageConverter = uriConverter;
121 		if (pageConverterFactory != null) {
122 			// XXX: ad-hoc code - ContextResultURIConverterFactory should take ResultURIConverter
123 			// as argument, so that it can simply wrap the original.
124 			String replayURIPrefix = (uriConverter instanceof ArchivalUrlResultURIConverter ?
125 					((ArchivalUrlResultURIConverter)uriConverter).getReplayURIPrefix() : "");
126 			ResultURIConverter ruc = pageConverterFactory.getContextConverter(replayURIPrefix);
127 			if (ruc != null)
128 				pageConverter = ruc;
129 		}
130 		// Load content into an HTML page, and resolve load-time URLs:
131 		TextDocument page = new TextDocument(decodedResource, result,
132 				uriConverter);
133 		page.readFully(charSet);
134 
135 		updatePage(page, httpRequest, httpResponse, wbRequest, result,
136 				decodedResource, pageConverter, results);
137 
138 		// set the corrected length:
139 		int bytes = page.getBytes().length;
140 		headers.put(HttpHeaderOperation.HTTP_LENGTH_HEADER, String.valueOf(bytes));
141 		if (guessedCharsetHeader != null) {
142 			headers.put(guessedCharsetHeader, page.getCharSet());
143 		}
144 
145 		// send back the headers:
146 		HttpHeaderOperation.sendHeaders(headers, httpResponse);
147 
148 		// Tomcat will always send a charset... It's trying to be smarter than
149 		// we are. If the original page didn't include a "charset" as part of
150 		// the "Content-Type" HTTP header, then Tomcat will use the default..
151 		// who knows what that is, or what that will do to the page..
152 		// let's try explicitly setting it to what we used:
153 		httpResponse.setCharacterEncoding(page.getCharSet());
154 
155 		page.writeToOutputStream(httpResponse.getOutputStream());
156 	}

207 	/**
208 	 * Returns a compression-decoding wrapper Resource if Resource has {@code Content-Encoding: gzip} or {@code br}.
209 	 * Otherwise returns {@code payloadResource} unmodified.
210 	 * <p>As a side-effect, {@code Content-Encoding} and
211 	 * {@code Transfer-Encoding} headers are removed from {@code headersResource} (this happens only when
212 	 * the resource was decompressed). It is assumed that {@code headerResource} and
213 	 * {@code payloadResource} are captures of identical response content.</p>
214 	 * <p>TODO: XArchiveHttpHeaderProcessor also does HTTP header removal. Check for refactoring case.</p>
215 	 * @param headersResource Resource to read HTTP headers from.
216 	 * @param payloadResource Resource to read content from (same as {@code headerResource} for regular captures,
217 	 * 	different Resource if headersResource is a revisit record.)
218 	 * @return The decoded Resource.
219 	 */
220 	public static Resource decodeResource(Resource headersResource, Resource payloadResource) throws IOException {
221 		Map<String, String> headers = headersResource.getHttpHeaders();
222 
223 		if (headers != null) {
224 			String encoding = HttpHeaderOperation.getHeaderValue(headers,
225 					HttpHeaderOperation.HTTP_CONTENT_ENCODING);
226 			if (encoding != null) {
227 				// mark current position so we can recover if decoding fails
228 				if (payloadResource.markSupported()) {
229 					payloadResource.mark(32);
230 				}
231 
232 				Resource decodingResource;
233 				try {
234 					decodingResource = DecodingResource.forEncoding(encoding, payloadResource);
235 				} catch (IOException e) {
236 					// this might be because the content is not actually compressed and the header
237 					// is wrong so may as well reset and attempt to continue without decoding
238 					if (payloadResource.markSupported()) {
239 						payloadResource.reset();
240 					}
241 					decodingResource = payloadResource;
242 				}
243 
244 				if (decodingResource != null) {
245 					headers.put(ORIG_ENCODING, encoding);
246 					HttpHeaderOperation.removeHeader(headers,
247 							HttpHeaderOperation.HTTP_CONTENT_ENCODING);
248 
249 					if (HttpHeaderOperation.isChunkEncoded(headers)) {
250 						HttpHeaderOperation.removeHeader(headers,
251 								HttpHeaderOperation.HTTP_TRANSFER_ENC_HEADER);
252 					}
253 
254 					return decodingResource;
255 				}
256 
257 				// TODO: check for other encodings?
258 			}
259 		}
260 
261 		return payloadResource;
262 	}

276 	/**
277 	 * return text to insert.
278 	 * <p>{@code jspInserts} are executed in sequence, concatenating
279 	 * their output.</p>
280 	 * @param page for calling {@code includeJspString} method
281 	 * @param httpRequest incoming request
282 	 * @param httpResponse outgoing response
283 	 * @param wbRequest wayback request info
284 	 * @param results captures
285 	 * @param result capture being replayed
286 	 * @param resource resource being replayed
287 	 * @return concatenated output of {@code jspInserts}
288 	 *  (empty if no {@code jspInserts} is configured.)
289 	 * @throws IOException error from {@link TextDocument#includeJspString}
290 	 * @throws ServletException error from {@link TextDocument#includeJspString}
291 	 */
292 	protected CharSequence buildInsertText(TextDocument page,
293 			HttpServletRequest httpRequest, HttpServletResponse httpResponse,
294 			WaybackRequest wbRequest, CaptureSearchResults results,
295 			CaptureSearchResult result, Resource resource) throws IOException, ServletException {
296 		if (jspInserts == null || jspInserts.isEmpty())
297 			return "";
298 		StringBuilder toInsert = new StringBuilder(300);
299 		for (String jspName : jspInserts) {
300 			toInsert.append(page.includeJspString(jspName, httpRequest, httpResponse, wbRequest, results, result, resource));
301 		}
302 		return toInsert;
303 	}
