3   import org.archive.wayback.core.CaptureSearchResult;
4   import org.archive.wayback.core.CaptureSearchResults;
5   import org.archive.wayback.core.WaybackRequest;
6   import org.archive.wayback.webapp.AccessPoint;

8   /**
9    * Given {@link WaybackRequest} and {@link CaptureSearchResults},
10   * {@code ReplayCaptureSelector} returns {@link CaptureSearchResult} in the
11   * order of preference to fulfill the replay request.
12   * <p>
13   * After setting {@code request} and {@code captures}, the first call to
14   * {@link #next()} returns the best capture to fulfill the request (typically
15   * the one <em>closest</em> to the requested date). As it may disregard some
16   * captures not suitable for the request, this first call to {@code next()} may
17   * return {@code null} even though there are one or more captures in
18   * {@code captures}.
19   * </p>
20   * <p>
21   * Subsequent call to {@code next()} returns the second best candidate, third
22   * best candidate, and so on, until it returns {@code null}, which indicates
23   * there's no more captures suitable for the request.
24   * </p>
25   * <p>
26   * {@link #setCaptures(CaptureSearchResults)} may be called to re-initialize it
27   * with new set of captures. Next call to {@code next()} will return the best
28   * capture within the capture set.
29   * </p>
30   * <p>
31   * Note that {@code ReplayCaptureSelector} evaluates captures' suitability for
32   * the request solely on the information available in {@code WaybackRequest} and
33   * {@code CaptureSearchResults}. Returned captures may turn out to be unsuitable
34   * for the request for other reasons. For example, a capture may be unusable
35   * because its record is inaccessible or it is a self-redirect.
36   * </p>
37   * @see AccessPoint
38   */
39  public interface ReplayCaptureSelector {
40  	/**
41  	 * set request information.
42  	 * <p>
43  	 * {@code AccessPoint} calls this method just once on each instance.
44  	 * </p>
45  	 * @param wbRequest WaybackRequest for the replay request
46  	 */
47  	public abstract void setRequest(WaybackRequest wbRequest);
48  
49  	/**
50  	 * set a list of captures to select replay candidate from.
51  	 * <p>
52  	 * {@code AccessPoint} may call this method more than once with different
53  	 * instance of {@code CaptureSearchResults} within a request. When this
54  	 * method is called, next {@code next()} call shall return the best capture
55  	 * in new {@code CaptureSearchResults}.
56  	 * </p>
57  	 * <p>
58  	 * Implementation can make changes to CaptureSearchResults for bookkeeping
59  	 * purposes etc.
60  	 * </p>
61  	 * @param captures CaptureSearchResults, non-{@code null}
62  	 */
63  	public abstract void setCaptures(CaptureSearchResults captures);
64  
65  	/**
66  	 * return next best capture.
67  	 * @return CaptureSearchResult, or {@code null} if there's no more captures
68  	 *         suitable for the request.
69  	 */
70  	public abstract CaptureSearchResult next();
71  }
