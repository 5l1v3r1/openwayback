20  package org.archive.wayback.replay;
21  
22  import java.io.IOException;
23  import java.io.OutputStream;
24  import java.util.Map;
25  
26  import javax.servlet.ServletException;
27  import javax.servlet.http.HttpServletRequest;
28  import javax.servlet.http.HttpServletResponse;
29  
30  import org.archive.wayback.ReplayRenderer;
31  import org.archive.wayback.ResultURIConverter;
32  import org.archive.wayback.core.CaptureSearchResult;
33  import org.archive.wayback.core.CaptureSearchResults;
34  import org.archive.wayback.core.Resource;
35  import org.archive.wayback.core.WaybackRequest;
36  import org.archive.wayback.exception.SpecificCaptureReplayException;
37  
38  /**
39   * ReplayRenderer implementation which returns the archive document as 
40   * pristinely as possible -- no modifications to response code, HTTP headers,
41   * or original byte-stream.
42   *
43   * <p>
44   * Now this class has a limited support for range requests. If the request
45   * has {@code Range} header, it tries to fulfill it by extracting requested
46   * byte range from the archived content (either 200 or 206 responses).
47   * If the Resource is not suitable for replaying the request ranges,
48   * {@code renderResource} throws {@link RangeNotSatisfiable}, so that
49   * replay process can try another capture, or take other actions.
50   * </p>
51   *
52   * @author brad
53   * @version $Date$, $Revision$
54   */
55  public class TransparentReplayRenderer implements ReplayRenderer {
56  	private HttpHeaderProcessor httpHeaderProcessor;
57  	
58  	// TODO: Figure out best way to generalize this, but probably good default
59  	// Add special don't cache header in case of at least 100M
60  	private final static long NOCACHE_THRESHOLD = 100000000L;
61  
62  	private final static String NOCACHE_HEADER_NAME = "X-Accel-Buffering";
63  	private final static String NOCACHE_HEADER_VALUE = "no";
64  	
65  	private final static int BUFFER_SIZE = 4096;
66  	public TransparentReplayRenderer(HttpHeaderProcessor httpHeaderProcessor) {
67  		this.httpHeaderProcessor = httpHeaderProcessor;
68  	}
69  
70  	@Override
71  	public void renderResource(HttpServletRequest httpRequest,
72  			HttpServletResponse httpResponse, WaybackRequest wbRequest,
73  			CaptureSearchResult result, Resource resource,
74  			ResultURIConverter uriConverter, CaptureSearchResults results)
75  			throws ServletException, IOException,
76  			SpecificCaptureReplayException {
77  		String rangeHeader = httpRequest.getHeader(HttpHeaderOperation.HTTP_RANGE_HEADER_UP);
78  		if (rangeHeader != null) {
79  			long[][] ranges = HttpHeaderOperation.parseRanges(rangeHeader);
80  			@SuppressWarnings("resource")
81  			RangeResource rangeResource = new RangeResource(resource,
82  				ranges);
83  			rangeResource.parseRange();
84  			resource = rangeResource;
85  		}
86  
87  		HttpHeaderOperation.copyHTTPMessageHeader(resource, httpResponse);
88  
89  		// Note: httpHeaderProcessor may remove Content-Length.
90  		Map<String, String> headers = HttpHeaderOperation.processHeaders(
91  			resource, result, uriConverter, httpHeaderProcessor);
92  
93  		String origLength = HttpHeaderOperation.getContentLength(resource.getHttpHeaders());
94  		if (origLength != null) {
95  			HttpHeaderOperation.replaceHeader(headers, HttpHeaderOperation.HTTP_LENGTH_HEADER, origLength);
96  
97  			long contentLength = -1;
98  			
99  			try {
100 			    contentLength = Long.parseLong(origLength);
101 			} catch (NumberFormatException n) {
102 			    
103 			}
104 			
105 			//TODO: Generalize? Don't buffer NOCACHE_THRESHOLD
106 			if ((contentLength >= NOCACHE_THRESHOLD)) {
107 			    headers.put(NOCACHE_HEADER_NAME, NOCACHE_HEADER_VALUE);
108 			}
109 		}
110 
111 		HttpHeaderOperation.sendHeaders(headers, httpResponse);
112 
113 		// and copy the raw byte-stream.
114 		OutputStream os = httpResponse.getOutputStream();
115 		byte[] buffer = new byte[BUFFER_SIZE];
116 		long total = 0;
117 		for (int r = -1; (r = resource.read(buffer, 0, BUFFER_SIZE)) != -1;) {
118 			os.write(buffer, 0, r);
119 			total += r;
120 		}
121 		if (total == 0) {
122 			if (headers.size() == 0) {
123 				// totally empty response
124 				httpResponse.setContentLength(0);
125 			}
126 		}
127 	}
128 
129 	@Override
130 	public void renderResource(HttpServletRequest httpRequest,
131 			HttpServletResponse httpResponse, WaybackRequest wbRequest,
132 			CaptureSearchResult result, Resource httpHeadersResource,
133 			Resource payloadResource, ResultURIConverter uriConverter,
134 			CaptureSearchResults results) throws ServletException, IOException,
135 			SpecificCaptureReplayException {
136 		if (httpHeadersResource != payloadResource) {
137 			httpHeadersResource = new CompositeResource(
138 				httpHeadersResource, payloadResource);
139 		}
140 		renderResource(httpRequest, httpResponse, wbRequest, result,
141 			httpHeadersResource, uriConverter, results);
142 	}
143 }
