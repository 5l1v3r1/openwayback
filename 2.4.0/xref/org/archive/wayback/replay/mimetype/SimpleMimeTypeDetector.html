14  /**
15   * Simple {@link MimeTypeDetector} implementation.
16   * It's ad-hoc and not customizable, just tested against many samples.
17   */
18  public class SimpleMimeTypeDetector implements MimeTypeDetector {
19  	private static final Logger logger = Logger.getLogger(SimpleMimeTypeDetector.class.getName());

58  	private static final String BINARY_FILE = "application/octet-stream";
59  
60  	/**
61  	 * Return {@code true} if {@code bytes} looks like a beginning of a
62  	 * binary file.
63  	 * <p>Looks for well-known binary format MAGICs.</p>
64  	 * @param bytes array of bytes.
65  	 * @return detected mimetype, or {@code null} if no binary
66  	 * format is detected.
67  	 */
68  	private String detectBinaryTypes(byte[] bytes) {
69  		// NB: there are a lot of specific mimetypes we can detect by file
70  		// magic, but we don't want to make too much effort here.
71  		// "application/octet-stream" is fine if file looks very much like
72  		// binary but don't know what exactly. Returned value is not really
73  		// used beyond being non-null. We're only concerned with file formats
74  		// frequently sent out with imprecise Content-Type.
75  		switch (bytes[0]) {
76  		case (byte)0xFF:
77  			if (bytes[1] == (byte)0xFE) {
78  				// UTF-16LE BOM
79  				return null;
80  			} else if ((bytes[1] & 0xFE) == (byte)0xFA) {
81  				// audio/mp3
82  				return "audio/mp3";
83  			} else if (bytes[1] == (byte)0xD8) {
84  				// image/jpeg - commonly <FF><D8><FF><E0> or <FF><D8><FF><E1>
85  				return "image/jpeg";
86  			}
87  			// WordPerfect (<FF>WPC) falls in this category.
88  			// (WordPerfect also has <D8>WPC
89  			return BINARY_FILE;
90  		case (byte)0xFE:
91  			if (bytes[1] == (byte)0xFF) {
92  				// UTF-16BE BOM
93  				return null;
94  			}
95  			break;
96  		case (byte)0xF7:
97  			if (bytes[1] == 0x02 && bytes[2] == 0x01) {
98  				// unconfirmed.
99  				return "application/x-dvi";
100 			}
101 			break;
102 		case (byte)0xEF:
103 			if (bytes[1] == (byte)0xBB && bytes[2] == (byte)0xBF) {
104 				// UTF-8 BOM
105 				return null;
106 			}
107 			break;
108 		case (byte)0xD0:
109 			if (bytes[1] == (byte)0xCF && bytes[2] == 0x11 && bytes[2] == (byte)0xE1) {
110 				// MS Word.Document.8
111 				return BINARY_FILE;
112 			}
113 			break;
114 		case (byte)0xCA:
115 			if (bytes[1] == (byte)0xFE && bytes[2] == (byte)0xBA && bytes[3] == (byte)0xBE) {
116 				// application/java (class files)
117 				return "application/java";
118 			}
119 			break;
120 		case (byte)0x89:
121 			if (bytes[1] == 'P' && bytes[2] == 'N' && bytes[3] == 'G') {
122 				return "image/png";
123 			}
124 			break;
125 		case 0x00:
126 			if (bytes[1] == 0x00) {
127 				// Windows icon resource falls in this category.
128 				return BINARY_FILE;
129 			} else if (bytes[1] == 0x01) {
130 				// TTF?
131 				return BINARY_FILE;
132 			}
133 			break;
134 		case 0x01:
135 			if (bytes[1] == (byte)0xB3 || bytes[1] == (byte)0xBA) {
136 				return "video/mpeg";
137 			} else if (bytes[1] == 0x00) {
138 				// very likely is a binary file
139 				return BINARY_FILE;
140 			}
141 			break;
142 		case 0x1F:
143 			if (bytes[1] == (byte)0x8B) {
144 				return "application/x-gzip";
145 			} else if (bytes[1] == (byte)0x9D) {
146 				// followed by 0x90
147 				return "application/x-compress";
148 			}
149 			break;
150 		case '%':
151 			if (bytes[1] == 'P' && bytes[2] == 'D' && bytes[3] == 'F' && bytes[4] == '-') {
152 				return "application/pdf";
153 			} else if (bytes[1] == '!' && bytes[2] == 'P' && bytes[3] == 'S' && bytes[4] == '-') {
154 				// application/postscript, Type 1 font.
155 				return "application/postscript";
156 			}
157 			break;
158 		case 'B':
159 			if (bytes[1] == 'Z' && bytes[2] == 'h') {
160 				return "application/x-bzip2";
161 			}
162 			break;
163 		case 'F':
164 			if (bytes[1] == 'W' && bytes[2] == 'S') {
165 				// followed by <04> or <05>. Also 'CWS<07>?
166 				return "application/x-shockwave-flash";
167 			} else if (bytes[1] == 'L' && bytes[2] == 'V' && bytes[3] == 0x01) {
168 				return "video/x-flv";
169 			}
170 			break;
171 		case 'G':
172 			if (bytes[1] == 'I' && bytes[2] == 'F' && bytes[3] == '8') {
173 				// GIF87a or GIF89a
174 				return "image/gif";
175 			}
176 			break;
177 		case 'M':
178 			if (bytes[1] == 'Z') {
179 				// MS-DOS Executable (MZ <00-FF><00-01>)
180 				if (bytes[3] == 0x00 || bytes[3] == 0x01) {
181 					return "application/x-dosexec";
182 				}
183 			} else if (bytes[1] == 'S' && bytes[2] == 'C' && bytes[3] == 'F') {
184 				// MS cab file
185 				return "application/vnd.ms-cab-compressed";
186 			} else if (bytes[1] == 'T' && bytes[2] == 'h' && bytes[3] == 'd') {
187 				// MIDI
188 				return "audio/midi"; // or "application/x-midi"
189 			}
190 			break;
191 		case 'P':
192 			if (bytes[1] == 'K' && bytes[2] == 0x03 && bytes[3] == 0x04) {
193 				return "application/zip";
194 			} else if (bytes[1] == 'E' && bytes[2] == 0x00 && bytes[3] == 0x00 && bytes[4] == 'M' && bytes[5] == 'S') {
195 				// Windows PE
196 				return BINARY_FILE;
197 			}
198 			break;
199 		case 'm':
200 			if (bytes[1] == 'o' && bytes[2] == 'o' && bytes[3] == 'v') {
201 				return "video/quicktime";
202 			} else if (bytes[1] == 'd' && bytes[2] == 'a' && bytes[3] == 't') {
203 				return "video/quicktime";
204 			}
205 			break;
206 		case '{':
207 			if (bytes[1] == '\\' && bytes[2] == 'r' && bytes[3] == 't' && bytes[4] == 'f' && bytes[5] == '1') {
208 				return "application/rtf";
209 			}
210 			break;
211 		}
212 
213 		if (bytes[2] == '-' && bytes[3] == 'l' && bytes[4] == 'h' && bytes[5] == '5' && bytes[6] == '-') {
214 			// LZH archive
215 			return BINARY_FILE;
216 		}
217 		// Other formats we may want to add
218 		// "RIFF" <?><?><?><?> "WAVEfmt " - "audio/wav"
219 		// <DB><A5><2D><00> - commonly with suffix .doc. files command doesn't know this format.
220 		// <C5><D0><D3><C6><1E><00><00><00> - some EPS has a binary header starting with this, before "%!PS-"
221 		return null;
222 	}

224 	/**
225 	 * Read first {@code sniffLength} bytes of {@code resource}'s payload,
226 	 * decoding {@code Content-Encoding} if any. Reset {@code resource}'s
227 	 * read position back to zero.
228 	 * @param resource Resource to load bytes from
229 	 * @return bytes, zero-padded if payload is shorter.
230 	 * @throws IOException
231 	 */
232 	protected byte[] peekContent(Resource resource) throws IOException {
233 		byte[] bbuffer = new byte[Math.max(sniffLength, MINIMUM_SNIFF_BUFFER_SIZE)];
234 		String encoding = resource.getHeader("content-encoding");
235 		// use larger readlimit, because gzip-ed data can be larger than the original
236 		// at low compression level.
237 		resource.mark(sniffLength + 100);
238 		if (encoding != null) {
239 			Resource decoded = DecodingResource.forEncoding(encoding, resource);
240 			if (decoded != null) {
241 				decoded.read(bbuffer, 0, sniffLength);
242 				resource.reset();
243 				return bbuffer;
244 			}
245 		}
246 		resource.read(bbuffer, 0, sniffLength);
247 		resource.reset();
248 		return bbuffer;
249 	}

251 	@Override
252 	public String sniff(Resource resource) {
253 		// This sniffer only works with HTTP response record.
254 		// TODO: check record type.
255 
256 		byte[] bbuffer;
257 		try {
258 			bbuffer = peekContent(resource);
259 		} catch (IOException ex) {
260 			// Caveat: IOException from reset() (i.e. mark got invalidated) will have major
261 			// consequences. Should we re-throw some runtime exception?
262 			logger.warning("error reading " + sniffLength + " from resource: " + ex.getMessage());
263 			return null;
264 		}
265 
266 		// Spare decoding and regexp-matching for clearly-binary files.
267 		// Most mimetype detector libraries are overkill since we don't
268 		// need to know the details (ex. bitrate of MP3, PDF version).
269 		// So we use our own detector.
270 		String ctype = detectBinaryTypes(bbuffer);
271 		if (ctype != null) {
272 			// Very like be a binary file.
273 			return ctype;
274 		}
275 
276 		// Try decoding as text and look for signature patterns.
277 		// It doesn't need to be too complex, since want to differentiate
278 		// only a handful of text types: HTML, JAVASCRIPT, JSON and CSS.
279 		String encoding;
280 		try {
281 			encoding = charsetDetector.getCharset(resource, null);
282 		} catch (IOException ex1) {
283 			// IO error at this stage means we won't be able to sniff
284 			// content type either.
285 			// TODO: log
286 			return null;
287 		}
288 		//System.err.println("detected encoding: " + encoding);
289 		String text;
290 		try {
291 			text = new String(bbuffer, encoding);
292 		} catch (UnsupportedEncodingException ex) {
293 			// likely to happen, already checked by CharsetDetector.
294 			return null;
295 		}
296 		// strip off BOM - all variants are decoded into \ufeff.
297 		if (text.length() > 0 && text.charAt(0) == '\ufeff') {
298 			text = text.substring(1);
299 		}
300 
301 		ctype = detectHTML(text);
302 		if (ctype != null) return ctype;
303 
304 		ctype = detectJavaScript(text);
305 		if (ctype != null) return ctype;
306 
307 		ctype = detectCSS(text);
308 		if (ctype != null) return ctype;
309 
310 		return null;
311 	}

313 	private static final Pattern RE_XML_PROLOGUE = Pattern
314 			.compile("\\s*<\\?xml\\s+version=\"[.\\d]+\"\\s+.*\\?>");
315 	private static final Pattern RE_HTML_ELEMENTS = Pattern
316 			.compile("(?i)\\s*<(HTML|HEAD|STYLE|SCRIPT|META|BODY)(\\s|>)");
317 	private static final Pattern RE_DOCTYPE_HTML = Pattern
318 			.compile("(?i)\\s*<!DOCTYPE\\s+HTML");
319 	private static final Pattern RE_SGML_COMMENT = Pattern
320 			.compile("(?s)\\s*<!--.*?-->");
321 	private static final Pattern RE_END_TAG = Pattern
322 			.compile("(?i)</[a-z][a-z0-9]*>");
323 
324 	protected String detectHTML(String text) {
325 		int pos = 0;
326 		{
327 			Matcher m = RE_XML_PROLOGUE.matcher(text);
328 			if (m.lookingAt()) {
329 				pos = m.end();
330 			}
331 		}
332 		{
333 			Matcher m = RE_SGML_COMMENT.matcher(text);
334 			m.region(pos, text.length());
335 			while (m.lookingAt()) {
336 				m.region(pos = m.end(), text.length());
337 			}
338 		}
339 		{
340 			Matcher m = RE_DOCTYPE_HTML.matcher(text);
341 			m.region(pos, text.length());
342 			if (m.lookingAt())
343 				return "text/html";
344 		}
345 		{
346 			Matcher m = RE_HTML_ELEMENTS.matcher(text);
347 			m.region(pos, text.length());
348 			if (m.lookingAt())
349 				return "text/html";
350 		}
351 		{
352 			Matcher m = RE_END_TAG.matcher(text);
353 			m.region(pos, text.length());
354 			if (m.find())
355 				return "text/html";
356 		}
357 		return null;
358 	}

360 	private static final Pattern RE_JS_VAR = Pattern
361 		.compile("(?m)^var\\s+[_a-zA-Z$][_a-zA-Z$0-9]+");
362 	private static final Pattern RE_JS_FUNCTION = Pattern
363 		.compile("(?s)function(?:\\s+[a-zA-Z0-9_$]+\\s*)?\\(");
364 	private static final Pattern RE_JSON_HEAD = Pattern
365 		.compile("\\s*\\{\\s*\"");
366 
367 	protected String detectJavaScript(String text) {
368 		{
369 			Matcher m = RE_JS_VAR.matcher(text);
370 			if (m.find())
371 				return "text/javascript";
372 		}
373 		{
374 			Matcher m = RE_JS_FUNCTION.matcher(text);
375 			if (m.find())
376 				return "text/javascript";
377 		}
378 		{
379 			Matcher m = RE_JSON_HEAD.matcher(text);
380 			if (m.lookingAt()) {
381 				// TODO: if resource has content-type "text/javascript", just
382 				// use it.
383 				return "application/json";
384 			}
385 		}
386 		return null;
387 	}

389 	private static final Pattern RE_CSS_COMMENT = Pattern.compile("\\s*/\\*.*?\\*/");
390 	private static final Pattern RE_CSS_AT_RULE = Pattern.compile("\\s*@(import|media|document|charset|font-face|keyframes|namespace|supports)\\s+");
391 	private static final String RE_CSS_SIMPLE_SELECTOR = "(?:(?:[-a-z0-9]+|\\*)(?:[.#:][-_a-z0-9]+|\\[.+?\\])*|(?:[.#:][-_a-z0-9]+|\\[.+?\\])+)";
392 	private static final Pattern RE_CSS_RULESET_START = Pattern
393 		.compile("(?i)\\s*" + RE_CSS_SIMPLE_SELECTOR + "(?:[\\s,+>]+" + RE_CSS_SIMPLE_SELECTOR + ")*\\s*\\{");
394 	private static final Pattern RE_CSS_DECLARATION = Pattern
395 		.compile("(?i)\\s*[-a-z]+\\s*:\\s*[^;}]+[;}]");
396 
397 	protected String detectCSS(String text) {
398 		// CSS (they are rarely returned with mimetype "unk" or "text/html")
399 		int pos = 0;
400 		Matcher cm = RE_CSS_COMMENT.matcher(text);
401 		{
402 			cm.region(pos, text.length());
403 			while (cm.lookingAt()) {
404 				cm.region(pos = cm.end(), text.length());
405 			}
406 		}
407 		{
408 			Matcher m = RE_CSS_AT_RULE.matcher(text);
409 			m.region(pos,  text.length());
410 			if (m.lookingAt()) {
411 				return "text/css";
412 			}
413 		}
414 		{
415 			Matcher m = RE_CSS_RULESET_START.matcher(text);
416 			m.region(pos,  text.length());
417 			if (m.lookingAt()) {
418 				cm.region(pos = m.end(), text.length());
419 				while (cm.lookingAt()) {
420 					cm.region(pos = cm.end(), text.length());
421 				}
422 				Matcher sm = RE_CSS_DECLARATION.matcher(text);
423 				sm.region(pos, text.length());
424 				if (sm.lookingAt()) {
425 					return "text/css";
426 				}
427 			}
428 		}
429 		return null;
430 	}
