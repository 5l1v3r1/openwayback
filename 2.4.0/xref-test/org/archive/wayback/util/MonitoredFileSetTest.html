1   package org.archive.wayback.util;

3   import java.io.File;
4   import java.io.FileOutputStream;
5   import java.io.IOException;
6   import java.util.ArrayList;
7   
8   import junit.framework.TestCase;
9   
10  public class MonitoredFileSetTest extends TestCase {
11  
12  	public void testIsChanged() throws IOException, InterruptedException {
13  		File f1 = new File(System.getProperty("java.io.tmpdir"), "file-set-1.tmp");
14  		File f2 = new File(System.getProperty("java.io.tmpdir"), "file-set-2.tmp");
15  		writeFile(f1,"one");
16  		writeFile(f2,"two");
17  		ArrayList<String> l = new ArrayList<String>();
18  		l.add(f1.getAbsolutePath());
19  		l.add(f2.getAbsolutePath());
20  		
21  		MonitoredFileSet fs = new MonitoredFileSet(l);
22  		MonitoredFileSet.FileState s1 = fs.getFileState();
23  		MonitoredFileSet.FileState s2 = fs.getFileState();
24  		assertFalse(fs.isChanged(s1));
25  		assertFalse(fs.isChanged(s2));
26  		Thread.sleep(1001);
27  		writeFile(f2,"two2");
28  		MonitoredFileSet.FileState s3 = fs.getFileState();
29  		assertTrue(fs.isChanged(s2));
30  		assertTrue(s3.isChanged(s2));
31  		Thread.sleep(1001);
32  		assertTrue(fs.isChanged(s2));
33  		assertFalse(fs.isChanged(s3));
34  	}
35  	private void writeFile(File f, String stuff) throws IOException {
36  		if(f.exists()) {
37  			f.delete();
38  		}
39  		FileOutputStream fos = new FileOutputStream(f,false);
40  		fos.write(stuff.getBytes());
41  		fos.close();
42  	}
43  }
