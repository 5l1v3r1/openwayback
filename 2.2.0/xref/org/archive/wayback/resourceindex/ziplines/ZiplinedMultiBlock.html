3   import java.io.BufferedReader;
4   import java.io.ByteArrayInputStream;
5   import java.io.IOException;
6   import java.io.InputStream;
7   import java.io.InputStreamReader;
8   import java.io.SequenceInputStream;
9   import java.util.ArrayList;
10  
11  import org.archive.util.zip.OpenJDK7GZIPInputStream;
12  
13  public class ZiplinedMultiBlock extends ZiplinedBlock {
14  	
15  	//long offset = 0;
16  	String partName;
17  	//String[] locations;
18  	BlockLoader loader;
19  	ArrayList<Integer> blockSizes;
20  	//int totalSize = 0;
21  	
22  	public ZiplinedMultiBlock(long offset, BlockLoader loader, String partName, String[] locations)
23  	{
24  		super(locations, offset, 0);
25  		setLoader(loader);
26  		this.partName = partName;
27  		blockSizes = new ArrayList<Integer>();
28  	}
29  	
30  	public boolean isSameBlock(long nextOffset, String nextPartName)
31  	{
32  		return ((offset + super.count) == nextOffset) && partName.equals(nextPartName);
33  	}
34  	
35  	public void addOffset(int size)
36  	{
37  		blockSizes.add(size);
38  		super.count += size;
39  	}
40  	
41  	@Override
42  	public String toString()
43  	{
44  		return "Multiblock from " + partName + " of " + blockSizes.size() + " segments (" + offset + "," + super.count + ")";
45  	}
46  	
47  	@Override
48  	public BufferedReader readBlock()
49  	throws IOException {
50  		byte bytes[] = attemptBlockLoad(super.loader);
51  		
52  		if(bytes == null) {
53  			throw new IOException("Unable to load block(s)!");
54  		}
55  		
56  		int count = 0;
57  		InputStream currChain = null;
58  		
59  		for (int blockSize : blockSizes) {
60  			InputStream nextStream = new ByteArrayInputStream(bytes, count, blockSize);
61  			nextStream = new OpenJDK7GZIPInputStream(nextStream);
62  			
63  			if (currChain == null) {
64  				currChain = nextStream;
65  			} else {
66  				currChain = new SequenceInputStream(currChain, nextStream);
67  			}
68  			
69  			count += blockSize;
70  		}
71  		
72  		return new BufferedReader(new InputStreamReader(currChain));
73  	}
74  }
