41      /**
42       * create a test {@link Resource} from {@link WARCRecordInfo} {@code recinfo}.
43       * <p>Override this method to run tests on different implementations of
44       * Resource.</p>
45       * @param recinfo
46       * @return Resource
47       * @see JWATResourceTest
48       */
49      protected Resource createResource(WARCRecordInfo recinfo) throws Exception {
50      	TestWARCReader ar = new TestWARCReader(recinfo);
51      	WARCRecord rec = ar.get(0);
52      	WarcResource res = new WarcResource(rec, ar);
53      	return res;
54      }

56      /**
57       * plain HTTP response (without any transfer/content-encoding)
58       * @throws Exception
59       */
60      public void testPlainHttpRecord() throws Exception {
61          String payload = "hogehogehogehogehoge";
62          WARCRecordInfo recinfo = TestWARCRecordInfo.createHttpResponse(payload);
63          Resource res = createResource(recinfo);
64          res.parseHeaders();
65          
66          assertEquals("statusCode", 200, res.getStatusCode());
67          assertEquals("content-type", "text/plain", res.getHeader("Content-Type"));
68          byte[] buf = new byte[payload.getBytes().length + 1];
69          int n = res.read(buf);
70          assertEquals("content length", buf.length - 1, n);
71          
72          res.close();
73      }

150     /**
151      * metadata record with render-able content like site screenshot image.
152      * 
153      * HTTP status is assumed to be 200, and Content-Type of WARC header
154      * becomes Content-Type of replay response.
155      * @throws Exception
156      */
157     public void testMetadataRecord() throws Exception {
158         // 1-dot transparent GIF found everywhere if you wonder :-)
159         final byte[] block = new byte[] {
160                 71, 73, 70, 56, 57, 97, 1, 0, 1, 0, -128, 0, 0, -64, -64, -64,
161                 0, 0, 0, 33, -7, 4, 1, 0, 0, 0, 0, 44, 0, 0, 0, 0,
162                 1, 0, 1, 0, 0, 2, 2, 68, 1, 0, 59, 13, 10, 13, 10
163         };
164         final String ct = "image/gif";
165         WARCRecordInfo recinfo = new TestWARCRecordInfo(block);
166         recinfo.setType(WARCRecordType.metadata);
167         recinfo.setMimetype(ct);
168         Resource res = createResource(recinfo);
169         // must not fail
170         res.parseHeaders();
171         
172         // should return assumed 200
173         assertEquals("statusCode", 200, res.getStatusCode());
174         // content-type is what's specified in WARC header.
175         assertEquals("content-type", ct, res.getHeader("Content-Type"));
176         // must have Date header, in HTTP Date format.
177         String date = res.getHeader("Date");
178         assertNotNull("has date header", date);
179         new SimpleDateFormat("EEE, dd MMM yyyy HH:mm:ss z", Locale.ENGLISH).parse(date);
180         
181         // block as content
182         byte[] buf = new byte[block.length + 1];
183         int n = res.read(buf);
184         assertEquals("content length", block.length, n);
185         for (int i = 0; i < block.length; i++) {
186             assertEquals("byte " + i, block[i], buf[i]);
187         }
188         
189         res.close();
190     }

224      * old revisit record, which has zero-length block (no HTTP response
225      * line, no HTTP headers).
226      * 
227      * in this case, {@link WarcResource#getStatusCode()} should not fail, but
228      * either return special value or throw an appropriate exception signifying
229      * there's no HTTP status line recorded in this resource, and thus ReplayRenderer
230      * should fallback on using payloadResource for the info instead.
231      * {@link WarcResource#getHttpHeaders()} must not return null, but should
232      * return empty Map object, so that {@link CharsetDetector} can return null
233      * without failing.
234      * 
235      * for the better, this fallback may be encapsulated in
236      * virtual Resource combining httpHeaderResource and payloadResource.
237      * 
238      * related issue: https://webarchive.jira.com/browse/ACC-126
239      * @throws Exception
240      * @see TextReplayRenderer
241      * @see StandardCharsetDetector#getCharset(org.archive.wayback.core.Resource, org.archive.wayback.core.Resource, org.archive.wayback.core.WaybackRequest)
242      */
243     public void testOldRevisitRecord() throws Exception {
244         final String ct = "text/html";
245         WARCRecordInfo recinfo = TestWARCRecordInfo.createRevisitHttpResponse(ct, 1345, false);
246         Resource res = createResource(recinfo);
247         res.parseHeaders();
248         
249         // should either return special value or throw appropriate exception (TBD)
250         int scode = res.getStatusCode();
251         assertEquals("status code", 0, scode);
252         
253         Map<String, String> headers = res.getHttpHeaders();
254         //assertNotNull("headers", headers);
255         assertNull("headers", headers);
256         
257         res.close();
258     }

281     /**
282      * resource record, typically used for archiving ftp fetches.
283      * @throws Exception
284      */
285     public void testResourceRecord() throws Exception {
286         final String ct = "text/plain";
287         final byte[] block = "blahblahblah\n".getBytes();
288         WARCRecordInfo recinfo = new TestWARCRecordInfo(block);
289         recinfo.setType(WARCRecordType.resource);
290         recinfo.setUrl("ftp://ftp.example.com/afile.txt");
291         recinfo.setMimetype(ct);
292         Resource res = createResource(recinfo);
293         res.parseHeaders();
294                 
295         int scode = res.getStatusCode();
296         assertEquals("statusCode", 200, scode);
297 
298         Map<String, String> headers = res.getHttpHeaders();
299         assertNotNull("headers", headers);
300         
301         assertEquals("content-type", ct, res.getHeader("Content-Type"));
302 
303         // must have Date header, in HTTP Date format.
304         String date = res.getHeader("Date");
305         assertNotNull("has date header", date);
306         new SimpleDateFormat("EEE, dd MMM yyyy HH:mm:ss z", Locale.ENGLISH).parse(date);
307         
308         res.close();
309     }
