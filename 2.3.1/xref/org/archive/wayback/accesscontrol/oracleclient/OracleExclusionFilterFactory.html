20  package org.archive.wayback.accesscontrol.oracleclient;
21  
22  import org.archive.wayback.accesscontrol.ExclusionFilterFactory;
23  import org.archive.wayback.resourceindex.filters.ExclusionFilter;
24  
25  /**
26   * ExclusionFilterFactory for {@link OracleExclusionFilter}.
27   * @author brad
28   * @see OracleExclusionFilter
29   */
30  public class OracleExclusionFilterFactory implements ExclusionFilterFactory {
31  
32  	private String oracleUrl = null;
33  	private String accessGroup = null;
34  	private String proxyHostPort = null;
35  
36  	public ExclusionFilter get() {
37  		OracleExclusionFilter filter = new OracleExclusionFilter(oracleUrl,
38  			accessGroup, proxyHostPort);
39  		return filter;
40  	}
41  
42  	public void shutdown() {
43  		// no-op... yet..
44  	}
45  
46  	/**
47  	 * @return String URL where Oracle HTTP server is located
48  	 */
49  	public String getOracleUrl() {
50  		return oracleUrl;
51  	}
52  
53  	/**
54  	 * @param oracleUrl String URL where Oracle HTTP server is located
55  	 */
56  	public void setOracleUrl(String oracleUrl) {
57  		this.oracleUrl = oracleUrl;
58  	}
59  
60  	/**
61  	 * @return String group to use with requests to the Oracle
62  	 */
63  	public String getAccessGroup() {
64  		return accessGroup;
65  	}
66  
67  	/**
68  	 * @param accessGroup String group to use with requests to the Oracle
69  	 */
70  	public void setAccessGroup(String accessGroup) {
71  		this.accessGroup = accessGroup;
72  	}
73  
74  	/**
75  	 * @return the proxyHostPort
76  	 */
77  	public String getProxyHostPort() {
78  		return proxyHostPort;
79  	}
80  
81  	/**
82  	 * @param proxyHostPort the proxyHostPort to set, ex. "localhost:3128"
83  	 */
84  	public void setProxyHostPort(String proxyHostPort) {
85  		this.proxyHostPort = proxyHostPort;
86  	}
87  
88  }
