3   import java.net.InetAddress;
4   import java.net.UnknownHostException;
5   import java.util.concurrent.Callable;
6   import java.util.concurrent.ExecutionException;
7   import java.util.concurrent.ExecutorService;
8   import java.util.concurrent.Executors;
9   import java.util.concurrent.Future;
10  import java.util.concurrent.TimeUnit;
11  import java.util.concurrent.TimeoutException;
12  import java.util.logging.Logger;
13  
14  class TimedDNSLookup
15  {
16  	private final static Logger LOGGER = 
17  		Logger.getLogger(TimedDNSLookup.class.getName());
18  	
19  	private ExecutorService executor;
20  	private int dnsTimeoutMS;
21  	
22  	public ExecutorService getExecutor() {
23  		return executor;
24  	}
25  
26  	public int getDnsTimeoutMS() {
27  		return dnsTimeoutMS;
28  	}
29  
30  	public TimedDNSLookup(int poolSize, int dnsTimeoutMS) {		
31  		executor = Executors.newFixedThreadPool(poolSize);
32  		this.dnsTimeoutMS = dnsTimeoutMS;
33  	}
34  
35  	public InetAddress[] resolveHostname(String host)
36  			throws UnknownHostException {
37  		
38  		Future<InetAddress[]> future = executor.submit(new InetLookup(host));
39  		
40  		try {
41  			return future.get(dnsTimeoutMS, TimeUnit.MILLISECONDS);
42  			
43  		} catch (ExecutionException e) {
44  			throw (UnknownHostException)e.getCause();
45  		} catch (InterruptedException e) {
46  			future.cancel(true);
47  			LOGGER.warning("DNS INTERRUPTED: " + host);
48  			throw new UnknownHostException(host);
49  		} catch (TimeoutException e) {
50  			future.cancel(true);
51  			LOGGER.warning("DNS TIMEOUT: " + host);
52  			throw new UnknownHostException(host);
53  		}
54  	}
55  	
56  	private class InetLookup implements Callable<InetAddress[]>
57  	{
58  		String host;
59  		
60  		private InetLookup(String host)
61  		{
62  			this.host = host;
63  		}
64  		
65  		@Override
66  		public InetAddress[] call() throws Exception {
67  			return InetAddress.getAllByName(host);
68  		}
69  	}
70  }
