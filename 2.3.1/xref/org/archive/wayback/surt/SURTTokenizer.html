20  package org.archive.wayback.surt;
21  
22  import org.apache.commons.httpclient.URIException;
23  import org.archive.url.UsableURI;
24  import org.archive.url.UsableURIFactory;
25  import org.archive.util.ArchiveUtils;
26  import org.archive.util.SURT;
27  
28  /**
29   * provides iterative Url reduction for prefix matching to find ever coarser
30   * grained URL-specific configuration. Assumes that a prefix binary search is
31   * being attempted for each returned value. First value is the entire SURT
32   * url String, with TAB appended. Second removes CGI ARGs. Then each subsequent
33   * path segment ('/' separated) is removed. Then the login:password, if present
34   * is removed. Then the port, if not :80 or omitted on the initial URL. Then
35   * each subsequent authority segment(. separated) is removed.
36   * 
37   * the nextSearch() method will return null, finally, when no broader searches
38   * can be attempted on the URL.
39   *
40   * @author brad
41   * @version $Date$, $Revision$
42   */
43  public class SURTTokenizer {
44  
45  	private final static String EXACT_SUFFIX = "\t";
46  	private String remainder;
47  	private boolean triedExact;
48  	private boolean triedFull;
49  	private boolean choppedArgs;
50  	private boolean choppedPath;
51  	private boolean choppedLogin;
52  	private boolean choppedPort;
53  	
54  	/**
55  	 * constructor
56  	 * 
57  	 * @param url String URL
58  	 * @throws URIException 
59  	 */
60  	public SURTTokenizer(final String url, boolean isSurt) throws URIException {
61  		if(url.startsWith("(") || isSurt) {
62  			remainder = url;
63  		} else {
64  			remainder = getKey(url,false);
65  		}
66  	}
67  	/**
68  	 * update internal state and return the next smaller search string
69  	 * for the url
70  	 * 
71  	 * @return string to lookup for prefix match for relevant information.
72  	 */
73  	public String nextSearch() {
74  		if(!triedExact) {
75  			triedExact = true;
76  			return remainder + EXACT_SUFFIX;
77  		}
78  		if(!triedFull) {
79  			triedFull = true;
80  			if(remainder.endsWith(")/")) {
81  				choppedPath = true;
82  			}
83  			return remainder;
84  		}
85  		if(!choppedArgs) {
86  			choppedArgs = true;
87  			int argStart = remainder.indexOf('?');
88  			if(argStart != -1) {
89  				remainder = remainder.substring(0,argStart);
90  				return remainder;
91  			}
92  		}
93  		// we have already returned remainder as-is, so we have slightly
94  		// special handling here to make sure we continue to make progress:
95  		// (com,foo,www,)/         => (com,foo,www,
96  		// (com,foo,www,)/bar      => (com,foo,www,)/
97  		// (com,foo,www,)/bar/     => (com,foo,www,)/bar
98  		// (com,foo,www,)/bar/foo  => (com,foo,www,)/bar
99  		// (com,foo,www,)/bar/foo/ => (com,foo,www,)/bar/foo
100 		if(!choppedPath) {
101 			int lastSlash = remainder.lastIndexOf('/');
102 			if(lastSlash != -1) {
103 				if(lastSlash == (remainder.length()-1)) {
104 					if(remainder.endsWith(")/")) {
105 						String tmp = remainder;
106 						remainder = remainder.substring(0,lastSlash-1);
107 						choppedPath = true;
108 						return tmp;
109 					} else {
110 						remainder = remainder.substring(0,lastSlash);
111 						return remainder;
112 					}
113 				}
114 				if((lastSlash > 0) && remainder.charAt(lastSlash-1) == ')') {
115 					String tmp = remainder.substring(0,lastSlash+1);
116 					remainder = remainder.substring(0,lastSlash-1);
117 					return tmp;
118 				} else {
119 					remainder = remainder.substring(0,lastSlash);
120 					return remainder;
121 				}
122 			}
123 			choppedPath = true;
124 		}
125 		if(!choppedLogin) {
126 			choppedLogin = true;
127 			int lastAt = remainder.lastIndexOf('@');
128 			if(lastAt != -1) {
129 				String tmp = remainder;
130 				remainder = remainder.substring(0,lastAt);
131 				return tmp;
132 			}
133 		}
134 		if(!choppedPort) {
135 			choppedPort = true;
136 			int lastColon = remainder.lastIndexOf(':');
137 			if(lastColon != -1) {
138 				return remainder;
139 			}
140 		}
141 		// now just remove ','s
142 		int lastComma = remainder.lastIndexOf(',');
143 		if(lastComma == -1) {
144 			return null;
145 		}
146 		remainder = remainder.substring(0,lastComma);
147 		return remainder;
148 	}
149 	
150 	/**
151 	 * @param url
152 	 * @return String SURT which will match exactly argument url
153 	 * @throws URIException
154 	 */
155 	public static String exactKey(String url) throws URIException {
156 		return getKey(url,false);
157 	}
158 
159 	/**
160 	 * @param url
161 	 * @return String SURT which will match urls prefixed with the argument url
162 	 * @throws URIException
163 	 */
164 	public static String prefixKey(String url) throws URIException {
165 		return getKey(url,true);
166 	}
167 	
168 	private static String getKey(String url, boolean prefix)
169 	throws URIException {
170 
171 		String key = ArchiveUtils.addImpliedHttpIfNecessary(url);
172 		UsableURI uuri = UsableURIFactory.getInstance(key);
173 		key = uuri.getScheme() + "://" + uuri.getAuthority() + 
174 			uuri.getEscapedPathQuery();
175 
176 		key = SURT.fromURI(key);
177 		
178 		int hashPos = key.indexOf('#');
179 		if(hashPos != -1) {
180 			key = key.substring(0,hashPos);
181 		}
182 		
183 		if(key.startsWith("http://")) {
184 			key = key.substring(7);
185 		}
186 		if(prefix) {
187 			if(key.endsWith(",)/")) {
188 				key = key.substring(0,key.length()-3);
189 			}
190 		}
191 		return key;
192 	}
193 }
