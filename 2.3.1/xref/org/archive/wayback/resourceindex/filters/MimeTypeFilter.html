20  package org.archive.wayback.resourceindex.filters;
21  
22  import java.util.HashMap;
23  
24  import org.archive.wayback.core.CaptureSearchResult;
25  import org.archive.wayback.util.ObjectFilter;
26  
27  /**
28   * SearchResultFilter which includes only records matching one or more supplied
29   * Mime-Types. All comparision is case-insensitive.
30   *
31   * @author brad
32   * @version $Date$, $Revision$
33   */
34  public class MimeTypeFilter implements ObjectFilter<CaptureSearchResult> {
35  	private HashMap<String,Integer> validMimes = null;
36  	private boolean includeIfContains = true; 
37  	
38  	/**
39  	 * @param mime String which is valid match for mime-type field
40  	 */
41  	public void addMime(final String mime) {
42  		if(validMimes == null) {
43  			validMimes = new HashMap<String, Integer>();
44  		}
45  		validMimes.put(mime.toLowerCase(),null);
46  	}
47  
48  	/* (non-Javadoc)
49  	 * @see org.archive.wayback.util.ObjectFilter#filterObject(java.lang.Object)
50  	 */
51  	public int filterObject(CaptureSearchResult r) {
52  		String mime = r.getMimeType().toLowerCase();
53  		return validMimes.containsKey(mime) == includeIfContains ? 
54  				FILTER_INCLUDE : FILTER_EXCLUDE;
55  	}
56  
57  	/**
58  	 * @return the includeIfContains
59  	 */
60  	public boolean isIncludeIfContains() {
61  		return includeIfContains;
62  	}
63  
64  	/**
65  	 * @param includeIfContains the includeIfContains to set
66  	 */
67  	public void setIncludeIfContains(boolean includeIfContains) {
68  		this.includeIfContains = includeIfContains;
69  	}
70  }
