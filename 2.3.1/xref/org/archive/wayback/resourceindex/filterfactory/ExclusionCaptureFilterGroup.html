21  
22  import java.util.List;
23  
24  import org.archive.wayback.UrlCanonicalizer;
25  import org.archive.wayback.core.CaptureSearchResult;
26  import org.archive.wayback.core.SearchResults;
27  import org.archive.wayback.core.WaybackRequest;
28  import org.archive.wayback.exception.AccessControlException;
29  import org.archive.wayback.exception.AdministrativeAccessControlException;
30  import org.archive.wayback.exception.ResourceNotInArchiveException;
31  import org.archive.wayback.exception.RobotAccessControlException;
32  import org.archive.wayback.exception.RobotNotAvailableException;
33  import org.archive.wayback.exception.RobotTimedOutAccessControlException;
34  import org.archive.wayback.resourceindex.filters.CounterFilter;
35  import org.archive.wayback.resourceindex.filters.ExclusionFilter;
36  import org.archive.wayback.util.ObjectFilter;
37  import org.archive.wayback.util.ObjectFilterChain;
38  
39  public class ExclusionCaptureFilterGroup implements CaptureFilterGroup {
40  
41  	private ObjectFilterChain<CaptureSearchResult> chain = null;
42  	private CounterFilter preCounter = null;
43  	private CounterFilter postCounter = null;
44  	String requestUrl = null;
45  	private boolean sawRobots = false;
46  	private boolean passedRobots = false;
47  	private boolean robotTimedOut = false;
48  	private boolean liveWebGone = false;
49  	private boolean sawAdministrative = false;
50  	private boolean passedAdministrative = false;
51  	private UrlCanonicalizer canonicalizer = null;
52  	
53  	public ExclusionCaptureFilterGroup(WaybackRequest request, UrlCanonicalizer canonicalizer) {
54  		
55  		this.canonicalizer = canonicalizer;
56  		
57  		// checks an exclusion service for every matching record
58  		ExclusionFilter exclusion = request.getExclusionFilter();
59  		chain = new ObjectFilterChain<CaptureSearchResult>();
60  		if(exclusion != null) {
61  			exclusion.setFilterGroup(this);
62  //			preCounter = new CounterFilter();
63  //			// count how many results got to the ExclusionFilter:
64  //			chain.addFilter(preCounter);
65  			chain.addFilter(exclusion);
66  			// count how many results got past the ExclusionFilter:
67  			requestUrl = request.getRequestUrl();
68  		}
69  //		postCounter = new CounterFilter();
70  //		chain.addFilter(postCounter);
71  	}
72  	
73  	public UrlCanonicalizer getCaptureFilterGroupCanonicalizer()
74  	{
75  		return canonicalizer;
76  	}
77  	
78  	public List<ObjectFilter<CaptureSearchResult>> getFilters() {
79  		return chain.getFilters();
80  	}
81  
82  	public void annotateResults(SearchResults results)
83  			throws AccessControlException, ResourceNotInArchiveException,
84  			RobotNotAvailableException {
85  		if(getRobotTimedOut()) {
86  			throw new RobotTimedOutAccessControlException("Unable to check" +
87  					" robots.txt for " + requestUrl);
88  		}
89  		if(getLiveWebGone()) {
90  			throw new RobotNotAvailableException("The URL " + requestUrl +
91  			" is blocked by the sites robots.txt file");
92  		}
93  		if(isSawRobots() && !isPassedRobots()) {
94  			throw new RobotAccessControlException("The URL " + requestUrl +
95  					" is blocked by the sites robots.txt file");
96  		}
97  		if(isSawAdministrative() && !isPassedAdministrative()) {
98  			throw new AdministrativeAccessControlException(requestUrl +
99  					"  is not available in the Wayback Machine.");
100 		}
101 	}
102 
103 	public void setPassedRobots() {
104 		passedRobots = true;
105 	}
106 	public void setSawRobots() {
107 		sawRobots = true;
108 	}
109 	
110 	public void setPassedAdministrative(boolean passed) {
111 		passedAdministrative = passed;
112 	}
113 
114 	public void setPassedAdministrative() {
115 		passedAdministrative = true;
116 	}
117 	public void setSawAdministrative() {
118 		sawAdministrative = true;
119 	}
120 
121 	public void setRobotTimedOut() {
122 		robotTimedOut = true;
123 	}
124 	public boolean getRobotTimedOut() {
125 		return robotTimedOut;
126 	}
127 
128 	public void setLiveWebGone() {
129 		liveWebGone = true;
130 	}
131 	public boolean getLiveWebGone() {
132 		return liveWebGone;
133 	}
134 
135 	public boolean isSawRobots() {
136 		return sawRobots;
137 	}
138 
139 	public boolean isPassedRobots() {
140 		return passedRobots;
141 	}
142 
143 	public boolean isSawAdministrative() {
144 		return sawAdministrative;
145 	}
146 
147 	public boolean isPassedAdministrative() {
148 		return passedAdministrative;
149 	}
150 }
