53  	 * X canonized url in other href tages
54  	 * Y canonized url in other src tags
55  	 * Z canonized url found in script
56  	 * a original url **
57  	 * b date **
58  	 * c old style checksum *
59  	 * d uncompressed dat file offset
60  	 * e IP **
61  	 * f frame *

63  	 * h original host

86  	protected CDXField[] fields = null;
87  	protected char delimiter = ' ';
88  	protected String delimiterS = null;
89  
90  	public static String CDX_MAGIC = " CDX";
91  	
92  	public static char URL_KEY = 'A';
93  	public static char TIMESTAMP = 'b';
94  	public static char ORIGINAL_URL = 'a';
95  	public static char MIME_TYPE = 'm';
96  	public static char HTTP_CODE = 's';
97  	public static char DIGEST = 'k';
98  	public static char REDIRECT = 'r';
99  	public static char ROBOT_FLAGS = 'M';
100 	public static char COMPRESSED_OFFSET = 'V';
101 	public static char COMPRESSED_LENGTH = 'n';
102 	public static char FILE = 'g';
103 	
104 	/**
105 	 * Construct a CDXFormat reader/writer based on the specification argument
106 	 * @param cdxSpec
107 	 * @throws CDXFormatException
108 	 */
109 	public CDXFormat(String cdxSpec) throws CDXFormatException {
110 		if(!cdxSpec.startsWith(CDX_MAGIC)) {
111 			throw new CDXFormatException("Spec '" + cdxSpec 
112 					+ "' does not start with '" + CDX_MAGIC + "'");
113 		}
114 		delimiter = cdxSpec.charAt(CDX_MAGIC.length());
115 		String fieldsString = cdxSpec.substring(CDX_MAGIC.length()+1);
116 		int fieldCount = (fieldsString.length() + 1) / 2;
117 		if(fieldsString.length() != (fieldCount * 2) - 1) {
118 			throw new CDXFormatException("Extra char after spec '"
119 					+ cdxSpec + "'");
120 		}
121 		fields = new CDXField[fieldCount];
122 		for(int i = 0; i < fieldCount; i++) {
123 			char f = fieldsString.charAt(i * 2);
124 			if(i < fieldCount - 1) {
125 				char d = fieldsString.charAt((i*2)+1);
126 				if(d != delimiter) {
127 					throw new CDXFormatException("Non-delimiter char in '" 
128 							+ fieldsString + "'");
129 				}
130 			}
131 			fields[i] = getField(f);
132 		}
133 		delimiterS = new String(""+delimiter);
134 	}
135 	
136 	protected CDXField getField(char fieldChar) throws CDXFormatException {
137 		CDXField field = null;
138 		switch (fieldChar) {
139 			case 'A':  field = new URLKeyCDXField(); break;
140 			// backvards compat with Alexa tools:
141 			case 'N':  field = new URLKeyCDXField(); break;
142     		case 'b':  field = new TimestampCDXField(); break;
143     		case 'a':  field = new OriginalURLCDXField(); break;
144     		case 'm':  field = new MIMETypeCDXField(); break;
145     		case 's':  field = new HTTPCodeCDXField(); break;
146     		case 'k':  field = new DigestCDXField(); break;
147     		case 'r':  field = new RedirectURLCDXField(); break;
148     		case 'M':  field = new RobotFlagsCDXField(); break;
149     		case 'V':  field = new StartOffsetCDXField(); break;
150     		// Experimental..
151     		case 'S':  field = new CompressedLengthCDXField(); break;
152     		case 'g':  field = new FilenameCDXField(); break;
153 		}
154 		if(field == null) {
155 			throw new CDXFormatException("Unknown field '"+fieldChar+"'");
156 		}
157 		return field;
158 	}
159 	
160 	/**
161 	 * @param line
162 	 * @return CaptureSearchResult containing data from the 'line' argument
163 	 *         parsed according the the specification for this CDXFormat
164 	 * @throws CDXFormatException
165 	 */
166 	public CaptureSearchResult parseResult(String line) 
167 	throws CDXFormatException {
168 		CaptureSearchResult result = new CaptureSearchResult();
169 		String[] parts = line.split(delimiterS);
170 
171 		if(parts.length != fields.length) {
172 			throw new CDXFormatException("Wrong number of fields");
173 		}
174 		for(int i = 0; i < fields.length; i++) {
175 			fields[i].apply(parts[i], result);
176 		}
177 		return result;
178 	}
179 
180 	/**
181 	 * @param result
182 	 * @return String representation of the data in 'result' formatted according
183 	 *         to the specification for this CDXFormat
184 	 */
185 	public String serializeResult(CaptureSearchResult result) {
186 		StringBuilder sb = new StringBuilder(100);
187 		for(int i = 0; i < fields.length; i++) {
188 			String value = fields[i].serialize(result);
189 			if((value == null) || (value.length() == 0)) {
190 				sb.append(CDXField.DEFAULT_VALUE);
191 			} else {
192 				sb.append(value);
193 			}
194 			if(i < fields.length - 1) {
195 				sb.append(delimiter);
196 			}
197 		}
198 		return sb.toString();
199 	}
