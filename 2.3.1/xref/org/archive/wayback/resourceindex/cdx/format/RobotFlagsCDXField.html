20  package org.archive.wayback.resourceindex.cdx.format;
21  
22  import org.archive.wayback.core.CaptureSearchResult;
23  
24  public class RobotFlagsCDXField implements CDXField {
25  
26  	public void apply(String field, CaptureSearchResult result)
27  			throws CDXFormatException {
28  		result.setRobotFlags(field);
29  	}
30  
31  	public String serialize(CaptureSearchResult result) {
32  		String r = result.getRobotFlags();
33  		return r == null ? DEFAULT_VALUE : r;
34  	}
35  }
