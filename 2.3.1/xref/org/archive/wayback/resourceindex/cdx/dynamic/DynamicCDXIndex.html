40  /**
41   * A CompositeSearchResultSource that autmatically manages it's list of sources
42   * based on 3 configuration files, and a background thread:
43   * Config 1: Mapping of ranges to hosts responsible for that range
44   *               this class is aware of the local host name, so uses this file
45   *               to determin which range(s) should be local
46   *
47   * Config 2: Mapping of ranges to one or more MD5s that compose that range
48   *               when all of these MD5s have been copied local, this index 
49   *               becomes active, and each request uses a composite of these
50   *               local files
51   *
52   * Config 3: Mapping of MD5s to locations from which they can be retrieved
53   *               when a file that should be local is missing, these locations
54   *               will be used to retrieve a copy of that file
55   * 
56   * Background Thread: compares current set of files to the various 
57   *               configurations files, gets files local that need to be and
58   *               updates the composite set searched when the correct set of
59   *               MD5s are localized. 
60   *
61   * The Thread maintains the state of the sychronization with the desired file
62   * set:
63   *   UNKNOWN: If the desired state is unknown
64   *   SYNCHING: If the local state does not match the desired state
65   *   SYNCHED: If the local stat matches the desired state
66   * 
67   * This class forwards all method requests to the superclass, if the state is
68   * SYNCHED, otherwise throws a ResourceIndexNotAvailableException.
69   * 
70   * @author brad
71   * @version $Date$, $Revision$
72   */
73  public class DynamicCDXIndex extends CompositeSearchResultSource {
74  
75  	private static final Logger LOGGER =
76          Logger.getLogger(DynamicCDXIndex.class.getName());
77  
78  	protected static int STATE_UNKNOWN = 0;
79  	protected static int STATE_SYNCHING = 1;
80  	protected static int STATE_SYNCHED = 2;
81  	
82  	private int state = STATE_UNKNOWN;
83  	private File dataDir;
84  	private static Thread syncherThread = null;
85  	protected static String MD5_PATTERN = "^[0-9a-f_.-]{32}$";
86      protected static final Pattern MD5_REGEX = Pattern.compile(MD5_PATTERN);
87  	
88  	/**
89  	 * @param nodeNames 
90  	 * @param interval
91  	 * @param dataDir 
92  	 * @param rangeFile
93  	 * @param definitionFile
94  	 * @param md5File
95  	 */
96  	public DynamicCDXIndex(Object nodeNames[], int interval, File dataDir,
97  			RangeAssignmentFile rangeFile, CDXDefinitionFile definitionFile,
98  			MD5LocationFile md5File) {
99  		super();
100 		this.dataDir = dataDir;
101 		startUpThread(nodeNames,interval,rangeFile,definitionFile,md5File);
102 	}
103 	
104 	protected Object[] getLocalMD5s() {
105 		return dataDir.list(new md5FilenameFilter());
106 	}
107 
108 	protected File dataFileForMD5(String md5) {
109 		return new File(dataDir,md5);
110 	}
111 	
112 	protected void setCDXFiles(Object md5s[]) {
113 		sources.clear();
114 		for(int i = 0; i< md5s.length; i++) {
115 			File cdx = dataFileForMD5((String) md5s[i]);
116 			CDXIndex index = new CDXIndex();
117 			index.setPath(cdx.getAbsolutePath());
118 			addSource(index);
119 		}
120 	}
121 	
122 	private synchronized void startUpThread(Object nodeNames[], int interval,
123 			RangeAssignmentFile rangeFile, CDXDefinitionFile definitionFile,
124 			MD5LocationFile md5File) {
125 		if (syncherThread != null) {
126 			return;
127 		}
128 		syncherThread = new DynamicCDXSyncherThread(this, nodeNames, interval,
129 				rangeFile, definitionFile,md5File);
130 		syncherThread.start();
131 		
132 	}
133 
134 	protected synchronized void setState(int newState) {
135 		state = newState;
136 	}
137 
138 	protected synchronized int getState() {
139 		return state;
140 	}

168 	private class md5FilenameFilter implements FilenameFilter {
169 
170 		/* (non-Javadoc)
171 		 * @see java.io.FilenameFilter#accept(java.io.File, java.lang.String)
172 		 */
173 		public boolean accept(File dir, String name) {
174 			return name.matches(MD5_PATTERN);
175 		}
176 	}

178 	private class DynamicCDXSyncherThread extends Thread {
179 		private RangeAssignmentFile rangeFile = null;
180 		private CDXDefinitionFile definitionFile = null;
181 		private MD5LocationFile md5File = null;
182 		private DynamicCDXIndex index = null;
183 
184 		private Object nodeNames[];
185 		private int runInterval;
186 		private FileDownloader downloader;
187 
188 		/**
189 		 * @param index
190 		 * @param nodeNames 
191 		 * @param runInterval
192 		 * @param rangeFile
193 		 * @param definitionFile
194 		 * @param md5File
195 		 */
196 		public DynamicCDXSyncherThread(DynamicCDXIndex index, Object nodeNames[],
197 				int runInterval, RangeAssignmentFile rangeFile, 
198 				CDXDefinitionFile definitionFile, MD5LocationFile md5File ) {
199 			
200 			super("DynamicCDXSyncherThread");
201 			super.setDaemon(true);
202 			this.index = index;
203 			this.nodeNames = nodeNames;
204 			this.runInterval = runInterval;
205 			this.rangeFile = rangeFile;
206 			this.definitionFile = definitionFile;
207 			this.md5File = md5File;
208 			this.downloader = new FileDownloader();
209 			this.downloader.setDigest(true);
210 			LOGGER.info("DynamicCDXSyncherThread is alive.");
211 		}
212 		
213 		private Object[] getDesiredMD5s() throws IOException {
214 			ArrayList<String> allRanges = new ArrayList<String>();
215 			for(int i = 0; i < nodeNames.length; i++) {
216 				Object ranges[] = rangeFile.getRangesForNode((String) nodeNames[i]);
217 				for(int j=0; j<ranges.length; j++) {
218 					allRanges.add((String)ranges[j]);
219 				}
220 			}
221 			Object ranges[] = allRanges.toArray();
222 			ArrayList<String> md5sNeeded = new ArrayList<String>();
223 			for(int i = 0; i < ranges.length; i++) {
224 				Object rangeMD5s[] = definitionFile.getMD5sForRange((String) ranges[i]);
225 				for(int j=0; j < rangeMD5s.length; j++) {
226 					md5sNeeded.add((String)rangeMD5s[j]);
227 				}
228 			}
229 			return md5sNeeded.toArray();
230 		}
231 
232 		private Object[] getCurrentMD5s() {
233 			return index.getLocalMD5s();
234 		}
235 
236 		private void removeFiles(Object toBeRemoved[]) throws IOException {
237 			for(int i=0; i< toBeRemoved.length; i++) {
238 				File toDelete = index.dataFileForMD5((String) toBeRemoved[i]);
239 				if(!toDelete.delete()) {
240 					throw new IOException("Failed to remove " +
241 							toDelete.getAbsolutePath());
242 				}
243 			}
244 		}
245 		
246 		private void downloadFiles(Object toBeDownloaded[]) throws IOException {
247 			for(int i=0; i< toBeDownloaded.length; i++) {
248 				String neededMD5 = (String) toBeDownloaded[i];
249 				File target = index.dataFileForMD5(neededMD5);
250 				File tmpTarget = new File(target.getAbsolutePath() + ".TMP");
251 				Object locs[] = md5File.getLocationsForMD5(neededMD5);
252 				boolean gotFile = false;
253 				for(int j=0; j< locs.length; j++) {
254 					String loc = (String) locs[j];
255 					URL u = new URL(loc);
256 					try {
257 						if(loc.endsWith(".gz")) {
258 							downloader.downloadGZ(u,tmpTarget);
259 						} else {
260 							downloader.download(u,tmpTarget);
261 						}
262 						String gotMD5 = downloader.getLastDigest();
263 						if(gotMD5.equals(neededMD5)) {
264 							gotFile = true;
265 							tmpTarget.renameTo(target);
266 							break;
267 						} else {
268 							tmpTarget.delete();
269 							LOGGER.warning("Bad file contents. Location(" + 
270 									loc +") should have MD5(" + neededMD5 +
271 									") but has MD5(" + gotMD5 +")");
272 						}
273 					} catch (IOException e) {
274 						e.printStackTrace();
275 					} catch (NoSuchAlgorithmException e) {
276 						e.printStackTrace();
277 					}
278 				}
279 				if(!gotFile) {
280 					throw new IOException("Unable to get MD5 " +
281 							neededMD5);
282 				}
283 			}			
284 		}
285 		
286 		public void run() {
287 			int sleepInterval = runInterval;
288 			boolean setInitial = false;
289 			while (true) {
290 				try {
291 					// get desired index state:
292 					Object desired[] = getDesiredMD5s();
293 					// get current index state:
294 					Object current[] = getCurrentMD5s();
295 
296 					// work to do?
297 					HashMap<String,Object> desiredMap = 
298 						new HashMap<String, Object>();
299 					ArrayList<String> extra = new ArrayList<String>();
300 					for(int i=0; i< desired.length; i++) {
301 						desiredMap.put((String)desired[i],null);
302 					}
303 					for(int i=0; i< current.length; i++) {
304 						if(desiredMap.containsKey(current[i])) {
305 							desiredMap.remove(current[i]);
306 						} else {
307 							extra.add((String)current[i]);
308 						}
309 					}
310 					Set<String> needed = desiredMap.keySet();
311 					if((needed.size() + extra.size()) > 0) {
312 						// whoops -- we're off:
313 						index.setState(DynamicCDXIndex.STATE_SYNCHING);
314 
315 						// first remove extras:
316 						removeFiles(extra.toArray());
317 						
318 						// now get needed:
319 						downloadFiles(needed.toArray());
320 
321 						index.setCDXFiles(desired);
322 						index.setState(DynamicCDXIndex.STATE_SYNCHED);
323 					} else if(!setInitial && current.length > 0) {
324 						index.setCDXFiles(desired);
325 						index.setState(DynamicCDXIndex.STATE_SYNCHED);
326 					}
327 					sleep(sleepInterval);
328 				} catch (InterruptedException e) {
329 					e.printStackTrace();
330 					return;
331 				} catch (IOException e) {
332 					// TODO Auto-generated catch block
333 					e.printStackTrace();
334 				}
335 			}
336 		}
337 	}
