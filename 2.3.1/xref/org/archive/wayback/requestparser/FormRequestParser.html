20  package org.archive.wayback.requestparser;
21  
22  import java.util.Iterator;
23  import java.util.Map;
24  import java.util.Set;
25  
26  import javax.servlet.http.HttpServletRequest;
27  
28  import org.archive.wayback.core.WaybackRequest;
29  import org.archive.wayback.exception.BetterRequestException;
30  import org.archive.wayback.util.Timestamp;
31  import org.archive.wayback.util.url.UrlOperations;
32  import org.archive.wayback.webapp.AccessPoint;
33  
34  /**
35   * RequestParser which attempts to extract data from an HTML form, that is, from
36   * HTTP GET request arguments
37   *
38   * @author brad
39   */
40  public class FormRequestParser extends WrappedRequestParser {
41  	/**
42  	 * @param wrapped the BaseRequestParser being wrapped
43  	 */
44  	public FormRequestParser(BaseRequestParser wrapped) {
45  		super(wrapped);
46  	}
47  
48  	/**
49  	 * CGI argument name for Submit button...
50  	 */
51  	private final static String SUBMIT_BUTTON = "Submit";
52  
53  	/*
54  	 * Stuff whatever GET/POST arguments are sent up into the returned
55  	 * WaybackRequest object, except the Submit button argument.
56  	 */
57  	public WaybackRequest parse(HttpServletRequest httpRequest, 
58  			AccessPoint accessPoint) throws BetterRequestException {
59  
60  		WaybackRequest wbRequest = null;
61  		@SuppressWarnings("unchecked")
62  		Map<String, String[]> queryMap = httpRequest.getParameterMap();
63  		if (queryMap.size() > 0) {
64  			wbRequest = new WaybackRequest();
65  
66  			String base = accessPoint.translateRequestPath(httpRequest);
67  			if (base.startsWith(REPLAY_BASE)) {
68  				wbRequest.setReplayRequest();
69  			} else if (base.startsWith(QUERY_BASE)) {
70  				wbRequest.setCaptureQueryRequest();
71  			} else if (base.startsWith(XQUERY_BASE)) {
72  				wbRequest.setCaptureQueryRequest();
73  				wbRequest.setXMLMode(true);
74  
75  			} else {
76  				return null;
77  			}
78  			wbRequest.setResultsPerPage(getMaxRecords());
79  			Set<String> keys = queryMap.keySet();
80  			Iterator<String> itr = keys.iterator();
81  			while (itr.hasNext()) {
82  				String key = itr.next();
83  				if (key.equals(SUBMIT_BUTTON)) {
84  					continue;
85  				}
86  				// just jam everything else in:
87  				String val = AccessPoint.getMapParam(queryMap, key);
88  				if (key.equals(WaybackRequest.REQUEST_URL)) {
89  					String scheme = UrlOperations.urlToScheme(val);
90  					if (scheme == null) {
91  						val = UrlOperations.HTTP_SCHEME + val;
92  					}
93  				}
94  				wbRequest.put(key, val);
95  			}
96  			String partialTS = wbRequest.getReplayTimestamp();
97  			if (partialTS != null) {
98  				if (wbRequest.getStartTimestamp() == null) {
99  					String startTS = Timestamp.parseBefore(partialTS)
100 							.getDateStr();
101 					wbRequest.setStartTimestamp(startTS);
102 				}
103 				if (wbRequest.getEndTimestamp() == null) {
104 					String endTS = Timestamp.parseAfter(partialTS).getDateStr();
105 					wbRequest.setEndTimestamp(endTS);
106 				}
107 			} else {
108 				if (wbRequest.getStartTimestamp() == null) {
109 					wbRequest.setStartTimestamp(getEarliestTimestamp());
110 				}
111 				if (wbRequest.getEndTimestamp() == null) {
112 					wbRequest.setEndTimestamp(getLatestTimestamp());
113 				}
114 			}
115 		}
116 		return wbRequest;
117 	}
118 }
