69  		Date first = results.getFirstResultDate();
70  		Date last = results.getLastResultDate();

81  		}
82  		// end
83  
84  		pw.print(makeLink(requestUrl, ORIGINAL));
85  		pw.println(",");
86  		// ludab nov 30 2012
87

89  		// TIMEMAP,APPLICATION_LINK_FORMAT));

91  			getTimemapDateUrl(ap, FORMAT_LINK, pagedate, requestUrl), "self",
92  			APPLICATION_LINK_FORMAT) + "; " + captureDate("from", first) + "; " + captureDate("until", last));
93  		// end
94  		pw.println(",");
95  		pw.print(makeLink(getTimegateUrl(ap, requestUrl), TIMEGATE));
96  		pw.println(",");
97  
98  		if (first.compareTo(last) == 0) {
99  			// special handling of single result:
100 			CaptureSearchResult result = results.getResults().get(0);
101 			pw.print(makeLink(ap, result.getOriginalUrl(), FIRST_LAST_MEMENTO,
102 				result));
103 		} else {
104 			List<CaptureSearchResult> lr = results.getResults();
105 			int count = lr.size();
106 			String rel;
107 			for (int i = 0; i < count; i++) {
108 				CaptureSearchResult result = lr.get(i);
109 				if (i == 0) {
110 					rel = FIRST_MEMENTO;
111 				} else if (i == count - 1) {
112 					pw.println(",");
113 					rel = LAST_MEMENTO;
114 				} else {
115 					pw.println(",");
116 					rel = MEMENTO;
117 				}
118 				pw.print(makeLink(ap, result.getOriginalUrl(), rel, result));
119 			}
120 		}
121 		// ludab nov 30 2012
122 		if (results.getMatchingCount() > results.getReturnedCount()) {
123 			int sec = last.getSeconds() + 1;
124 			last.setSeconds(sec);
125 			pw.println(",");
126 			pw.print(makeLink(
127 				getTimemapDateUrl(ap, FORMAT_LINK,
128 					ArchiveUtils.get14DigitDate(last) + "/", requestUrl),
129 				TIMEMAP, APPLICATION_LINK_FORMAT) +
130 				"; " + captureDate("from", last));
131 		}
132 		// end
133 
134 		pw.flush();
135 	}

138 	 * Add {@code Link} header value. Includes: {@code first}, {@code prev},
139 	 * {@code next}, {@code last}, {@code original} (if {@code includeOriginal}
140 	 * is {@code true}), and {@code timegate} (if {@code includeTimegateLink} is
141 	 * {@code true}).
142 	 * 
143 	 */
144 	public static String generateMementoLinkHeaders(
145 			CaptureSearchResults results, WaybackRequest wbr,
146 			boolean includeTimegateLink, boolean includeOriginalLink) {
147 		NotableResultExtractor nre = getNotableResults(results);
148 		CaptureSearchResult first = nre.getFirst();
149 		CaptureSearchResult prev = nre.getPrev();
150 		CaptureSearchResult closest = nre.getClosest();
151 		CaptureSearchResult next = nre.getNext();
152 		CaptureSearchResult last = nre.getLast();
153 		ArrayList<String> rels = new ArrayList<String>();
154 
155 		AccessPoint ap = wbr.getAccessPoint();
156 
157 		String requestUrl = wbr.getRequestUrl();
158 
159 		// add generics:
160 		// rels.add(makeLink(getTimebundleUrl(ap, requestUrl), TIMEBUNDLE));
161 		if (includeOriginalLink) {
162 			rels.add(makeLink(requestUrl, ORIGINAL));
163 		}
164 
165 		rels.add(makeLink(getTimemapUrl(ap, FORMAT_LINK, requestUrl), TIMEMAP,
166 			APPLICATION_LINK_FORMAT));
167 
168 		// Spec says not to include timegate link for timegate
169 		if (includeTimegateLink) {
170 			rels.add(makeLink(getTimegateUrl(ap, requestUrl), TIMEGATE));
171 		}
172 
173 		// add first/prev/next/last:
174 		if (first == last) {
175 			// only one capture.. are we sure we want the "actual" memento here?
176 			rels.add(makeLink(ap, requestUrl, FIRST_LAST_MEMENTO, first));
177 		} else {
178 			if (first == closest) {
179 				// no previous:
180 				rels.add(makeLink(ap, requestUrl, FIRST_MEMENTO, first));
181 				if (next == last) {
182 					rels.add(makeLink(ap, requestUrl, NEXT_LAST_MEMENTO, last));
183 				} else {
184 					rels.add(makeLink(ap, requestUrl, NEXT_MEMENTO, next));
185 					rels.add(makeLink(ap, requestUrl, LAST_MEMENTO, last));
186 				}
187 			} else if (last == closest) {
188 				// no next:
189 				rels.add(makeLink(ap, requestUrl, LAST_MEMENTO, last));
190 				if (prev == first) {
191 					rels.add(makeLink(ap, requestUrl, PREV_FIRST_MEMENTO, first));
192 				} else {
193 					rels.add(makeLink(ap, requestUrl, FIRST_MEMENTO, first));
194 					rels.add(makeLink(ap, requestUrl, PREV_MEMENTO, prev));
195 				}
196 			} else {
197 				// somewhere in the middle:
198 
199 				if (prev == first) {
200 					rels.add(makeLink(ap, requestUrl, PREV_FIRST_MEMENTO, first));
201 				} else {
202 					// add both prev and first:
203 					rels.add(makeLink(ap, requestUrl, FIRST_MEMENTO, first));
204 					rels.add(makeLink(ap, requestUrl, PREV_MEMENTO, prev));
205 				}
206 				// add "actual" memento:
207 				rels.add(makeLink(ap, requestUrl, MEMENTO, closest));
208 				if (next == last) {
209 					rels.add(makeLink(ap, requestUrl, NEXT_LAST_MEMENTO, last));
210 				} else {
211 					rels.add(makeLink(ap, requestUrl, NEXT_MEMENTO, next));
212 					rels.add(makeLink(ap, requestUrl, LAST_MEMENTO, last));
213 				}
214 			}
215 		}
216 		return StringFormatter.join(", ", rels.toArray(a));
217 	}

448 	private static NotableResultExtractor getNotableResults(
449 			CaptureSearchResults r) {
450 		// eventually, the NotableResultExtractor will be part of the standard
451 		// ResourceIndex.query() but for now, we'll just do an extra traversal
452 		// of the whole set of results:
453 
454 		Iterator<CaptureSearchResult> itr = r.iterator();
455 		Date want = r.getClosest().getCaptureDate();
456 		NotableResultExtractor nre = new NotableResultExtractor(want);
457 
458 		ObjectFilterIterator<CaptureSearchResult> ofi = new ObjectFilterIterator<CaptureSearchResult>(
459 			itr, nre);
460 		while (ofi.hasNext()) {
461 			ofi.next();
462 		}
463 		return nre;
464 	}
