20  package org.archive.wayback.domainprefix;
21  
22  import java.net.URI;
23  import java.net.URISyntaxException;
24  
25  import org.archive.wayback.ResultURIConverter;
26  
27  /**
28   *
29   *
30   * @author brad
31   * @version $Date$, $Revision$
32   */
33  public class DomainPrefixResultURIConverter implements ResultURIConverter {
34  
35  	private String hostPort = "localhost:8081";
36  	
37  	/* (non-Javadoc)
38  	 * @see org.archive.wayback.ResultURIConverter#makeReplayURI(java.lang.String, java.lang.String)
39  	 */
40  	public String makeReplayURI(String datespec, String url) {
41  		String replayURI = "";
42  		try {
43  			if(url.contains(hostPort)) {
44  				return url;
45  			}
46  			URI uri = new URI(url);
47  			StringBuilder sb = new StringBuilder(90);
48  			sb.append("http://");
49  			sb.append(datespec).append(".");
50  			sb.append(uri.getHost()).append(".");
51  			sb.append(hostPort);
52  			sb.append(uri.getPath());
53  			String query = uri.getQuery();
54  			if(query != null && query.length() > 0) {
55  				sb.append("?").append(query);
56  			}
57  			replayURI = sb.toString();
58  
59  		} catch (URISyntaxException e) {
60  			e.printStackTrace();
61  		}
62  		return replayURI;
63  	}
64  
65  	/**
66  	 * @return the hostPort
67  	 */
68  	public String getHostPort() {
69  		return hostPort;
70  	}
71  
72  	/**
73  	 * @param hostPort the hostPort to set
74  	 */
75  	public void setHostPort(String hostPort) {
76  		this.hostPort = hostPort;
77  	}
78  
79  }
