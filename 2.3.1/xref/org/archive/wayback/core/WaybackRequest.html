662 	private void setBoolean(String key, boolean value) {
663 		if(value) {
664 			put(key,REQUEST_YES);
665 		} else {
666 			remove(key);
667 		}
668 	}

670 	private int getInt(String key) {
671 		String value = get(key);
672 		if(value == null) {
673 			return -1;
674 		}
675 		return Integer.parseInt(value);
676 	}	
677 	private void setInt(String key, int value) {
678 		put(key,String.valueOf(value));
679 	}	
680 	private boolean getBoolean(String key) {
681 		String value = get(key);
682 		return(value != null && value.equals(REQUEST_YES));
683 	}

699 	private boolean isRequestType(String requestType) {
700 		String type = get(REQUEST_TYPE);
701 		if(type != null && type.equals(requestType)) {
702 			return true;
703 		}
704 		return false;
705 	}

747     /**
748      * Set the request URL.
749      * @param urlStr Request URL.
750      */
751 	public void setRequestUrl(String urlStr) {
752 
753 		// This looks a little confusing: We're trying to fixup an incoming
754 		// request URL that starts with: 
755 		//       "http:/www.archive.org"
756 		// so it becomes:
757 		//       "http://www.archive.org"
758 		// (note the missing second "/" in the first)
759 		// 
760 		// if that is not the case, then see if the incoming scheme
761 		// is known, adding an implied "http://" scheme if there doesn't appear
762 		// to be a scheme..
763 		// TODO: make the default "http://" configurable.
764 		if (!urlStr.startsWith(UrlOperations.HTTP_SCHEME)) {
765 	    	if(urlStr.startsWith("http:/")) {
766 	    		urlStr = UrlOperations.HTTP_SCHEME + urlStr.substring(6);
767 	    	} else {
768 	    		if(UrlOperations.urlToScheme(urlStr) == null) {
769 	    			urlStr = UrlOperations.HTTP_SCHEME + urlStr;
770 	    		}
771 	    	}
772 	    }
773 
774         try {
775             String decodedUrlStr = URLDecoder.decode(urlStr, "UTF-8");
776 
777             String idnEncodedHost = UsableURIFactory.getInstance(decodedUrlStr, "UTF-8").getHost();
778             
779             if (idnEncodedHost != null) {
780                 // If url is absolute, replace host with IDN-encoded host.
781                 String unicodeEncodedHost = URLEncoder.encode(IDNA.toUnicode(idnEncodedHost), "UTF-8");
782                 urlStr = urlStr.replace(unicodeEncodedHost, idnEncodedHost);
783             }
784         } catch (UnsupportedEncodingException ex) {
785             // Should never happen as UTF-8 is required to be present
786             throw new RuntimeException(ex);
787         } catch (URIException ex) {
788             throw new RuntimeException(ex);
789         }
790 
791         put(REQUEST_URL, urlStr);
792 	}

1103 	/**
1104 	 * extract REFERER, remote IP and authorization information from the
1105 	 * HttpServletRequest
1106 	 * 
1107 	 * @param httpRequest
1108 	 * @throws BadQueryException 
1109 	 */
1110 	public void extractHttpRequestInfo(HttpServletRequest httpRequest) {
1111 		
1112 		putUnlessNull(REQUEST_REFERER_URL, httpRequest.getHeader("REFERER"));
1113 		
1114 		String remoteAddr = httpRequest.getHeader("X-Forwarded-For");
1115 		if (remoteAddr == null) {
1116 			remoteAddr = httpRequest.getRemoteAddr();
1117 		}
1118 		putUnlessNull(REQUEST_REMOTE_ADDRESS, remoteAddr);
1119 		
1120 		// Check for AJAX
1121 		String x_req_with = httpRequest.getHeader("X-Requested-With");
1122 		if (x_req_with != null) {
1123 			if (x_req_with.equals("XMLHttpRequest")) {
1124 				this.setAjaxRequest(true);
1125 			}
1126 		} else if (this.getRefererUrl() != null && httpRequest.getParameter("ajaxpipe") != null) {
1127 			this.setAjaxRequest(true);
1128 		}
1129 		
1130 		if (isMementoEnabled()) {
1131 			// Check for Memento Accept-Datetime
1132 			String acceptDateTime = httpRequest.getHeader(MementoUtils.ACCEPT_DATETIME);
1133 			if (acceptDateTime != null) {
1134 				this.setMementoAcceptDatetime(true);
1135 			}
1136 		}
1137 		
1138 		putUnlessNull(REQUEST_WAYBACK_HOSTNAME, httpRequest.getLocalName());
1139 		putUnlessNull(REQUEST_AUTH_TYPE, httpRequest.getAuthType());
1140 		putUnlessNull(REQUEST_REMOTE_USER, httpRequest.getRemoteUser());
1141 		
1142 		putUnlessNull(REQUEST_AUTHORIZATION,
1143 				httpRequest.getHeader(REQUEST_AUTHORIZATION));
1144 		putUnlessNull(REQUEST_WAYBACK_PORT, 
1145 				String.valueOf(httpRequest.getLocalPort()));
1146 		putUnlessNull(REQUEST_WAYBACK_CONTEXT, httpRequest.getContextPath());
1147 
1148 		Locale l = null;
1149 		if (accessPoint != null) {
1150 			l = accessPoint.getLocale();
1151 		}
1152 		if (l == null) {
1153 			l = httpRequest.getLocale();
1154 		}
1155 		//setLocale(l);
1156 		this.locale = l;
1157 
1158 		putUnlessNull(REQUEST_LOCALE_LANG,l.getDisplayLanguage());
1159 
1160 		Cookie[] cookies = httpRequest.getCookies();
1161 		if(cookies != null) {
1162 			for(Cookie cookie : cookies) {
1163 				String name = cookie.getName();
1164 				String value = cookie.getValue();
1165 				String oldVal = get(name);
1166 				if(oldVal == null || oldVal.length() == 0) {
1167 					put(name,value);
1168 				}
1169 			}
1170 		}
1171 	}

1181 	/**
1182 	 * @param pageNum
1183 	 * @return String hex-encoded GET CGI arguments which will duplicate the
1184 	 *         same request, but for page 'pageNum' of the results
1185 	 */
1186 	public String getQueryArguments(int pageNum) {
1187 		int numPerPage = resultsPerPage;
1188 
1189 		try {
1190 			StringBuffer queryString = new StringBuffer("");
1191 			Iterator<String> itr = filters.keySet().iterator();
1192 			while(itr.hasNext()) {
1193 				String key = itr.next();
1194 				boolean isStandard = false;
1195 				for(int i=0; i<standardHeaders.length; i++) {
1196 					if(standardHeaders[i].equals(key)) {
1197 						isStandard = true;
1198 						break;
1199 					}
1200 				}
1201 				if(isStandard) continue;
1202 				String val = filters.get(key);
1203 				if(val == null) continue;
1204 				if (queryString.length() > 0) {
1205 					queryString.append(" ");
1206 				}
1207 				key = URLEncoder.encode(key, "UTF-8");
1208 				val = URLEncoder.encode(val, "UTF-8");
1209 				queryString.append(key + ":" + val);
1210 			}
1211 			String escapedQuery = queryString.toString();
1212 
1213 			escapedQuery = URLEncoder.encode(escapedQuery, "UTF-8");
1214 
1215 			return OpenSearchRequestParser.SEARCH_QUERY + "=" + escapedQuery + "&"
1216 				+ OpenSearchRequestParser.SEARCH_RESULTS + "=" + numPerPage + "&"
1217 				+ OpenSearchRequestParser.START_PAGE + "=" + pageNum;
1218 
1219 		} catch (UnsupportedEncodingException e) {
1220 			// oops.. what to do?
1221 			e.printStackTrace();
1222                         throw new RuntimeException(e);
1223 		}
1224 	}
