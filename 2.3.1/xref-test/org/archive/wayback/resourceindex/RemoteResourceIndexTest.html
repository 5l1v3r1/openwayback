30  import org.archive.wayback.core.CaptureSearchResult;
31  import org.archive.wayback.core.CaptureSearchResults;

37  import junit.framework.TestCase;

39  /**
40   * @author brad
41   *
42   */
43  public class RemoteResourceIndexTest extends TestCase {
44  	private DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
45  
46  	/**
47  	 * Test method for {@link org.archive.wayback.resourceindex.RemoteResourceIndex#documentToSearchResults(org.w3c.dom.Document, org.archive.wayback.util.ObjectFilter)}.
48  	 */
49  	public void testDocumentToSearchResults() {
50  		RemoteResourceIndex r = new RemoteResourceIndex();
51  		DocumentBuilder db;
52  		try {
53  			db = factory.newDocumentBuilder();
54  			
55  			String testXML = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>"
56  				+ "<wayback>"
57  				+ "<request>"
58  				+ "<startdate>20090101000000</startdate>"
59  				+ "<numreturned>1</numreturned>"
60  				+ "<type>urlquery</type>"
61  				+ "<enddate>20100810191351</enddate>"
62  				+ "<numresults>1</numresults>"
63  				+ "<firstreturned>0</firstreturned>"
64  				+ "<url>dixienet.org/</url>"
65  				+ "<resultsrequested>1000</resultsrequested>"
66  				+ "<resultstype>resultstypecapture</resultstype>"
67  				+ "</request>"
68  				+ "<results>"
69  				+ "<result>"
70  				+ "<compressedoffset>36717460</compressedoffset>"
71  				+ "<mimetype>text/html</mimetype>"
72  				+ "<file>LOC-TRANSITION-001-20090204213039-00004-crawling108.us.archive.org.warc.gz</file>"
73  				+ "<redirecturl>http://dixienet.org/New%20Site/index.shtml</redirecturl>"
74  				+ "<urlkey>dixienet.org/</urlkey>"
75  				+ "<digest>3I42H3S6NNFQ2MSVX7XZKYAYSCX5QBYJ</digest>"
76  				+ "<httpresponsecode>302</httpresponsecode>"
77  				+ "<url>http://dixienet.org/</url>"
78  				+ "<capturedate>20090204213443</capturedate>"
79  				+ "</result>"
80  				+ "</results>"
81  				+ "</wayback>";
82  			ByteArrayInputStream bais = new ByteArrayInputStream(testXML.getBytes());
83  			Document document = db.parse(bais);
84  			
85  			ObjectFilterChain<CaptureSearchResult> filter = new ObjectFilterChain<CaptureSearchResult>();
86  			filter.addFilter(new ObjectFilter<CaptureSearchResult>() {
87  				
88  				public int filterObject(CaptureSearchResult o) {
89  					return ObjectFilter.FILTER_INCLUDE;
90  				}
91  			});
92  			CaptureSearchResults sr = (CaptureSearchResults) r.documentToSearchResults(document, filter);
93  			assertEquals(1,sr.getResults().size());
94  			assertEquals("20090204213443",sr.getResults().get(0).getCaptureTimestamp());
95  		} catch (Exception e) {
96  			// TODO Auto-generated catch block
97  			e.printStackTrace();
98  			fail(e.getLocalizedMessage());
99  		}
100 	}
