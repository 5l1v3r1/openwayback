27  /**
28   * unit test for {@link ArchivalCSSReplayRenderer}.
29   * <p>Uses a mock for {@link ResultURIConverter}. ResultURIConverter
30   * must be test separately. ArchivalCSSReplayRenderer does not use
31   * BlockCSSStringTransformer, but relies on TextDocument for extracting
32   * URLs in CSS. So this unit test does not verify those
33   * <code>StringTransformer</code>s are working correctly.</p>
34   * <p>ArchivalCSSReplayRenderer is still a primary ReplayRenderer
35   * for CSS resources (vs. CSS embedded in HTML) for both archival-URL
36   * and proxy mode.</p>
37   *
38   * @author kenji
39   *
40   */
41  public class ArchivalUrlCSSReplayRendererTest extends TestCase {
42  
43      ResultURIConverter uriConverter;
44      HttpServletResponse response;
45      WaybackRequest wbRequest;
46      CaptureSearchResult result;
47      TestServletOutputStream servletOutput = new TestServletOutputStream();
48      
49      ArchivalUrlCSSReplayRenderer cut;
50          
51      /* (non-Javadoc)
52       * @see junit.framework.TestCase#setUp()
53       */
54      protected void setUp() throws Exception {
55          super.setUp();
56          RedirectRewritingHttpHeaderProcessor httpHeaderProcessor = new RedirectRewritingHttpHeaderProcessor();
57          httpHeaderProcessor.setPrefix("X-Archive-Orig-");
58          
59          cut = new ArchivalUrlCSSReplayRenderer(httpHeaderProcessor);
60          
61          uriConverter = EasyMock.createMock(ResultURIConverter.class);
62          
63          response = EasyMock.createMock(HttpServletResponse.class);
64          EasyMock.expect(response.getOutputStream()).andReturn(servletOutput);
65          
66          wbRequest = new WaybackRequest();
67          wbRequest.setFrameWrapperContext(false);
68          
69          result = new CaptureSearchResult();
70          result.setOriginalUrl("http://www.example.com/");
71          result.setCaptureTimestamp("20100101123456");
72      }
73      
74      public static Resource createTestHtmlResource(byte[] payloadBytes) throws IOException {
75          WARCRecordInfo recinfo = TestWARCRecordInfo.createCompressedHttpResponse("text/html", payloadBytes);
76          TestWARCReader ar = new TestWARCReader(recinfo);
77          WARCRecord rec = ar.get(0);
78          WarcResource resource = new WarcResource(rec, ar);
79          resource.parseHeaders();
80          return resource;
81      }
82      
83      /**
84       * test basic behavior with simple input.
85       * expectations:
86       * <ul>
87       * <li>reads <em>decoded (uncompressed)</em> contents from archive record.</li>
88       * <li>calls HttpServletResponse.setHeader() for Content-Type, Content-Length and 
89       *   {@link TextReplayRenderer#getGuessedCharsetHeader()}.</li>
90       * <li>calls HttpServletResponse.setCharsetEncoding() with value "utf-8"</li>
91       * </ul>
92       * @throws Exception
93       */
94      public void testBasicBehavior() throws Exception {
95          final String payload = 
96                  "@import \"style1.css\";\n" +
97          		"@import 'style2.css';\n" +
98                  "@import 'http://archive.org/common.css';\n" +
99          		"BODY {\n" +
100         		"  color: #fff;\n" +
101         		"  background: transparent url(bg.gif);\n" +
102         		"}\n";
103         final byte[] payloadBytes = payload.getBytes("UTF-8");
104         Resource payloadResource = createTestHtmlResource(payloadBytes);
105         
106         response.setStatus(200);
107         response.setCharacterEncoding("UTF-8");
108         response.setHeader(EasyMock.eq("Content-Length"), EasyMock.<String>notNull());
109         response.setHeader(cut.getGuessedCharsetHeader(), "UTF-8");
110         response.setHeader("Content-Type", "text/html");
111         response.setHeader(EasyMock.matches("X-Archive-Orig-.*"), EasyMock.<String>notNull());
112         EasyMock.expectLastCall().anyTimes();
113         
114         final String ts = result.getCaptureTimestamp();
115         final String url1 = result.getOriginalUrl() + "style1.css";
116         EasyMock.expect(uriConverter.makeReplayURI(ts, url1)).andReturn(
117                 "/web/" + ts + "/" + url1);
118         final String url2 = result.getOriginalUrl() + "style2.css";
119         EasyMock.expect(uriConverter.makeReplayURI(ts, url2)).andReturn(
120                 "/web/" + ts + "/" + url2);
121         final String url3 = result.getOriginalUrl() + "bg.gif";
122         EasyMock.expect(uriConverter.makeReplayURI(ts, url3)).andReturn(
123                 "/web/" + ts + "/" + url3);
124         EasyMock.expect(uriConverter.makeReplayURI(ts, "http://archive.org/common.css")).andReturn(
125                 "http://archive.org/common.css");
126         
127         EasyMock.replay(response, uriConverter);
128         
129         cut.renderResource(null, response, wbRequest, result, payloadResource, payloadResource, uriConverter, null);
130         
131         EasyMock.verify(response, uriConverter);
132         
133         final String expected = 
134                 "@import \"/web/20100101123456/http://www.example.com/style1.css\";\n" +
135                 "@import '/web/20100101123456/http://www.example.com/style2.css';\n" +
136                 "@import 'http://archive.org/common.css';\n" +
137                 "BODY {\n" +
138                 "  color: #fff;\n" +
139                 "  background: transparent url(/web/20100101123456/http://www.example.com/bg.gif);\n" +
140                 "}\n";
141         String out = servletOutput.getString();
142         Assert.assertTrue( out.startsWith( expected ) );
143     }
