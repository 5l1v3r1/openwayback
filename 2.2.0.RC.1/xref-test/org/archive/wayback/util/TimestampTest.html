35  public class TimestampTest extends TestCase {
36  	/**
37  	 * run several padding tests
38  	 */
39  	public void testPadDateStr() {
40  
41  		String curYear = String.valueOf(Calendar.getInstance(TimeZone.getTimeZone("GMT")).get(Calendar.YEAR));
42  		
43  		assertEquals("padStart '1'","19960101000000",Timestamp.padStartDateStr("1"));
44  		assertEquals("padEnd '1'","19991231235959",Timestamp.padEndDateStr("1"));
45  		assertEquals("padStart '2'","20000101000000",Timestamp.padStartDateStr("2"));
46  		assertEquals("padEnd",curYear + "1231235959",Timestamp.padEndDateStr("2"));
47  		assertEquals("padEnd",curYear + "1231235959",Timestamp.padEndDateStr("3"));
48  		assertEquals("padEnd","20061231235959",Timestamp.padEndDateStr("2006"));
49  		assertEquals("padEnd","20061231235959",Timestamp.padEndDateStr("200613"));
50  		assertEquals("padEnd","20071231235959",Timestamp.padEndDateStr("2007"));
51  		
52  
53  		// day of month stuff:
54  		assertEquals("padEnd","20060131235959",Timestamp.padEndDateStr("200601"));
55  		assertEquals("padEnd","20060228235959",Timestamp.padEndDateStr("200602"));
56  		assertEquals("padEnd","20060331235959",Timestamp.padEndDateStr("200603"));
57  		assertEquals("padEnd","20060430235959",Timestamp.padEndDateStr("200604"));
58  		assertEquals("padEnd","20060430235959",Timestamp.padEndDateStr("2006044"));
59  
60  		assertEquals("padEnd","20050228235959",Timestamp.padEndDateStr("200502"));
61  		assertEquals("padEnd","20040229235959",Timestamp.padEndDateStr("200402"));
62  		assertEquals("padEnd","20030228235959",Timestamp.padEndDateStr("200302"));
63  
64  		assertEquals("padEnd","19960229235959",Timestamp.padEndDateStr("199602"));
65  		assertEquals("padStart","19960201000000",Timestamp.padStartDateStr("199602"));
66  		
67  		assertEquals("padStart","19960101000000",Timestamp.padStartDateStr("19960"));
68  		assertEquals("padEnd","19960930235959",Timestamp.padEndDateStr("19960"));
69  		
70  		assertEquals("padStart","19961001000000",Timestamp.padStartDateStr("19961"));
71  		assertEquals("padEnd","19961231235959",Timestamp.padEndDateStr("19961"));
72  
73  		assertEquals("padStart","19961001000000",Timestamp.padStartDateStr("19962"));
74  		assertEquals("padEnd","19961231235959",Timestamp.padEndDateStr("19962"));
75  		
76  		assertEquals("padStart","19960101000050",Timestamp.padStartDateStr("19960101000060"));
77  		assertEquals("padEnd","19960101000050",Timestamp.padEndDateStr("19960101000060"));
78  		
79  
80  	}
