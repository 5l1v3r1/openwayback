7   import java.util.logging.Logger;

11  import org.archive.cdxserver.auth.AuthToken;

13  import org.archive.cdxserver.filter.CDXAccessFilter;

15  import org.archive.cdxserver.filter.FilenamePrefixFilter;
16  import org.archive.format.cdx.CDXLine;
17  import org.archive.wayback.accesscontrol.ExclusionFilterFactory;
18  import org.archive.wayback.accesscontrol.oracleclient.CustomPolicyOracleFilter;
19  import org.archive.wayback.resourceindex.filters.ExclusionFilter;

22  /**
23   * {@link AuthChecker} implementation that runs {@link ExclusionFilter} provided
24   * through {@link AccessPoint#getExclusionFactory()} as well as the filtering
25   * provided by {@link WaybackAPAuthChecker}.
26   * <p>
27   * This is primarily meant for running {@link CustomPolicyOracleFilter} in
28   * {@link CDXServer} rather than in {@link EmbeddedCDXServerIndex}, making
29   * {@link WaybackAPAuthChecker} obsolete. Current implementation is very
30   * specific to this usage.
31   * </p>
32   * Needs {@link APContextAuthToken} as token.</p>
33   */
34  public class AccessPointAuthChecker extends PrivTokenAuthChecker {
35  
36  	private static final Logger logger = Logger
37  		.getLogger(AccessPointAuthChecker.class.getName());
38  
39  	protected ExclusionFilterFactory fallbackExclusionFactory;
40  
41  	/**
42  	 * {@link ExclusionFilterFactory} used if token passed to
43  	 * {@link #createAccessFilter(AuthToken)} is not an instance of
44  	 * {@link APContextAuthToken} (CDX server query, for example).
45  	 * @param fallbackExclusionFactory
46  	 */
47  	public void setFallbackExclusionFactory(
48  			ExclusionFilterFactory fallbackExclusionFactory) {
49  		this.fallbackExclusionFactory = fallbackExclusionFactory;
50  	}
51  
52  	public ExclusionFilterFactory getFallbackExclusionFactory() {
53  		return fallbackExclusionFactory;
54  	}
55  
56  	/**
57  	 * {@link CDXFilter} on prefix of filename field.
58  	 * <p>
59  	 * It first accepts all captures that have any of {@code includePrefixes} as
60  	 * filename prefixes, then rejects captures that have any of
61  	 * {@code excludePrefixes}. If {@code includePrefixes} is {@code null} (as
62  	 * opposed to empty), all captures are accepted.
63  	 * </p>
64  	 */
65  	protected class CombinedFilenamePrefixFilter implements CDXFilter {
66  		FilenamePrefixFilter includeFilter;
67  		FilenamePrefixFilter excludeFilter;
68  
69  		public CombinedFilenamePrefixFilter(List<String> includePrefixes,
70  				List<String> excludePrefixes) {
71  			if (includePrefixes != null)
72  				this.includeFilter = new FilenamePrefixFilter(includePrefixes,
73  					false);
74  			if (excludePrefixes != null)
75  				this.excludeFilter = new FilenamePrefixFilter(excludePrefixes,
76  					true);
77  		}
78  
79  		@Override
80  		public boolean include(CDXLine line) {
81  			return (includeFilter == null || includeFilter.include(line)) &&
82  					(excludeFilter == null || excludeFilter.include(line));
83  		}
84  	}
85  
86  	@Override
87  	public CDXAccessFilter createAccessFilter(AuthToken token) {
88  		if (token instanceof APContextAuthToken) {
89  			AccessPoint ap = ((APContextAuthToken)token).getAccessPoint();
90  
91  			CDXFilter prefixFilter = new CombinedFilenamePrefixFilter(
92  				ap.getFileIncludePrefixes(), ap.getFileExcludePrefixes());
93  
94  			ExclusionFilter apFilter = null;
95  			try {
96  				apFilter = ap.createExclusionFilter();
97  			} catch (Exception ex) {
98  				// FIXME: createExclusionFilter throws a sub-class of
99  				// AccessControlException when it fails to initialize
100 				// the exclusion component. Unfortunately it cannot be
101 				// thrown from this method because AuthChecker interface
102 				// is part of cdx-server project which cannot depend on
103 				// wayback-core, where AccessControlException is defined.
104 				// Yet, we should not just ignore this error because
105 				// exclusion system failure can be highly risky.
106 				// For now, we throw AccessControlException wrapped in
107 				// RuntimeException, assuming caller is catching it.
108 				// (See EmbeddedCDXServerIndex#doQuery(WaybackRequest))
109 				// Probably we should define a new exception in cdx-server.
110 				throw new RuntimeException(ex);
111 			}
112 			return new AccessCheckFilter(token, apFilter, null, prefixFilter);
113 		} else {
114 			ExclusionFilter apFilter = null;
115 			if (fallbackExclusionFactory != null) {
116 				apFilter = fallbackExclusionFactory.get();
117 			}
118 			return new AccessCheckFilter(token, apFilter, null, null);
119 		}
120 	}
121 }
