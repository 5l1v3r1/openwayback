1   package org.archive.wayback.resourceindex.filters;

3   import java.util.LinkedHashMap;
4   import java.util.Map;
5   
6   import org.archive.wayback.core.CaptureSearchResult;
7   import org.archive.wayback.util.ObjectFilter;
8   
9   public class DuplicateHashFilter implements ObjectFilter<CaptureSearchResult> {
10  
11  	private int maxDupeHashes = 10;
12  	protected int maxTrackedHashes = 3;
13  	private int numCaptures = 0;
14  	private int minThreshold = 100;
15  	
16  	public class LRUHashCache extends LinkedHashMap<String, Integer>
17  	{
18  		private static final long serialVersionUID = 1L;
19  
20  		public boolean removeEldestEntry(Map.Entry<String, Integer> eldest) {
21  	    	 return (size() > maxTrackedHashes);
22  	    }
23  	}
24  	
25  	LRUHashCache cache = new LRUHashCache();
26  	
27  	@Override
28  	public int filterObject(CaptureSearchResult o) {
29  		String thisHash = o.getDigest();
30  		int result = FILTER_INCLUDE;
31  		
32  		// Only start filtering after minThreshold captures
33  		if (++numCaptures <= minThreshold) {
34  			return result;
35  		}
36  				
37  		Integer count = cache.remove(thisHash);
38  		
39  		if (count == null) {
40  			cache.put(thisHash, 1);
41  		} else {
42  			if (count >= maxDupeHashes) {
43  				result = FILTER_EXCLUDE;
44  				cache.put(thisHash, count);
45  			} else {
46  				cache.put(thisHash, count + 1);
47  			}
48  		}
49  		
50  		return result;
51  	}
52  
53  
54  }
