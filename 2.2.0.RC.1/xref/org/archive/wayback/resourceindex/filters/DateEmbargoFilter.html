1   package org.archive.wayback.resourceindex.filters;
2   
3   import java.util.Date;
4   
5   import org.archive.wayback.core.CaptureSearchResult;
6   import org.archive.wayback.util.ObjectFilter;
7   
8   public class DateEmbargoFilter implements ObjectFilter<CaptureSearchResult> {
9   	protected Date embargoDate = null;
10  	public DateEmbargoFilter(long minAge) {
11  		embargoDate = new Date(System.currentTimeMillis() - minAge);
12  	}
13  	public int filterObject(CaptureSearchResult o) {
14  		return o.getCaptureDate().compareTo(embargoDate) < 0 
15  			? FILTER_INCLUDE : FILTER_EXCLUDE;
16  	}
17  }
