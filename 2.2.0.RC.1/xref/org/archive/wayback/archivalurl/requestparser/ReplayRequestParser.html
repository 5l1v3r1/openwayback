22  import java.util.regex.Matcher;
23  import java.util.regex.Pattern;
24  
25  import org.archive.wayback.archivalurl.ArchivalUrl;
26  import org.archive.wayback.core.WaybackRequest;
27  import org.archive.wayback.requestparser.BaseRequestParser;
28  import org.archive.wayback.util.Timestamp;

30  /**
31   * RequestParser implementation that extracts request info from a Replay
32   * Archival Url path.
33   * 
34   * @author brad
35   */
36  public class ReplayRequestParser extends DateUrlPathRequestParser {
37  //	/**
38  //	 * Regex which parses Archival URL replay requests into:
39  //	 *      timestamp, flags, & url
40  //	 */
41  //	public final static Pattern WB_REQUEST_REGEX = Pattern
42  //			.compile("^(\\d{1,14})(([a-z]{2}[0-9]*_)*)/(.*)$");
43  	public final static Pattern TIMESTAMPCTX_REGEX = Pattern.compile("(\\d{1,14})((?:[a-z]{2}[0-9]*_)*)");
44  	/**
45  	 * @param wrapped BaseRequestParser which provides general configuration
46  	 */
47  	public ReplayRequestParser(BaseRequestParser wrapped) {
48  		super(wrapped);
49  	}
50  
51  	@Override
52  	protected WaybackRequest parseDateUrl(String dateStr, String urlStr) {
53  		Matcher matcher = TIMESTAMPCTX_REGEX.matcher(dateStr);
54  		if (!matcher.matches())
55  			return null;
56  
57  		dateStr = matcher.group(1);
58  		String flags = matcher.group(2);
59  		
60  		// The logic of the classic WM wrt timestamp bounding:
61  		// if 14-digits are specified, assume min-max range boundaries
62  		// if less than 14 are specified, assume min-max range boundaries
63  		// based upon amount given (2001 => 20010101... - 20011231...)
64  		// AND assume the user asked for the LATEST possible date
65  		// within that range...
66  
67  		String startDate = null;
68  		String endDate = null;
69  		if (dateStr.length() == 12) {
70  			// assume this is one of those old old alexa ARCs which has 
71  			// some 12-digit dates. Pad with "00";
72  			dateStr = dateStr.concat("00");
73  		}
74  		if (dateStr.length() == 14) {
75  			startDate = getEarliestTimestamp();
76  			endDate = getLatestTimestamp();
77  			if(endDate == null) {
78  				endDate = Timestamp.currentTimestamp().getDateStr();
79  			}
80  		} else {
81  			// classic behavior:
82  			startDate = Timestamp.parseBefore(dateStr).getDateStr();
83  			endDate = Timestamp.parseAfter(dateStr).getDateStr();
84  			dateStr = endDate;
85  
86  			// maybe "better" behavior:
87  //			startDate = getEarliestTimestamp();
88  //			endDate = getLatestTimestamp();
89  //			dateStr = Timestamp.parseAfter(dateStr).getDateStr();
90  		}
91  
92  		WaybackRequest wbRequest = WaybackRequest.createReplayRequest(urlStr, dateStr, startDate, endDate);
93  		ArchivalUrl.assignFlags(wbRequest, flags);
94  
95  		return wbRequest;
96  	}
