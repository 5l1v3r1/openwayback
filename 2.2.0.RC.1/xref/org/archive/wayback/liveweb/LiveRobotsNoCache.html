19  public class LiveRobotsNoCache extends ArcRemoteLiveWebCache {
20  
21  	protected int maxRobotsSize = 512000;
22  
23  	public int getMaxRobotsSize() {
24  		return maxRobotsSize;
25  	}
26  
27  	public void setMaxRobotsSize(int maxRobotsSize) {
28  		this.maxRobotsSize = maxRobotsSize;
29  	}
30  
31  	@Override
32      public Resource getCachedResource(URL url, long maxCacheMS,
33              boolean record) throws LiveDocumentNotAvailableException,
34              LiveWebCacheUnavailableException, LiveWebTimeoutException,
35              IOException {
36  		
37  		HttpClient http = super.getHttpClient();
38  		
39  		GetMethod method = new GetMethod(url.toString());
40  		method.setFollowRedirects(true);
41  		method.getParams().setCookiePolicy(CookiePolicy.IGNORE_COOKIES);
42  		
43  		InputStream in = null;
44  
45  		try {
46  			int responseStatus = http.executeMethod(method);
47  			
48  			if (responseStatus >= 400 || responseStatus < 200) {
49  				throw new LiveDocumentNotAvailableException("Invalid Status: " + responseStatus);
50  			}		
51  			
52  			in = ByteStreams.limit(method.getResponseBodyAsStream(), maxRobotsSize);
53  			return new RobotsTxtResource(IOUtils.toString(in));
54  			
55  		} catch (IOException io) {
56  			throw new LiveDocumentNotAvailableException(io.toString());
57  		} finally {
58  			if (in != null) {
59  				in.close();
60  			}
61  			
62  			method.abort();
63  			method.releaseConnection();
64  		}
65      }
66  
67  	@Override
68      public void shutdown() {
69  	    // TODO Auto-generated method stub
70      }
71  }
