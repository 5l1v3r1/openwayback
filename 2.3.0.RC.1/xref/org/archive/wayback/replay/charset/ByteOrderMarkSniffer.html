1   package org.archive.wayback.replay.charset;

3   import java.io.IOException;
4   
5   import org.archive.wayback.core.Resource;
6   
7   /**
8    * {@link EncodingSniffer} that peek the content for
9    * Byte Order Mark bytes.
10   * <p>This is the step 3 of character encoding sniffing
11   * prescribed by WHAT-NG.</p>
12   */
13  public class ByteOrderMarkSniffer extends BaseEncodingSniffer {
14  	public static final int MAX_BOM_LEN = 3;
15  	@Override
16  	public String sniff(Resource resource) {
17  		byte[] bbuffer = new byte[MAX_BOM_LEN];
18  		resource.mark(MAX_BOM_LEN);
19  		try {
20  			resource.read(bbuffer, 0, MAX_BOM_LEN);
21  			resource.reset();
22  		} catch (IOException ex) {
23  			return null;
24  		}
25  		if (bbuffer[0] == (byte)0xFE && bbuffer[1] == (byte)0xFF)
26  			return "UTF-16BE";
27  		if (bbuffer[0] == (byte)0xFF && bbuffer[1] == (byte)0xFE)
28  			return "UTF-16LE";
29  		if (bbuffer[0] == (byte)0xEF && bbuffer[1] == (byte)0xBB && bbuffer[2] == (byte)0xBF)
30  			return "UTF-8";
31  		return null;
32  	}
33  }
