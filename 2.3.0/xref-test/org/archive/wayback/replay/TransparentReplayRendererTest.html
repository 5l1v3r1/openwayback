163     public void testRenderResource_Redirect() throws Exception {
164         String location = "http://www.example.com/index.html";
165         WARCRecordInfo recinfo = new TestWARCRecordInfo(TestWARCRecordInfo.buildHttpRedirectResponseBlock(location));
166         TestWARCReader ar = new TestWARCReader(recinfo);
167         WARCRecord rec = ar.get(0);
168         Resource payloadResource = new WarcResource(rec, ar);
169         payloadResource.parseHeaders();
170         
171         final String originalUrl = "http://www.example.com/";
172         final String captureTimestamp = "20130101123456";
173         
174         result.setOriginalUrl(originalUrl);
175         result.setCaptureTimestamp(captureTimestamp);
176         
177         // makeReplayURI() is called through RedirectRewritingHttpHeaderProcessor.
178         // TODO: perhaps HttpheaderProcessor is the right class to make fixture?
179         EasyMock.expect(uriConverter.makeReplayURI(captureTimestamp, location))
180         .andReturn("/web/" + captureTimestamp + "/" + location);
181 
182         TestServletOutputStream servletOutput = new TestServletOutputStream();
183         response.setStatus(302);
184         EasyMock.expect(response.getOutputStream()).andReturn(servletOutput);
185         response.setHeader("Content-Type", "text/html");
186         response.setHeader(EasyMock.eq("Location"), EasyMock.matches("/web/" + captureTimestamp + "/" + location));
187         // RedirectRewritingHttpHeaderProcessor drops Content-Length.
188         // response.setHeader("Content-Length", "0");
189         response.setHeader(EasyMock.<String>notNull(), EasyMock.<String>notNull());
190         EasyMock.expectLastCall().anyTimes();
191         
192         EasyMock.replay(response, uriConverter);
193 
194         cut.renderResource(request, response, wbRequest, result,
195                 payloadResource, payloadResource, uriConverter, results);
196         
197         EasyMock.verify(response, uriConverter);
198 
199         byte[] content = servletOutput.getBytes();
200         assertEquals("payload length", 0, content.length);
201     }

203     /**
204      * test replay of capture with {@code Transfer-Encoding: chunked}.
205      *
206      * <p>TransparentReplayRenderer writes out chunk-decoded payload, because
207      * {@link WarcResource} always decodes chunked-entity. Point of this test
208      * is that response never have {@code Transfer-Encoding: chunked} header,
209      * even when initialized with {@link IdentityHttpHeaderProcessor}.
210      * so, this is not really a unit test for TransparentReplayRenderer, but
211      * a multi-component test placed here for convenience.</p>
212      * <p>This test does not use member object {@code cut}, in order to test
213      * with {@link IdentityHttpHeaderProcessor}.</p>
214      *
215      * @throws Exception
216      */
217     public void testRenderResource_Chunked() throws Exception {
218         final String ct = "text/xml";
219         final String payload = "<?xml version=\"1.0\"?>\n" +
220         		"<payload name=\"archive\">\n" +
221         		"  <inside/>\n" +
222         		"</payload>\n";
223         final byte[] recordBytes = TestWARCRecordInfo.buildHttpResponseBlock(
224         		"200 OK", ct, payload.getBytes("UTF-8"), true);
225         //System.out.println(new String(recordBytes, "UTF-8"));
226         WARCRecordInfo recinfo = new TestWARCRecordInfo(recordBytes);
227         TestWARCReader ar = new TestWARCReader(recinfo);
228         WARCRecord rec = ar.get(0);
229         Resource payloadResource = new WarcResource(rec, ar);
230         payloadResource.parseHeaders();
231         Resource headersResource = payloadResource;
232 
233         TestServletOutputStream servletOutput = new TestServletOutputStream();
234         // expectations
235         response.setStatus(200);
236         EasyMock.expect(response.getOutputStream()).andReturn(servletOutput);
237         // capture setHeader() call for "Transfer-Encoding"
238         Capture<String> transferEncodingCapture = new Capture<String>(CaptureType.FIRST);
239         response.setHeader(EasyMock.eq("Transfer-Encoding"), EasyMock.capture(transferEncodingCapture));
240         EasyMock.expectLastCall().anyTimes();
241         response.setHeader(EasyMock.<String>anyObject(), EasyMock.<String>anyObject());
242         EasyMock.expectLastCall().anyTimes();
243 
244         EasyMock.replay(response);
245 
246         // creating separate test object to use IdentityHttpHeaderProcessor
247         TransparentReplayRenderer cut2 = new TransparentReplayRenderer(new IdentityHttpHeaderProcessor());
248         cut2.renderResource(request, response, wbRequest, result,
249         		headersResource, payloadResource, uriConverter, results);
250 
251         EasyMock.verify(response);
252 
253         assertFalse("Transfer-Encoding header must not be set", transferEncodingCapture.hasCaptured());
254 
255         // content is the original gzip-compressed bytes for PAYLOAD_GIF.
256         String output = new String(servletOutput.getBytes(), "UTF-8");
257         assertEquals(payload, output);
258     }

424 	protected static byte[] buildPartialContentResponseBlock(String ctype,
425 			byte[] payloadBytes, long startPos, long instanceSize)
426 			throws IOException {
427 		assertTrue(startPos + payloadBytes.length <= instanceSize);
428 		final String CRLF = "\r\n";
429 		final String contentRange = String.format("bytes %d-%d/%d", startPos, startPos + payloadBytes.length - 1, instanceSize);
430 		ByteArrayOutputStream blockbuf = new ByteArrayOutputStream();
431 		Writer bw = new OutputStreamWriter(blockbuf);
432 		bw.write("HTTP/1.0 206 Partial Content" + CRLF);
433 		bw.write("Content-Type: " + ctype + CRLF);
434 		bw.write("Content-Range: " + contentRange + CRLF);
435 		bw.write("Content-Length: " + payloadBytes.length + CRLF);
436 		bw.write(CRLF);
437 		bw.close();
438 		blockbuf.write(payloadBytes);
439 		return blockbuf.toByteArray();
440     }
