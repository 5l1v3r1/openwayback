12  /**
13   * unit test for {@link BlockCSSStringTransformer}.
14   * 
15   * @author kenji
16   *
17   */
18  public class BlockCSSStringTransformerTest extends TestCase {
19  
20  	URL baseURL;
21  	RecordingReplayParseContext rc;
22  	BlockCSSStringTransformer st;
23  	
24  	/* (non-Javadoc)
25  	 * @see junit.framework.TestCase#setUp()
26  	 */
27  	protected void setUp() throws Exception {
28  		baseURL = new URL("http://foo.com");
29  		rc = new RecordingReplayParseContext(null, baseURL, null);
30  		st = new BlockCSSStringTransformer();
31  	}
32  
33  	public void testTransform() throws Exception {
34          final String input = 
35          		"@import \"style1.css\";\n" +
36          		"@import 'style2.css';\n" +
37          		"@import 'http://archive.org/common.css';\n" +
38          		"BODY {\n" +
39          		"  color: #fff;\n" +
40          		"  background: transparent url(bg.gif);\n" +
41          		"}\n";
42          st.transform(rc, input);
43          
44          assertEquals(4, rc.got.size());
45          assertTrue(rc.got.contains("style1.css"));
46          assertTrue(rc.got.contains("style2.css"));
47          assertTrue(rc.got.contains("http://archive.org/common.css"));
48          assertTrue(rc.got.contains("bg.gif"));
49  	}
50  	
51  	/**
52  	 * test of rewriteHttpsOnly.
53  	 * <p>Now it doesn't affect StringTransformer's behavior. This test
54  	 * will be dropped soon.</p>
55  	 * @throws Exception
56  	 */
57  	public void testRewriteHttpsOnly() throws Exception {
58  		rc.setRewriteHttpsOnly(true);
59  		
60          final String input = 
61          		"@import \"style1.css\";\n" +
62          		"@import 'style2.css';\n" +
63          		"@import 'http://archive.org/common.css';\n" +
64          		"@import 'https://secure.archive.org/common.css';\n" +
65          		"BODY {\n" +
66          		"  color: #fff;\n" +
67          		"  background: transparent url(bg.gif);\n" +
68          		"}" +
69          		".error {\n" +
70          		"  background-image: url(https://secure.archive.org/img/error-icon.gif);\n" +
71          		"}\n";
72          st.transform(rc, input);
73          
74          assertEquals(6, rc.got.size());
75          assertTrue(rc.got.contains("https://secure.archive.org/common.css"));
76          assertTrue(rc.got.contains("https://secure.archive.org/img/error-icon.gif"));
77          assertTrue(rc.got.contains("style1.css"));
78          assertTrue(rc.got.contains("style2.css"));
79          assertTrue(rc.got.contains("http://archive.org/common.css"));
80          assertTrue(rc.got.contains("bg.gif"));
81  	}
