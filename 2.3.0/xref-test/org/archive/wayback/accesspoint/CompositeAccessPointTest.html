137 	/**
138 	 * Test of {@code handleRequest()}.
139 	 * Only thing we test here is {@code createExclusionFilter()}.
140 	 * @throws Exception
141 	 */
142 	public void testHandleRequest() throws Exception {
143 		HttpServletRequest request = EasyMock.createNiceMock(HttpServletRequest.class);
144 		EasyMock.expect(request.getAttribute(RequestMapper.REQUEST_CONTEXT_PREFIX)).andStubReturn(null);
145 		// TODO: requestURI can start with "/"
146 		EasyMock.expect(request.getRequestURI()).andStubReturn("2604/2/http://example.com/");
147 		EasyMock.expect(request.getLocale()).andStubReturn(Locale.US);
148 
149 		HttpServletResponse response = EasyMock.createMock(HttpServletResponse.class);
150 
151 		// Wow, this is way too complex - something must be wrong with the design.
152 
153 		// Looks comples, but the point is to test if collection name is passed
154 		// to ContextExclusionFilterFactory.getExclusionFilter().
155 		CollectionContext expectedContext = new CollectionContext() {
156 			@Override
157 			public String getCollectionContextName() {
158 				return "2604";
159 			}
160 		};
161 		ContextExclusionFilterFactory exclusionFactory = EasyMock.createMock(ContextExclusionFilterFactory.class);
162 		// currently this is invoked twice for a replay request - should be just once.
163 		EasyMock.expect(
164 			exclusionFactory.getExclusionFilter(EasyMock.cmp(expectedContext,
165 				collectionContextComp, LogicalOperator.EQUAL))).andReturn(
166 			new ExclusionFilter() {
167 				@Override
168 				public int filterObject(CaptureSearchResult o) {
169 					return FILTER_INCLUDE;
170 				}
171 			}).atLeastOnce();
172 
173 		cut.setExclusionFactory(exclusionFactory);
174 
175 		Resource resource = AccessPointTest.createTestHtmlResource("http://example.com/", "20140101000000", "<html></html>".getBytes());
176 		final CaptureSearchResults results = AccessPointTest.setupCaptures(resourceIndex, resourceStore, 0, resource);
177 
178 		EasyMock
179 			.expect(resourceIndex.query(EasyMock.<WaybackRequest>notNull()))
180 			.andAnswer(new IAnswer<SearchResults>() {
181 				@Override
182 				public SearchResults answer() throws Throwable {
183 					WaybackRequest wbRequest = (WaybackRequest)EasyMock
184 						.getCurrentArguments()[0];
185 					// This emulates key ResourceIndex behavior 
186 					wbRequest.getAccessPoint().createExclusionFilter();
187 					return results;
188 				}
189 			});
190 		EasyMock.expect(replay.getClosest(EasyMock.<WaybackRequest>notNull(), EasyMock.<CaptureSearchResults>notNull())).andReturn(
191 			results.getClosest());
192 		
193 		response.setStatus(302);
194 		// Redirect URL shall have collection name, not "web" (see ResultURIConverter setup above).
195 		response.setHeader(EasyMock.matches("(?i)location"),
196 			EasyMock.eq("/2604/20140101000000/http://example.com/"));
197 		// XXX currently Memento is always enabled in AccessPointAdapter 
198 		// no matter what value CompositeAccessPoint.enableMemento has.
199 		response.setHeader(EasyMock.matches("(?i)link"), EasyMock.<String>notNull());
200 		EasyMock.expect(response.getWriter()).andReturn(new PrintWriter(new StringWriter()));
201 
202 		EasyMock.replay(request, response, exclusionFactory,
203 			resourceStore, resourceIndex, query, replay, replayRenderer);
204 		
205 		cut.handleRequest(request, response);
206 
207 	}
