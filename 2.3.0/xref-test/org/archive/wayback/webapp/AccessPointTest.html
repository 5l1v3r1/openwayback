333 	 * Transform input date to 14-digit timestamp: 2007-08-29T18:00:26Z =>
334 	 * 20070829180026 (stolen from WARCRecordToSearchResultAdapter - move that
335 	 * method to ArchiveUtils!)
336 	 * @param input date text in ISOZ format.
337 	 * @return date text in DT14 format.
338 	 */
339 	private static String transformWARCDate(final String input) {
340 		StringBuilder output = new StringBuilder(14);
341 		output.append(input.substring(0, 4));
342 		output.append(input.substring(5, 7));
343 		output.append(input.substring(8, 10));
344 		output.append(input.substring(11, 13));
345 		output.append(input.substring(14, 16));
346 		output.append(input.substring(17, 19));
347 		return output.toString();
348 	}

377 	 * given a sequence of {@link WarcResource}s,
378 	 * <ul>
379 	 * <li>build CaptureSearchResults, filled with CaptureSearchResult
380 	 * instances, which have auto-generated unique filename and offset of 0.
381 	 * this is necessary for equality (see {@link CaptureSearchResultMatcher}).</li>
382 	 * <li>setup ResourceStore mock to return Resource for each
383 	 * CaptureSearchResult.
384 	 * <li>
385 	 * <li>if {@code closestIndex} {@code >= 0}, set corresponding
386 	 * CaptureSearchResult's closest flag.</li>
387 	 * </ul>
388 	 * <p>
389 	 * It's left as caller's responsibility to setup {@link ResourceIndex#query(WaybackRequest)} mock
390 	 * to return {@code CaptureSearchResults} returned by this method.
391 	 * </p> 
392 	 * @param resourceIndex ResourceIndex mock
393 	 * @param resourceStore ResourceStore mock, can be {@code null}
394 	 * @param closestIndex 0-based index of resource to be marked as
395 	 *        <i>closest</i>
396 	 * @param resources sequence of WarcResources
397 	 * @return CaptureSearchResults populated with CaptureSearchResult objects.
398 	 * @throws Exception
399 	 */
400 	public static CaptureSearchResults setupCaptures(
401 			ResourceIndex resourceIndex, ResourceStore resourceStore,
402 			int closestIndex, Resource... resources) throws Exception {
403 		CaptureSearchResults results = new CaptureSearchResults();
404 		CaptureSearchResult prev = null;
405 		for (Resource res : resources) {
406 			CaptureSearchResult result = new FastCaptureSearchResult();
407 			if (prev != null) {
408 				prev.setNextResult(result);
409 				result.setPrevResult(prev);
410 			}
411 			// TODO: Resource should have methods for accessing URI and date
412 			if (res instanceof WarcResource) {
413 				// TODO: want to use WARCRecordToSearchResultAdapter?
414 				// WarcResource
415 				// has no method to retrieve underlining WARCRecord.
416 				ArchiveRecordHeader h = ((WarcResource)res).getWarcHeaders();
417 				String originalUrl = h.getUrl();
418 				String ts = (String)h.getHeaderValue("WARC-Date");
419 				// WARC-Date is in ISOZ format.
420 				ts = transformWARCDate(ts);
421 				result.setOriginalUrl(originalUrl);
422 				result.setCaptureTimestamp(ts);
423 				result.setOffset(0);
424 				// this is (W)ARC file name in real practice. here we use
425 				// DT14 timestamp as pseudo filename (.warc.gz suffix is not
426 				// essential).
427 				result.setFile(ts + ".warc.gz");
428 				if (res.getRefersToDate() != null) {
429 					// getRefersToDate() is supposed to return "yyyyMMddHHmmss"
430 					String refTimestamp = res.getRefersToDate();
431 					for (CaptureSearchResult r : results.getResults()) {
432 						if (r.getCaptureTimestamp().equals(refTimestamp)) {
433 							result.flagDuplicateDigest(r);
434 							refTimestamp = null;
435 							break;
436 						}
437 					}
438 					if (refTimestamp != null) {
439 						// no original capture found - just flag it
440 						result.flagDuplicateDigest();
441 					}
442 				}
443 			} else if (res instanceof ArcResource) {
444 				// TODO: should use ARCRecordToSearchResultAdapter? ArcResource
445 				// has getArcRecord() methods whose result may be cast to ARCRecord.
446 				// NB: ArcResource#getARCMetadata() creates a new Map object.
447 				Map<String, String> meta = ((ArcResource)res).getARCMetadata();
448 				String originalUrl = meta
449 					.get(ArchiveFileConstants.URL_FIELD_KEY);
450 				String ts = meta.get(ArchiveFileConstants.DATE_FIELD_KEY);
451 				result.setOriginalUrl(originalUrl);
452 				result.setCaptureTimestamp(ts);
453 			} else {
454 				throw new AssertionError("unexpected Resource type: " +
455 						res.getClass());
456 			}
457 			result.setHttpCode(Integer.toString(res.getStatusCode()));
458 			// CaptureSearchResultMatcher fails without this, but actual value
459 			// does not matter. so set it to 0.
460 			result.setOffset(0);
461 			assertTrue("invalid timestamp " + result.getCaptureTimestamp(),
462 				validTimestamp(result.getCaptureTimestamp()));
463 			if (closestIndex == 0) {
464 				result.setClosest(true);
465 				results.setClosest(result);
466 			}
467 
468 			if (resourceStore != null) {
469 				// Note AccessPoint passes a copy of CaptureSearchResult in some
470 				// case (ex. Replay_Revisit() test).
471 				// so we need to use custom argument matcher.
472 				EasyMock
473 					.expect(
474 						resourceStore
475 							.retrieveResource(eqCaptureSearchResult(result)))
476 					.andReturn(res).anyTimes();
477 			}
478 
479 			results.addSearchResult(result);
480 			--closestIndex;
481 		}
482 		return results;
483 	}
