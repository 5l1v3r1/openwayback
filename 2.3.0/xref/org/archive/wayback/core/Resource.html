30  /**
31   * Abstraction on top of a document stored in a WaybackCollection.
32   *
33   * TODO: This implementation needs some pretty drastic refactoring.. May have to wait
34   * for 2.0. This should be a byte-oriented record, and allow wrapping the
35   * interior byte-stream in on the more full featured HTTP libraries
36   * (jetty/apache-http-client/w3c-http-reference).
37   *
38   * For now, it is a system-wide assumption that all resources are HTTP based.
39   *
40   * TODO: Some code downcasts Resource to its sub-classes to gain access to
41   * methods only available in specific implementation.  Consider adding more methods
42   * to make downcast unnecessary.  More sub-classes are expected, for encapsulating
43   * revisit-original pair of Resources as single Resource, for example.
44   *
45   * @see org.archive.wayback.ResourceStore#retrieveResource(CaptureSearchResult)
46   *
47   * @author Brad Tofel
48   */
49  public abstract class Resource extends InputStream {
50  
51  	private InputStream is;
52  
53      public abstract void close() throws IOException;
54  	/**
55  	 * Assumes an HTTP resource - return the HTTP response code
56  	 * @return the HTTP response code from the HTTP message
57  	 */
58  	public abstract int getStatusCode();
59  	/**
60  	 * @return the size in bytes of the record payload, including HTTP header
61  	 */
62  	public abstract long getRecordLength();
63  	/**
64  	 * Assumes an HTTP response - return the HTTP headers, not including the
65  	 * HTTP Message header
66  	 * @return key-value Map of HTTP headers
67  	 */
68  	public abstract Map<String,String> getHttpHeaders();
69  
70  	// URL-Agnostic Revisit Support
71  
72  	/**
73  	 * return {@code WARC-Refer-To-Target-URI} WARC record header value or
74  	 * equivalent.
75  	 * Default implementation returns {@code null}.
76  	 * @return header value (URI)
77  	 */
78  	public String getRefersToTargetURI() {
79  		return null;
80  	}
81  
82  	/**
83  	 * return {@code WARC-Refers-To-Date} WARC record header value or
84  	 * equivalent.
85  	 * Default implementation returns {@code null}
86  	 * @return 14-digit timestamp string ({@code yyyyMMddHHmmss})
87  	 */
88  	public String getRefersToDate() {
89  		return null;
90  	}
91  
92  	public void parseHeaders() throws IOException {
93  		//Implemented in warc/arc reader
94  	}
95  
96  	public String getHeader(String headerName) {
97  		Map<String, String> httpHeaders = getHttpHeaders();
98  
99  		if (httpHeaders == null) {
100 			return null;
101 		}
102 
103 		Iterator<String> keys = httpHeaders.keySet().iterator();
104 
105 		String headerUp = headerName.toUpperCase();
106 
107 		while (keys.hasNext()) {
108 			String key = keys.next();
109 
110 			if (key.toUpperCase().equals(headerUp)) {
111 				return httpHeaders.get(key);
112 			}
113 		}
114 
115 		return null;
116 	}
117 
118 	private void validate() throws IOException {
119 		if(is == null) {
120 			throw new IOException("No InputStream");
121 		}
122 	}
123 
124 	protected void setInputStream(InputStream is) {
125 		if(is.markSupported()) {
126 			this.is = is;
127 		} else {
128 			this.is = new BufferedInputStream(is);
129 		}
130 	}
131 
132 	/**
133 	 * indicate that there is a {@code Transfer-Encoding: chunked} header, so the input
134 	 *   data should be dechunked as it is read. This method actually peeks
135 	 *   ahead to verify that there is a hex-encoded chunk length before
136 	 *   assuming the data is chunked.
137 	 * @throws IOException for usual reasons
138 	 */
139 	public void setChunkedEncoding() throws IOException {
140 		validate();
141 		// peek ahead and make sure we have a line with hex numbers:
142 		int max = 50;
143 		is.mark(max + 2);
144 		int cur = 0;
145 		int hexFound = 0;
146 		boolean isChunked = false;
147 		while (cur < max) {
148 			int nextC = is.read();
149 			// allow CRLF and plain ole LF:
150 			if ((nextC == 13) || (nextC == 10)) {
151 				// must have read at least 1 hex char:
152 				if (hexFound > 0) {
153 					if (nextC == 10) {
154 						isChunked = true;
155 						break;
156 					}
157 					nextC = is.read();
158 					if (nextC == 10) {
159 						isChunked = true;
160 						break;
161 					}
162 				}
163 				// keep looking to allow some blank lines.
164 			} else {
165 				// better be a hex character:
166 				if (isHex(nextC)) {
167 					hexFound++;
168 				} else if (nextC != ' ') {
169 					// allow whitespace before or after chunk...
170 					// not a hex digit: not a chunked stream.
171 					break;
172 				}
173 			}
174 			cur++;
175 		}
176 		is.reset();
177 		if (isChunked) {
178 			setInputStream(new ChunkedInputStream(is));
179 		}
180 	}
181 
182 	private boolean isHex(int c) {
183 		if ((c >= '0') && (c <= '9')) {
184 			return true;
185 		}
186 		if ((c >= 'a') && (c <= 'f')) {
187 			return true;
188 		}
189 		if ((c >= 'A') && (c <= 'F')) {
190 			return true;
191 		}
192 		return false;
193 	}
194 
195 	public int available() throws IOException {
196 		validate();
197 		return is.available();
198 	}
199 
200 	public void mark(int readlimit) {
201 		if (is != null) {
202 			is.mark(readlimit);
203 		}
204 	}
205 
206 	public boolean markSupported() {
207 		if (is == null) {
208 			return false;
209 		}
210 		return is.markSupported();
211 	}
212 
213 	public int read() throws IOException {
214 		validate();
215 		return is.read();
216 	}
217 
218 	public int read(byte[] b, int off, int len) throws IOException {
219 		validate();
220 		return is.read(b, off, len);
221 	}
222 
223 	public int read(byte[] b) throws IOException {
224 		validate();
225 		return is.read(b);
226 	}
227 
228 	public void reset() throws IOException {
229 		validate();
230 		is.reset();
231 	}
232 
233 	public long skip(long n) throws IOException {
234 		validate();
235 		return is.skip(n);
236 	}
