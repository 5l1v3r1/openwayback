89      public void testRenderResource_BasicCapture() throws Exception {
90          final String ct = "image/gif";
91          WARCRecordInfo recinfo = TestWARCRecordInfo.createHttpResponse(ct, TestWARCRecordInfo.PAYLOAD_GIF);
92          TestWARCReader ar = new TestWARCReader(recinfo);
93          WARCRecord rec = ar.get(0);
94          Resource payloadResource = new WarcResource(rec, ar);
95          payloadResource.parseHeaders();
96          Resource headersResource = payloadResource;
97          
98          TestServletOutputStream servletOutput = new TestServletOutputStream();
99          response.setStatus(200);
100         EasyMock.expect(response.getOutputStream()).andReturn(servletOutput);
101         response.setHeader("Content-Type", ct);
102         // ??? RedirectRewritingHttpHeaderProcessor drops Content-Length header. is this really
103         // it is supposed to do?
104         //response.setHeader("Content-Length", Integer.toString(payloadBytes.length));
105         response.setHeader(EasyMock.<String>notNull(), EasyMock.<String>notNull());
106         EasyMock.expectLastCall().anyTimes();
107         EasyMock.replay(response);
108 
109         cut.renderResource(request, response, wbRequest, result,
110                 headersResource, payloadResource, uriConverter, results);
111         
112         EasyMock.verify(response);
113         byte[] content = servletOutput.getBytes();
114         assertTrue("servlet output", Arrays.equals(TestWARCRecordInfo.PAYLOAD_GIF, content));
115     }

138         response.setStatus(200);
139         EasyMock.expect(response.getOutputStream()).andReturn(servletOutput);

142         // ??? RedirectRewritingHttpHeaderProcessor drops Content-Length header. is this really

161     public void testRenderResource_Redirect() throws Exception {
162         String location = "http://www.example.com/index.html";
163         WARCRecordInfo recinfo = new TestWARCRecordInfo(TestWARCRecordInfo.buildHttpRedirectResponseBlock(location));
164         TestWARCReader ar = new TestWARCReader(recinfo);
165         WARCRecord rec = ar.get(0);
166         Resource payloadResource = new WarcResource(rec, ar);
167         payloadResource.parseHeaders();
168         
169         final String originalUrl = "http://www.example.com/";
170         final String captureTimestamp = "20130101123456";
171         
172         result.setOriginalUrl(originalUrl);
173         result.setCaptureTimestamp(captureTimestamp);
174         
175         // makeReplayURI() is called through RedirectRewritingHttpHeaderProcessor.
176         // TODO: perhaps HttpheaderProcessor is the right class to make fixture?
177         EasyMock.expect(uriConverter.makeReplayURI(captureTimestamp, location))
178         .andReturn("/web/" + captureTimestamp + "/" + location);
179 
180         TestServletOutputStream servletOutput = new TestServletOutputStream();
181         response.setStatus(302);
182         EasyMock.expect(response.getOutputStream()).andReturn(servletOutput);
183         response.setHeader("Content-Type", "text/html");
184         response.setHeader(EasyMock.eq("Location"), EasyMock.matches("/web/" + captureTimestamp + "/" + location));
185         // RedirectRewritingHttpHeaderProcessor drops Content-Length.
186         // response.setHeader("Content-Length", "0");
187         response.setHeader(EasyMock.<String>notNull(), EasyMock.<String>notNull());
188         EasyMock.expectLastCall().anyTimes();
189         
190         EasyMock.replay(response, uriConverter);
191 
192         cut.renderResource(request, response, wbRequest, result,
193                 payloadResource, payloadResource, uriConverter, results);
194         
195         EasyMock.verify(response, uriConverter);
196 
197         byte[] content = servletOutput.getBytes();
198         assertEquals("payload length", 0, content.length);
199     }

201     /**
202      * test replay of capture with {@code Transfer-Encoding: chunked}.
203      *
204      * <p>TransparentReplayRenderer writes out chunk-decoded payload, because
205      * {@link WarcResource} always decodes chunked-entity. Point of this test
206      * is that response never have {@code Transfer-Encoding: chunked} header,
207      * even when initialized with {@link IdentityHttpHeaderProcessor}.
208      * so, this is not really a unit test for TransparentReplayRenderer, but
209      * a multi-component test placed here for convenience.</p>
210      * <p>This test does not use member object {@code cut}, in order to test
211      * with {@link IdentityHttpHeaderProcessor}.</p>
212      *
213      * @throws Exception
214      */
215     public void testRenderResource_Chunked() throws Exception {
216         final String ct = "text/xml";
217         final String payload = "<?xml version=\"1.0\"?>\n" +
218         		"<payload name=\"archive\">\n" +
219         		"  <inside/>\n" +
220         		"</payload>\n";
221         final byte[] recordBytes = TestWARCRecordInfo.buildHttpResponseBlock(
222         		"200 OK", ct, payload.getBytes("UTF-8"), true);
223         //System.out.println(new String(recordBytes, "UTF-8"));
224         WARCRecordInfo recinfo = new TestWARCRecordInfo(recordBytes);
225         TestWARCReader ar = new TestWARCReader(recinfo);
226         WARCRecord rec = ar.get(0);
227         Resource payloadResource = new WarcResource(rec, ar);
228         payloadResource.parseHeaders();
229         Resource headersResource = payloadResource;
230 
231         TestServletOutputStream servletOutput = new TestServletOutputStream();
232         // expectations
233         response.setStatus(200);
234         EasyMock.expect(response.getOutputStream()).andReturn(servletOutput);
235         // capture setHeader() call for "Transfer-Encoding"
236         Capture<String> transferEncodingCapture = new Capture<String>(CaptureType.FIRST);
237         response.setHeader(EasyMock.eq("Transfer-Encoding"), EasyMock.capture(transferEncodingCapture));
238         EasyMock.expectLastCall().anyTimes();
239         response.setHeader(EasyMock.<String>anyObject(), EasyMock.<String>anyObject());
240         EasyMock.expectLastCall().anyTimes();
241 
242         EasyMock.replay(response);
243 
244         // creating separate test object to use IdentityHttpHeaderProcessor
245         TransparentReplayRenderer cut2 = new TransparentReplayRenderer(new IdentityHttpHeaderProcessor());
246         cut2.renderResource(request, response, wbRequest, result,
247         		headersResource, payloadResource, uriConverter, results);
248 
249         EasyMock.verify(response);
250 
251         assertFalse("Transfer-Encoding header must not be set", transferEncodingCapture.hasCaptured());
252 
253         // content is the original gzip-compressed bytes for PAYLOAD_GIF.
254         String output = new String(servletOutput.getBytes(), "UTF-8");
255         assertEquals(payload, output);
256     }
