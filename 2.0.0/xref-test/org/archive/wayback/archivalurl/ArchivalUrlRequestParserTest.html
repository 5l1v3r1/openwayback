49      protected void setUp() throws Exception {
50          super.setUp();
51          cut = new ArchivalUrlRequestParser();
52          // this affects what WaybackRequest.getStartTimestamp() returns.
53          // here we leave lastestTimestamp null.
54          cut.setEarliestTimestamp(EARLIEST_TIMESTAMP);
55          // this needs to be called before use (but I don't see it called in
56          // Spring config.)
57          cut.init();
58          // refactoring note: RequestParser#parser takes AccessPoint as an argument.
59          // it is bad because AccessPoint is not mock-able and it is a big class.
60          // it'd be great if we can cut out an interface required for RequestParser
61          // and use it in RequestParser instead.
62          // (partial) list of AccessPoint methods used by RequestParser:
63          // - translateRequestPathQuery(HttpServletRequest) (defined in RequestHandler)
64          // - getMapParam(Map, String) (static)
65          // - isEnableMemento()
66          accessPoint = new AccessPoint();
67      }

69      protected HttpServletRequest getRequestMock(String requestURI, String query) {
70          HttpServletRequest mock = EasyMock.createNiceMock(HttpServletRequest.class);
71          // DatelessReplayRequestParser checks out Accept-Datetime HTTP header. If we
72          // want to test it, we need to setup return value for getHeader() method.
73          // RequestMapper accesses HttpServletequest.getAttribute(RequestMapper.REQUEST_CONTEXT_PREFIX)
74          // which can be null. so we leave it null.
75          // prefix must end with "/" (see RequestMapper#handleRequest(HttpServletRequest, HttpServletResponse)
76          EasyMock.expect(mock.getAttribute(EasyMock.eq("webapp-request-context-path-prefix"))).andStubReturn("/web/");
77          EasyMock.expect(mock.getRequestURI()).andStubReturn(requestURI);
78          EasyMock.expect(mock.getQueryString()).andStubReturn(query);
79          EasyMock.expect(mock.getHeader(MementoConstants.ACCEPT_DATETIME)).andStubReturn(acceptDatetimeHeader);
80          EasyMock.expect(mock.getHeader("Accept-Timestamp")).andStubReturn(acceptTimestampHeader);
81          // used by OpenSearchRequestParser
82          EasyMock.expect(mock.getParameterMap()).andStubReturn(Collections.EMPTY_MAP);
83          return mock;
84      }

86      protected WaybackRequest parse(String url) throws BadQueryException, BetterRequestException {
87  	    // note: url is set to HttpServletRequest's requestURI property. it should NOT contain
88  		// query part. RequestParser will receive value translated by RequestMapper.getRequestContextPath()
89  		// (i.e. "/web/" part removed.)
90  		HttpServletRequest req = getRequestMock(url, null);
91  		EasyMock.replay(req);
92  		return cut.parse(req, accessPoint);
93      }

123     public void testDatePrefix() throws Exception {
124     	// less-than-14-digit timestamp with "*": narrowed time range, highlight
125     	// the latest within the range.
126     	WaybackRequest wbr1 = parse("/web/20100101*/http://www.yahoo.com/?p=2");
127     	assertNotNull(wbr1);
128     	assertTrue(wbr1.isCaptureQueryRequest());
129     	assertEquals("20100101000000", wbr1.getStartTimestamp());
130     	assertEquals("20100101235959", wbr1.getEndTimestamp());
131     	assertEquals("20100101235959", wbr1.getReplayTimestamp());
132     	assertEquals("http://www.yahoo.com/?p=2", wbr1.getRequestUrl());
133     	
134     	// just "*": entire time range, replay the latest.
135     	WaybackRequest wbr2 = parse("/web/*/http://www.yahoo.com/");
136     	assertNotNull(wbr2);
137     	assertTrue(wbr2.isCaptureQueryRequest());
138     	assertEquals(EXPECTED_START_TIMESTAMP, wbr2.getStartTimestamp());
139     	assertEquals(EXPECTED_END_TIMESTAMP, wbr2.getEndTimestamp());
140     	assertEquals(null, wbr2.getReplayTimestamp());
141     	
142     	// full 14-digit timestamp with "*": entire time range, highlight the 
143     	// closest to the specified date.
144     	WaybackRequest wbr3 = parse("/web/20100101000000*/http://www.yahoo.com/");
145     	assertNotNull(wbr3);
146     	assertTrue(wbr3.isCaptureQueryRequest());
147     	assertEquals(EXPECTED_START_TIMESTAMP, wbr3.getStartTimestamp());
148     	assertEquals(EXPECTED_END_TIMESTAMP, wbr3.getEndTimestamp());
149     	assertEquals("20100101000000", wbr3.getReplayTimestamp());
150     }

322 	/**
323 	 * test of {@link DatelessReplayRequestParser}.
324 	 * <p>It also reads Replay/AchorDate from Accept-Date/Accept-Timestamp
325 	 * HTTP headers (Accept-Date is part of Memento protocol). This test is
326 	 * included here, even though it has some Memento-related code, because
327 	 * core part of processing is for Archival-URL.</p>
328 	 * <p>One notable difference from with-timestamp request is that
329 	 * {@link WaybackRequest#isBestLatestReplayRequest} is set to true, which used
330 	 * in several places for unclear purpose. This flag is not set in other
331 	 * RequestParser's, even if request is asking for the best latest capture.
332 	 * Probably flag's semantics is slightly different from what its name suggests.</p>
333 	 * <p>As this pattern of requests come from URL that escaped Archival-URL
334 	 * rewriting process, it runs extra check on it. Test includes a few negative cases
335 	 * for non-URL paths.</p>
336 	 * @throws Exception
337 	 */
338 	public void testPathDateless() throws Exception {
339 		{
340 			WaybackRequest wbr = parse("/web/http://www.yahoo.com/");
341 			checkPathDateless(wbr, "http://www.yahoo.com/");
342 		}
343 		{
344 			WaybackRequest wbr = parse("/web/https://www.yahoo.com/");
345 			checkPathDateless(wbr, "https://www.yahoo.com/");
346 		}
347 		{
348 			WaybackRequest wbr = parse("/web/http://www.yahoo.com:8080/");
349 			checkPathDateless(wbr, "http://www.yahoo.com:8080/");
350 		}
351 		// some client canonicalizes "//" in path into "/".
352 		// (why this isn't done for other patterns of requests?)
353 		{
354 			WaybackRequest wbr = parse("/web/http:/www.yahoo.com/");
355 			checkPathDateless(wbr, "http://www.yahoo.com/");
356 		}
357 		// doesn't repair "https:/"
358 		{
359 			WaybackRequest wbr = parse("/web/https:/www.yahoo.com/");
360 			//checkPathDateless(wbr, "https://www.yahoo.com/");
361 			assertNull(wbr);
362 		}
363 		// doesn't repair "ftp:/" either.
364 		{
365 			WaybackRequest wbr = parse("/web/ftp:/www.yahoo.com/afile");
366 			assertNull(wbr);
367 		}
368 		// scheme-relative - results in NullPointerException FIXME
369 		{
370 			try {
371 				WaybackRequest wbr = parse("/web///www.yahoo.com/");
372 				assertNull(wbr);
373 			} catch (NullPointerException ex) {
374 				// current behavior - FIXME.
375 			}
376 		}
377 		// regular case.
378 		{
379 			WaybackRequest wbr = parse("/web/www.yahoo.com/");
380 			checkPathDateless(wbr, "http://www.yahoo.com/");
381 		}
382 		// scheme-less URL with user info is rejected
383 		// TODO: why is this rejected?
384 		{
385 			try {
386 				@SuppressWarnings("unused")
387 				WaybackRequest wbr = parse("/web/user@www.yahoo.com/");
388 				fail("BadQueryException was not thrown");
389 			} catch (BadQueryException ex) {
390 				// expected;
391 			}
392 		}
393 		// but it's accepted with scheme.
394 		// TODO: should this be rejected as well?
395 		{
396 			WaybackRequest wbr = parse("/web/http://user@www.yahoo.com/");
397 			checkPathDateless(wbr, "http://user@www.yahoo.com/");
398 		}
399 		// just make sure path and query parts in requestUrl are preserved.
400 		{
401 			WaybackRequest wbr = parse("/web/www.yahoo.com/apis?v=2");
402 			checkPathDateless(wbr, "http://www.yahoo.com/apis?v=2");
403 		}
404 		// doesn't look like an URL.
405 		{
406 			WaybackRequest wbr = parse("/web/images/foo.gif");
407 			assertNull(wbr);
408 		}
409 		{
410 			WaybackRequest wbr = parse("/web/handler.php?url=http://www.yahoo.com/");
411 			assertNull(wbr);
412 		}
413 		// TODO: shouldn't this be parsed as dateless URL-Query?
414 		{
415 			WaybackRequest wbr = parse("/web/http://www.yahoo.com/*");
416 			checkPathDateless(wbr, "http://www.yahoo.com/*");
417 		}
418 		// ditto
419 		{
420 			WaybackRequest wbr = parse("/web/www.yahoo.com/*");
421 			checkPathDateless(wbr, "http://www.yahoo.com/*");
422 		}

425 	public void testPathDatelessWithDateHeader() throws Exception {
426 		final String dateHeader = "Thu, 24 Apr 2014 21:15:51 UTC+00:00";
427 		final Date date = (new SimpleDateFormat("EEE, dd MMM yyyy hh:mm:ss z", Locale.ENGLISH)).parse(dateHeader);
428 		{
429 			acceptDatetimeHeader = dateHeader;
430 			WaybackRequest wbr = parse("/web/http://www.yahoo.com/");
431 			assertNotNull(wbr);
432 			assertTrue(wbr.isReplayRequest());
433 			assertFalse(wbr.isBestLatestReplayRequest());
434 			assertEquals(EXPECTED_START_TIMESTAMP, wbr.getStartTimestamp());
435 			assertEquals(EXPECTED_END_TIMESTAMP, wbr.getEndTimestamp());
436 			assertEquals(date, wbr.getReplayDate());
437 			assertEquals(date, wbr.getAnchorDate());
438 		}
439 		// invalid Accept-Datetime header
440 		{
441 			acceptDatetimeHeader = "invalid date";
442 			try {
443 				@SuppressWarnings("unused")
444 				WaybackRequest wbr = parse("/web/http://www.yahoo.com/");
445 				fail("did not throw exception");
446 			} catch (TimeGateBadQueryException ex) {
447 				// expected
448 			}
449 		}
450 		// alternate Accept-Timestamp header. as long as it is valid,
451 		// invalid value in Accept-Datetime header doesn't cause exception.
452 		{
453 			acceptDatetimeHeader = "invalid date";
454 			acceptTimestampHeader = "20140424211551";
455 			WaybackRequest wbr = parse("/web/http://www.yahoo.com/");
456 			assertNotNull(wbr);
457 			assertTrue(wbr.isReplayRequest());
458 			assertFalse(wbr.isBestLatestReplayRequest());
459 			assertEquals(EXPECTED_START_TIMESTAMP, wbr.getStartTimestamp());
460 			assertEquals(EXPECTED_END_TIMESTAMP, wbr.getEndTimestamp());
461 			assertEquals(date, wbr.getReplayDate());
462 			assertEquals(date, wbr.getAnchorDate());
463 		}
464 		// invalid value in Accept-Timestamp header is silently ignored,
465 		// unless Accept-Datetime also has an invalid value. 
466 		{
467 			acceptDatetimeHeader = null;
468 			acceptTimestampHeader = "*INVALID*";
469 			WaybackRequest wbr = parse("/web/http://www.yahoo.com/");
470 			assertNotNull(wbr);
471 			assertTrue(wbr.isReplayRequest());
472 			assertTrue(wbr.isBestLatestReplayRequest());
473 			assertEquals(EXPECTED_START_TIMESTAMP, wbr.getStartTimestamp());
474 			assertEquals(EXPECTED_END_TIMESTAMP, wbr.getEndTimestamp());
475 		}
