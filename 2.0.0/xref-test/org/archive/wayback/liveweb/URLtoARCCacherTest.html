15  	public void testSocketTimeout() throws Exception {

18  	    HttpClient http = null;
19      	connectionManager = new MultiThreadedHttpConnectionManager();
20      	hostConfiguration = new HostConfiguration();
21      	http = new HttpClient(connectionManager);
22      	http.setHostConfiguration(hostConfiguration);
23  		HttpMethod method = null;
24  //		String urlString = "http://wayback.archive-it.org:6100/one";
25  		String urlString = "http://netpreserve.org/";
26  		int socketTimeoutMS = 100;
27  		int connectTimeoutMS = 200;
28      	connectionManager.getParams().setSoTimeout(socketTimeoutMS);
29      	connectionManager.getParams().setConnectionTimeout(connectTimeoutMS);
30  		try {
31  			method = new GetMethod(urlString);
32  		} catch(IllegalArgumentException e) {
33  			throw new LiveDocumentNotAvailableException("Url:" + urlString +
34  					"does not look like an URL?");
35  		}
36  	    try {
37  	    	int status = http.executeMethod(method);
38  	    	System.out.println("Got response code: " + status);
39  	    	ByteOp.copyStream(method.getResponseBodyAsStream(), System.out);
40  	    } catch (Exception e) {
41  	    	e.printStackTrace();
42  	    }

44  	}
