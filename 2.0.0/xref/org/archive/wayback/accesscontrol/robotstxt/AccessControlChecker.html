3   import java.io.IOException;
4   import java.io.PrintWriter;
5   
6   import javax.servlet.ServletException;
7   import javax.servlet.http.HttpServletRequest;
8   import javax.servlet.http.HttpServletResponse;
9   
10  import org.apache.commons.httpclient.URIException;
11  import org.archive.wayback.UrlCanonicalizer;
12  import org.archive.wayback.accesscontrol.ExclusionFilterFactory;
13  import org.archive.wayback.core.FastCaptureSearchResult;
14  import org.archive.wayback.resourceindex.filters.ExclusionFilter;
15  import org.archive.wayback.util.webapp.AbstractRequestHandler;
16  
17  public class AccessControlChecker extends AbstractRequestHandler {
18  	
19  	protected final static String HTTP_PREFIX = "http://";
20  	protected final static String HTTPS_PREFIX = "https://";
21  	
22  	protected ExclusionFilterFactory exclusionFactory;
23  	protected UrlCanonicalizer canonicalizer = null;
24  
25  	@Override
26  	public boolean handleRequest(HttpServletRequest httpRequest,
27  			HttpServletResponse httpResponse) throws ServletException,
28  			IOException {
29  		
30  		String url = this.translateRequestPath(httpRequest);
31  		
32  		if (!url.startsWith(HTTP_PREFIX) && !url.startsWith(HTTPS_PREFIX)) {
33  			url = HTTP_PREFIX + url;
34  		}
35  		
36  		PrintWriter writer = httpResponse.getWriter();
37  		httpResponse.setContentType("application/json");
38  		return checkAccess(url, writer);
39  	}
40  	
41  	protected boolean checkAccess(String url, PrintWriter writer) {
42  		if (exclusionFactory == null) {
43  			return false;
44  		}
45  		
46  		ExclusionFilter filter = exclusionFactory.get();
47  		FastCaptureSearchResult result = new FastCaptureSearchResult();
48  		result.setOriginalUrl(url);
49  		
50  		int status = ExclusionFilter.FILTER_EXCLUDE;
51  		
52  		String urlKey = url;
53  		
54  		try {
55  			// Must set canonicalizer
56  			urlKey = canonicalizer.urlStringToKey(url);			
57  			result.setUrlKey(urlKey);
58  			
59  			status = filter.filterObject(result);
60  		} catch (URIException e) {
61  			//Exclude invalid
62  		}
63  		
64  		if (status == ExclusionFilter.FILTER_INCLUDE) {
65  			writer.println("[\"allow\"]");
66  		} else {
67  			writer.println("[\"block\"]");
68  		}
69  		
70  		return true;
71  	}
72  
73  	public ExclusionFilterFactory getExclusionFactory() {
74  		return exclusionFactory;
75  	}
76  
77  	public void setExclusionFactory(ExclusionFilterFactory exclusionFactory) {
78  		this.exclusionFactory = exclusionFactory;
79  	}
80  
81  	public UrlCanonicalizer getCanonicalizer() {
82  		return canonicalizer;
83  	}
84  
85  	public void setCanonicalizer(UrlCanonicalizer canonicalizer) {
86  		this.canonicalizer = canonicalizer;
87  	}
88  }
