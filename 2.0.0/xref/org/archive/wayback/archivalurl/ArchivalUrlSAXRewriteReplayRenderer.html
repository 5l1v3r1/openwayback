89  	public void renderResource(HttpServletRequest httpRequest,
90  			HttpServletResponse httpResponse, WaybackRequest wbRequest,
91  			CaptureSearchResult result, Resource resource,
92  			ResultURIConverter uriConverter, CaptureSearchResults results)
93  					throws ServletException, IOException, WaybackException {
94  		renderResource(httpRequest, httpResponse, wbRequest, result, resource,
95  				resource, uriConverter, results);

98  	@Override
99  	public void renderResource(HttpServletRequest httpRequest,
100 			HttpServletResponse httpResponse, WaybackRequest wbRequest,
101 			CaptureSearchResult result, Resource httpHeadersResource,
102 			Resource payloadResource, ResultURIConverter uriConverter,
103 			CaptureSearchResults results) throws ServletException, IOException,
104 			WaybackException {
105 
106 		Resource decodedResource = TextReplayRenderer.decodeResource(httpHeadersResource, payloadResource);
107 
108 		// The URL of the page, for resolving in-page relative URLs: 
109 		URL url = null;
110 		try {
111 			url = new URL(result.getOriginalUrl());
112 		} catch (MalformedURLException e1) {
113 			// TODO: this shouldn't happen...
114 			e1.printStackTrace();
115 			throw new IOException(e1.getMessage());
116 		}
117 		// determine the character set used to encode the document bytes:
118 		String charSet = charsetDetector.getCharset(httpHeadersResource, decodedResource, wbRequest);
119 
120 		ContextResultURIConverterFactory fact = createConverterFactory(uriConverter, httpRequest, wbRequest);
121 		
122 		// set up the context:
123 		ReplayParseContext context = 
124 				new ReplayParseContext(fact,url,result.getCaptureTimestamp());
125 		
126 		context.setRewriteHttpsOnly(rewriteHttpsOnly);
127 
128 		if(!wbRequest.isFrameWrapperContext()) {
129 			// in case this is an HTML page with FRAMEs, peek ahead an look:
130 			// TODO: make ThreadLocal:
131 			byte buffer[] = new byte[FRAMESET_SCAN_BUFFER_SIZE];
132 
133 			decodedResource.mark(FRAMESET_SCAN_BUFFER_SIZE);
134 			int amtRead = decodedResource.read(buffer);
135 			decodedResource.reset();
136 
137 			if(amtRead > 0) {
138 				StringBuilder foo = new StringBuilder(new String(buffer,charSet));
139 				int frameIdx = TagMagix.getEndOfFirstTag(foo, "FRAMESET");
140 				if(frameIdx != -1) {
141 					// insert flag so we don't add FRAMESET:
142 					context.putData(FastArchivalUrlReplayParseEventHandler.FERRET_DONE_KEY,"");
143 
144 //					// top-level Frameset: Draw the frame wrapper thingy:
145 //					frameWrappingRenderer.renderResource(httpRequest, 
146 //							httpResponse, wbRequest, result, resource, 
147 //							uriConverter, results);
148 //					return;
149 				}
150 			}
151 		}
152 
153 
154 		// copy the HTTP response code:
155 		HttpHeaderOperation.copyHTTPMessageHeader(httpHeadersResource, httpResponse);
156 
157 		// transform the original headers according to our headerProcessor:
158 		Map<String,String> headers = HttpHeaderOperation.processHeaders(
159 				httpHeadersResource, result, uriConverter, httpHeaderProcessor);
160 
161 		// prepare several objects for the parse:
162 
163 		// a JSPExecutor:
164 		JSPExecutor jspExec = new JSPExecutor(uriConverter, httpRequest, 
165 				httpResponse, wbRequest, results, result, decodedResource);
166 
167 
168 		// To make sure we get the length, we have to buffer it all up...
169 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
170 
171 		context.setOutputCharset(OUTPUT_CHARSET);
172 		context.setOutputStream(baos);
173 		context.setJspExec(jspExec);
174 
175 
176 		// and finally, parse, using the special lexer that knows how to
177 		// handle javascript blocks containing unescaped HTML entities:
178 		Page lexPage = new Page(decodedResource,charSet);
179 		Lexer lexer = new Lexer(lexPage);
180 		Lexer.STRICT_REMARKS = false;
181 		ContextAwareLexer lex = new ContextAwareLexer(lexer, context);
182 		Node node;
183 		try {
184 			delegator.handleParseStart(context);
185 			while((node = lex.nextNode()) != null) {
186 				delegator.handleNode(context, node);
187 			}
188 			delegator.handleParseComplete(context);
189 		} catch (ParserException e) {
190 			e.printStackTrace();
191 			throw new IOException(e.getMessage());
192 		}
193 
194 		// At this point, baos contains the utf-8 encoded bytes of our result:
195 		byte[] utf8Bytes = baos.toByteArray();
196 		// set the corrected length:
197 		headers.put(HttpHeaderOperation.HTTP_LENGTH_HEADER, 
198 				String.valueOf(utf8Bytes.length));
199 		headers.put(TextReplayRenderer.GUESSED_CHARSET_HEADER, charSet);
200 
201 		// send back the headers:
202 		HttpHeaderOperation.sendHeaders(headers, httpResponse);
203 		// Tomcat will always send a charset... It's trying to be smarter than
204 		// we are. If the original page didn't include a "charset" as part of
205 		// the "Content-Type" HTTP header, then Tomcat will use the default..
206 		// who knows what that is, or what that will do to the page..
207 		// let's try explicitly setting it to what we used:
208 		httpResponse.setCharacterEncoding(OUTPUT_CHARSET);
209 		httpResponse.getOutputStream().write(utf8Bytes);
210 	}

212 	protected ContextResultURIConverterFactory createConverterFactory(ResultURIConverter uriConverter, HttpServletRequest httpRequest, WaybackRequest wbRequest)
213 	{
214 		// sam ecode in ArchivalURLJSStringTransformerReplayRenderer
215 		ContextResultURIConverterFactory fact = null;
216 		
217 		if (uriConverter instanceof ArchivalUrlResultURIConverter) {
218 			fact = new ArchivalUrlContextResultURIConverterFactory(
219 					(ArchivalUrlResultURIConverter) uriConverter);
220 		} else if (converterFactory != null) {
221 			fact = converterFactory;
222 		} else {
223 			fact = new IdentityResultURIConverterFactory(uriConverter);			
224 		}
225 		
226 		return fact;
227 	}
