183 	 * @return the original URL which resulted in the capture. If it is not 
184 	 * available, the urlKey and original Host will be used to reconstruct 
185 	 * something possibly closer to the original URL than the urlKey
186 	 */
187 	public String getOriginalUrl() {
188 		String url = get(CAPTURE_ORIGINAL_URL);
189 		if(url == null) {
190 			// convert from ORIG_HOST to ORIG_URL here:
191 			url = getUrlKey();
192 			String host = get(CAPTURE_ORIGINAL_HOST);
193 			if(url != null && host != null) {
194 				StringBuilder sb = new StringBuilder(url.length());
195 				sb.append(UrlOperations.DEFAULT_SCHEME);
196 				sb.append(host);
197 				sb.append(UrlOperations.getURLPath(url));
198 				url = sb.toString();
199 				// cache it for next time...?
200 				setOriginalUrl(url);
201 			}
202 		}
203 		return url;
204 	}

314 	public void flagDuplicateDigest(CaptureSearchResult payload) {
315 		flagDuplicateDigest();
316 		put(CAPTURE_DUPLICATE_STORED_TS, payload.getCaptureTimestamp());
317 		put(CAPTURE_DUPLICATE_PAYLOAD_FILE, payload.getFile());
318 		put(CAPTURE_DUPLICATE_PAYLOAD_OFFSET, String.valueOf(payload.getOffset()));
319 		if (payload.getCompressedLength() > 0) {
320 			put(CAPTURE_DUPLICATE_PAYLOAD_COMPRESSED_LENGTH, String.valueOf(payload.getCompressedLength()));
321 		}
322 	}

409 		String flags = getRobotFlags();
410 		if(flags == null) {

415 		}
416 		setRobotFlags(flags);
417 	}

477 	public void removeFromList()
478 	{
479 		if (nextResult != null) {
480 			nextResult.setPrevResult(prevResult);
481 		}
482 		
483 		if (prevResult != null) {
484 			prevResult.setNextResult(nextResult);
485 		}
486 		
487 		prevResult = null;
488 		nextResult = null;
489 	}

496 	public boolean isHttpError() {
497 		if (this.isDuplicateDigest() && (this.getDuplicatePayload() != null)) {
498 			return this.getDuplicatePayload().isHttpError();
499 		}
500 		String httpCode = this.getHttpCode();
501 		return (httpCode.startsWith("4") || httpCode.startsWith("5"));
502 	}
503 	
504 	public boolean isHttpRedirect() {
505 		if (this.isDuplicateDigest() && (this.getDuplicatePayload() != null)) {
506 			return this.getDuplicatePayload().isHttpRedirect();
507 		}		
508 		String httpCode = this.getHttpCode();
509 		return (httpCode.startsWith("3"));
510 	}
511 	
512 	public boolean isHttpSuccess() {
513 		if (this.isDuplicateDigest() && (this.getDuplicatePayload() != null)) {
514 			return this.getDuplicatePayload().isHttpSuccess();
515 		}			
516 		String httpCode = this.getHttpCode();
517 		return (httpCode.startsWith("2"));
518 	}
