20  package org.archive.wayback.webapp;
21  
22  import java.io.IOException;
23  import java.util.List;
24  
25  import org.archive.wayback.ResourceIndex;
26  import org.archive.wayback.ResourceStore;
27  import org.archive.wayback.Shutdownable;
28  import org.archive.wayback.exception.ConfigurationException;
29  
30  /**
31   * Composite class containing a ResourceStore, and a ResourceIndex, to simplify
32   * sharing them as a pair across multiple AccessPoints.
33   *
34   * @author brad
35   * @version $Date$, $Revision$
36   */
37  public class WaybackCollection {
38  	private ResourceStore resourceStore = null;
39  	private ResourceIndex resourceIndex = null;
40  	private List<Shutdownable> shutdownables = null;
41  	private boolean shutdownDone = false;
42  
43  	/**
44  	 * close/release any resources held by this WaybackCollection
45  	 * @throws IOException when thrown by an internal class being shut down.
46  	 */
47  	public void shutdown() throws IOException {
48  		if(shutdownDone) {
49  			return;
50  		}
51  		if(resourceStore != null) {
52  			resourceStore.shutdown();
53  		}
54  		if(resourceIndex != null) {
55  			resourceIndex.shutdown();
56  		}
57  		if(shutdownables != null) {
58  			for(Shutdownable s : shutdownables) {
59  				s.shutdown();
60  			}
61  		}
62  		shutdownDone = true;
63  	}
64  
65  	/**
66  	 * @return the ResourceStore used with this WaybackCollection
67  	 * @throws ConfigurationException if none is configured
68  	 */
69  	public ResourceStore getResourceStore() throws ConfigurationException {
70  		if(resourceStore == null) {
71  			throw new ConfigurationException("No resourceStore declared");
72  		}
73  		return resourceStore;
74  	}
75  	/**
76  	 * @param resourceStore the ResourceStore to use with this WaybackCollection
77  	 */
78  	public void setResourceStore(ResourceStore resourceStore) {
79  		this.resourceStore = resourceStore;
80  	}
81  	/**
82  	 * @return the ResourceIndex used with this WaybackCollection
83  	 * @throws ConfigurationException if none is configured
84  	 */
85  	public ResourceIndex getResourceIndex() throws ConfigurationException {
86  		if(resourceIndex == null) {
87  			throw new ConfigurationException("No resourceIndex declared");
88  		}
89  		return resourceIndex;
90  	}
91  	/**
92  	 * @param resourceIndex the ResourceIndex to use with this WaybackCollection
93  	 */
94  	public void setResourceIndex(ResourceIndex resourceIndex) {
95  		this.resourceIndex = resourceIndex;
96  	}
97  
98  	/**
99  	 * @return List of Shutdownable objects associated with this 
100 	 * WaybackCollection, or null, if none are configured
101 	 */
102 	public List<Shutdownable> getShutdownables() {
103 		return shutdownables;
104 	}
105 
106 	/**
107 	 * @param shutdownables set a List of Shutdownable objects associated with
108 	 * this WaybackCollection
109 	 */
110 	public void setShutdownables(List<Shutdownable> shutdownables) {
111 		this.shutdownables = shutdownables;
112 	}
113 }
