37  /**
38   * {@code ServerRelativeArchivalRedirect} is a {@link RequestHandler}
39   * that redirects <i>leaked</i> server-relative URL back to replay request
40   * URL.
41   * <p>For example, assuming {@code Referer} is
42   * {@code http://web.archive.org/web/20010203040506/http://example.com/index.html},
43   * it redirects request {@code http://web.archive.org/js/foo.js}
44   * to {@code http://web.archive.org/web/20010203040506/http://example.com/js/foo.js}.</p>
45   * <p>It is typically set up as catch-all {@code RequestHandler}</p>
46   * @author brad
47   */
48  public class ServerRelativeArchivalRedirect extends AbstractRequestHandler {
49  	private static final Logger LOGGER = Logger
50  		.getLogger(ServerRelativeArchivalRedirect.class.getName());
51  
52  	boolean useCollection = false;
53  	private String matchHost = null;
54  	private int matchPort = -1;
55  	private String replayPrefix;
56  
57  	private String handleRequestWithCollection(HttpServletRequest httpRequest,
58  			HttpServletResponse httpResponse) throws ServletException,
59  			IOException {
60  		// TODO: check if this works with non-empty context path.
61  		// I believe requestURI starts with context path. So non-empty
62  		// context path will break collection and timestamp extraction
63  		// code below. It'd be desirable for this class to get a pattern
64  		// of replay URL from somewhere - replayPreifx?
65  
66  		// hope that it's a server relative request, with a valid referrer:
67  		String referer = httpRequest.getHeader("Referer");
68  		if (referer == null) return null;
69  		
70  		final UsableURI refuri = UsableURIFactory.getInstance(referer);
71  
72  		// Check that the Referer is our current wayback path
73  		// before attempting to use referer as base archival url
74  
75  		if ((matchHost != null && !matchHost.equals(refuri.getHost())) ||
76  				(matchPort != -1 && refuri.getPort() != -1 && matchPort != refuri
77  				.getPort())) {
78  			LOGGER.info("Server-Relative-Redirect: Skipping, Referer " +
79  					refuri.getHost() + ":" + refuri.getPort() +
80  					" not from matching wayback host:port\t");
81  			return null;
82  		}
83  
84  		String path = refuri.getPath();
85  		int secondSlash = path.indexOf('/', 1);
86  		if (secondSlash == -1) return null;
87  		
88  		String collection = path.substring(0, secondSlash);
89  		collection = modifyCollection(collection);
90  
91  		String remainder = path.substring(secondSlash + 1);
92  		int thirdSlash = remainder.indexOf('/');
93  		if (thirdSlash == -1) return null;
94  
95  		String datespec = remainder.substring(0, thirdSlash);
96  		if (!datespec.isEmpty() &&
97  				!Character.isDigit(datespec.charAt(0))) {
98  			datespec = null;
99  		}
100 
101 		String url = remainder.substring(thirdSlash + 1);
102 		url = UrlOperations.fixupScheme(url);
103 		url = ArchiveUtils.addImpliedHttpIfNecessary(url);
104 
105 		String thisPath = httpRequest.getRequestURI();
106 		String queryString = httpRequest.getQueryString();
107 		if (queryString != null) {
108 			thisPath += "?" + queryString;
109 		}
110 		String resolved = UrlOperations.resolveUrl(url, thisPath);
111 
112 		String contextPath = httpRequest.getContextPath();
113 		StringBuilder sb = new StringBuilder(refuri.getScheme());
114 		sb.append("://");
115 		sb.append(refuri.getAuthority());
116 		sb.append(contextPath);
117 		sb.append(collection);
118 		sb.append("/");
119 		if (datespec != null) {
120 			sb.append(datespec);
121 			sb.append("/");
122 		}
123 		sb.append(resolved);
124 		
125 		return sb.toString();
126 	}
127 
128 	/**
129 	 * modify collection if necessary.
130 	 * <p>default implementation simply return {@code collection}.</p>
131 	 * @param collection (the first path component of Referer URL.)
132 	 * note value has leading slash, which must be retained.
133 	 * @return possibly modified collection.
134 	 */
135 	protected String modifyCollection(String collection) {
136 		return collection;
137 	}

139 	private String handleRequestWithoutCollection(
140 			HttpServletRequest httpRequest, HttpServletResponse httpResponse)
141 			throws ServletException, IOException {
142 
143 		// hope that it's a server relative request, with a valid referrer:
144 		String referer = httpRequest.getHeader("Referer");
145 		if (referer == null) return null;
146 		
147 		LOGGER.fine("referer:" + referer);
148 		UsableURI uri = UsableURIFactory.getInstance(referer);
149 		String path = uri.getPath();
150 
151 		String remainder = path.substring(1);
152 		int thirdSlash = remainder.indexOf('/');
153 		LOGGER.fine("referer:(" + referer + ") remain(" + remainder +
154 			") 3rd(" + thirdSlash + ")");
155 		if (thirdSlash == -1) return null;
156 
157 		String datespec = remainder.substring(0, thirdSlash);
158 		String url = remainder.substring(thirdSlash + 1);
159 		url = UrlOperations.fixupScheme(url);
160 		url = ArchiveUtils.addImpliedHttpIfNecessary(url);
161 		String thisPath = httpRequest.getRequestURI();
162 		String queryString = httpRequest.getQueryString();
163 		if (queryString != null) {
164 			thisPath += "?" + queryString;
165 		}
166 
167 		String resolved = UrlOperations.resolveUrl(url, thisPath);
168 		String contextPath = httpRequest.getContextPath();
169 		String finalUrl = uri.getScheme() + "://" + uri.getAuthority() +
170 				contextPath + "/" + datespec + "/" + resolved;
171 		// cross your fingers!!!
172 		LOGGER.info("Server-Relative-Redirect:\t" + referer + "\t" +
173 				thisPath + "\t" + finalUrl);
174 		return finalUrl;
175 	}

177 	@Override
178 	public boolean handleRequest(HttpServletRequest httpRequest,
179 			HttpServletResponse httpResponse) throws ServletException,
180 			IOException {
181 		if (matchHost != null) {
182 			if (!matchHost.equals(httpRequest.getServerName())) {
183 				LOGGER.fine("Wrong host for ServerRelativeRed(" +
184 						httpRequest.getServerName() + ")");
185 				return false;
186 			}
187 		}
188 		if (matchPort != -1) {
189 			if (matchPort != httpRequest.getLocalPort()) {
190 				LOGGER.fine("Wrong port for ServerRealtiveRed(" +
191 						httpRequest.getServerName() + ")(" +
192 						httpRequest.getLocalPort() + ") :" +
193 						httpRequest.getRequestURI());
194 				return false;
195 			}
196 		}
197 		String replayUrl = (useCollection ?
198 				handleRequestWithCollection(httpRequest, httpResponse) :
199 					handleRequestWithoutCollection(httpRequest, httpResponse));
200 		if (replayUrl == null && replayPrefix != null) {
201 			String thisPath = httpRequest.getRequestURI();
202 			String queryString = httpRequest.getQueryString();
203 			if (queryString != null) {
204 				thisPath += "?" + queryString;
205 			}
206 			// TODO: rethink this fallback, for now adding https support as
207 			// well
208 			if (thisPath.startsWith("/http://") ||
209 					thisPath.startsWith("/https://")) {
210 				// assume a replay request:
211 				StringBuilder sb = new StringBuilder(thisPath.length() +
212 					replayPrefix.length() + 16);
213 				sb.append(replayPrefix);
214 				sb.append(Timestamp.currentTimestamp().getDateStr());
215 				sb.append(thisPath);
216 				replayUrl = sb.toString();
217 			}
218 		}
219 		if (replayUrl != null) {
220 			// Gotta make sure this is properly cached, or
221 			// weird things happen:
222 			httpResponse.addHeader("Vary", "Referer");
223 			httpResponse.sendRedirect(replayUrl);
224 			return true;
225 		}
226 		return false;
227 	}

236 	/**
237 	 * whether replay URL has <i>collection</i> part.
238 	 * <p>set this to {@code true} if replay URL has <i>collection</i>
239 	 * part, path component between context path and timestamp (although
240 	 * it's called <i>collection</i> based on the common usage of this part,
241 	 * there's no need to have particular semantics.)</p>
242 	 * <p>collection part will be passed to {@link #modifyCollection(String)}
243 	 * before constructing final replay URL to redirect to.<p>
244 	 * @param useCollection the useCollection to set
245 	 */
246 	public void setUseCollection(boolean useCollection) {
247 		this.useCollection = useCollection;
248 	}
