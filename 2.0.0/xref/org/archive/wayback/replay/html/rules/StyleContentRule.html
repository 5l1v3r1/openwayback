20  package org.archive.wayback.replay.html.rules;
21  
22  import java.io.IOException;
23  
24  import org.archive.wayback.replay.html.ReplayParseEventDelegator;
25  import org.archive.wayback.replay.html.ReplayParseEventDelegatorVisitor;
26  import org.archive.wayback.replay.html.ReplayParseContext;
27  import org.archive.wayback.replay.html.StringTransformer;
28  import org.archive.wayback.util.htmllex.ParseContext;
29  import org.archive.wayback.util.htmllex.handlers.CSSTextHandler;
30  import org.htmlparser.nodes.TextNode;
31  
32  public class StyleContentRule implements ReplayParseEventDelegatorVisitor, CSSTextHandler {
33  	private StringTransformer transformer;
34  
35  	public void visit(ReplayParseEventDelegator rules) {
36  		rules.getModifyDelegator().addCSSTextHandler(this);
37  	}
38  	public void handleCSSTextNode(ParseContext context, TextNode node)
39  	throws IOException {
40  		node.setText(transformer.transform((ReplayParseContext)context, node.getText()));
41  	}
42  	/**
43  	 * @return the transformer
44  	 */
45  	public StringTransformer getTransformer() {
46  		return transformer;
47  	}
48  
49  	/**
50  	 * @param transformer the transformer to set
51  	 */
52  	public void setTransformer(StringTransformer transformer) {
53  		this.transformer = transformer;
54  	}
55  }
