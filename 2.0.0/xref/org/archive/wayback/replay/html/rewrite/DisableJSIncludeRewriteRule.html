5   public class DisableJSIncludeRewriteRule extends RewriteRule {
6   	
7   	@Override
8   	public String rewrite(ReplayParseContext context, String policy,
9   			String inputSrc) {
10  		
11  		int index = policy.indexOf('=', getBeanName().length());
12  		
13  		if (index >= 0) {
14  			String filename = policy.substring(index + 1);
15  			
16  			// If inputSrc contains filename, exclude it!
17  			if ((inputSrc != null) && inputSrc.indexOf(filename) >= 0) {
18  				return null;
19  			}
20  		}
21  		
22  		return inputSrc;
23  	}
24  }
