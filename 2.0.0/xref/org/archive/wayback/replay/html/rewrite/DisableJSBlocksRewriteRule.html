3   import java.util.StringTokenizer;
4   import java.util.logging.Logger;
5   
6   import org.archive.wayback.replay.html.ReplayParseContext;
7   
8   public class DisableJSBlocksRewriteRule extends RewriteRule {
9   	
10  	private static final Logger LOGGER = Logger.getLogger(
11  			DisableJSBlocksRewriteRule.class.getName());
12  	
13  	protected String removeMsg = "/* Script Removed for Wayback Machine replay */\n";
14  	
15  	public String getRemoveMsg() {
16  		return removeMsg;
17  	}
18  
19  	public void setRemoveMsg(String removeMsg) {
20  		this.removeMsg = removeMsg;
21  	}
22  
23  	public String rewrite(ReplayParseContext context, String policy, String input) {
24  		
25  		boolean rewrite = false;		
26  		String scriptIndexStr = String.valueOf(context.getJSBlockCount());
27  		
28  		// Test against all params specified, ex. disable-script(0,1)
29  		// will test to see if script block is 0 or 1 and if so, perform rewrite
30  		int index = policy.indexOf('[', getBeanName().length());
31  		
32  		if (index >= 0) {
33  			StringTokenizer tokens = new StringTokenizer(policy.substring(index + 1), " ]");
34  			while (tokens.hasMoreTokens()) {
35  				if (tokens.nextToken().equals(scriptIndexStr)) {
36  					rewrite = true;
37  					break;
38  				}
39  			}
40  		} else {
41  			//TODO: Do nothing? Disable first script block?
42  		}
43  				
44  		if (rewrite) {
45  			LOGGER.info(policy + ": disable script block " + scriptIndexStr + ", replacing it with \"" + removeMsg + "\"");		
46  			input = removeMsg;
47  		}
48  		
49  		return input;
50  	}
51  }
