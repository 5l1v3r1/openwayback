20  package org.archive.wayback.replay;
21  
22  import java.util.HashSet;
23  import java.util.Map;
24  import java.util.Set;
25  
26  import org.archive.wayback.ResultURIConverter;
27  import org.archive.wayback.core.CaptureSearchResult;
28  
29  /**
30   * {@link HttpHeaderProcessor} that renames all headers by prepending a prefix,
31   * except for a few <em>pass-through</em> headers.
32   * <p>
33   * Headers copied as-is:
34   * <ul>
35   * <li>{@code Content-Type}</li>
36   * <li>{@code Content-Disposition}</li>
37   * </ul>
38   * Headers dropped if {@code prefix} is set to {@code null} or empty:
39   * <ul>
40   * <li>{@code Transfer-Encoding}</li>
41   * </ul>
42   * </p>
43   * <p>This is only useful for proxy mode, because it does not translate URLs found
44   * in headers like {@code Location}.</p>
45   * @see RedirectRewritingHttpHeaderProcessor
46   */
47  public class XArchiveHttpHeaderProcessor extends PreservingHttpHeaderProcessor {
48  
49  	private static String DEFAULT_PREFIX = "X-Wayback-Orig-";
50  	private Set<String> passThrough = null;
51  	
52  	public XArchiveHttpHeaderProcessor() {
53  		passThrough = new HashSet<String>();
54  		passThrough.add(HTTP_CONTENT_TYPE_HEADER_UP);
55  		passThrough.add(HTTP_CONTENT_DISP_HEADER_UP);
56  
57  		prefix = DEFAULT_PREFIX;
58  	}
59  	
60  	public void filter(Map<String, String> output, String key, String value,
61  			ResultURIConverter uriConverter, CaptureSearchResult result) {
62  		String keyUp = key.toUpperCase();
63  
64  		if (key.equalsIgnoreCase(HTTP_TRANSFER_ENCODING_HEADER_UP))
65  			preserve(output, key, value);
66  		else
67  			preserveAlways(output, key, value);
68  		if (passThrough.contains(keyUp)) {
69  			// add this one as-is, too.
70  			output.put(key, value);
71  		}
72  	}
73  }
