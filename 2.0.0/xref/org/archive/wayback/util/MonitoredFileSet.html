1   package org.archive.wayback.util;
2   
3   import java.io.File;
4   import java.util.Date;
5   import java.util.HashMap;
6   import java.util.List;
7   
8   public class MonitoredFileSet {
9   	List<String> files;
10  	
11  	public MonitoredFileSet(List<String> files) {
12  		this.files = files;
13  	}
14  	public boolean isChanged(FileState fileState) {
15  		FileState currentFileState = getFileState();
16  		return currentFileState.isChanged(fileState);
17  	}
18  	public FileState getFileState() {
19  		FileState fileState = new FileState();
20  		
21  		for(String path : files) {
22  			File file = new File(path);
23  			if(file.isFile()) {
24  				fileState.put(path, new Date(file.lastModified()));
25  			} else {
26  				fileState.put(path, null);
27  			}
28  		}
29  		return fileState;
30  	}
31  	
32  	public class FileState extends HashMap<String,Date> {
33  		public boolean isChanged(FileState other) {
34  			for(String path : keySet()) {
35  				if(other.containsKey(path)) {
36  					Date otherDate = other.get(path);
37  					Date thisDate = get(path);
38  					if((otherDate == null) && (thisDate == null)) {
39  						// treat both missing as the same..
40  						continue;
41  					}
42  					if(!otherDate.equals(thisDate)) {
43  						return true;
44  					}
45  				}
46  			}
47  			return false;
48  		}
49  	}
50  }
