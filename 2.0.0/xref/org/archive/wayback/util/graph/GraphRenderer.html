20  package org.archive.wayback.util.graph;
21  
22  import java.awt.Graphics2D;
23  import java.awt.Rectangle;
24  import java.awt.image.BufferedImage;
25  import java.io.IOException;
26  import java.io.OutputStream;
27  
28  import javax.imageio.ImageIO;
29  
30  /**
31   * @author brad
32   *
33   */
34  public class GraphRenderer {
35  	/**
36  	 * appropriate Content-Type HTTP header value for graph image content
37  	 * produced by render(OutputStream,Graph)
38  	 */
39  	public final static String RENDERED_IMAGE_MIME = "image/png"; 
40  	/**
41  	 * Create both an HTML AREA map and an HTML IMG for a graph, using provided
42  	 * href targets, and titles within the AREA map.
43  	 * @param graph Graph to draw
44  	 * @param mapName name of HTML AREA map
45  	 * @param imgUrl URL to rendered Graph, see GraphEncoder.encode()
46  	 * @param targets URL href targets for the years in the Graph
47  	 * @param titles titles for the years in the graph.
48  	 * @return HTML String for the resulting AREA and IMG
49  	 */
50  	public static String renderHTML(Graph graph, String mapName, String imgUrl, 
51  			String targets[], String titles[]) {
52  
53  		StringBuilder sb = new StringBuilder();
54  		sb.append("<map name=\"").append(mapName).append("\">");
55  		RegionGraphElement rge[] = graph.getRegions();
56  		int count = rge.length;
57  		for(int i = 0; i < count; i++) {
58  			if(targets[i] != null) {
59  				Rectangle r = rge[i].getBoundingRectangle();
60  				sb.append("<area href=\"").append(targets[i]).append("\"");
61  				if(titles[i] != null) {
62  					sb.append(" title=\"").append(titles[i]).append("\"");
63  				}
64  				sb.append(" shape=\"rect\" coords=\"");
65  				sb.append(r.x).append(",");
66  				sb.append(r.y).append(",");
67  				sb.append(r.x+r.width).append(",");
68  				sb.append(r.y+r.height).append("\" border=\"1\" />");
69  			}
70  		}
71  		sb.append("</map>");
72  		sb.append("<image src=\"").append(imgUrl).append("\"");
73  		sb.append(" border=\"0\" width=\"").append(graph.width).append("\"");
74  		sb.append(" height=\"").append(graph.height).append("\"");
75  		sb.append(" usemap=\"#").append(mapName).append("\" />");
76  
77  		return sb.toString();
78  	}
79  		
80  	/**
81  	 * Send a PNG format byte stream for the argument Graph to the provided
82  	 * OutputStream  
83  	 * @param target OutputStream to write PNG format bytes
84  	 * @param graph Graph to send to the target
85  	 * @throws IOException for usual reasons.
86  	 */
87  	public void render(OutputStream target, Graph graph) throws IOException {
88  		
89  		BufferedImage bi = 
90  			new BufferedImage(graph.width, graph.height, 
91  					GraphConfiguration.imageType);
92  		Graphics2D g2d = bi.createGraphics();
93  		graph.draw(g2d);
94  	    ImageIO.write(bi, "png", target);
95  	}
96  
97  }
