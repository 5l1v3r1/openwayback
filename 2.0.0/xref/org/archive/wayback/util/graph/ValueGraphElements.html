20  package org.archive.wayback.util.graph;
21  
22  import java.awt.Graphics2D;
23  import java.util.NoSuchElementException;
24  
25  /**
26   * @author brad
27   *
28   */
29  public class ValueGraphElements extends RectangularGraphElement {
30  	private int values[] = null;
31  	private int highlightValue = -1;
32  	private int maxValue = -1;
33  	private GraphConfiguration config = null;
34  
35  	/**
36  	 * @param x the rectangles x in the global coordinate space
37  	 * @param y the rectangles y in the global coordinate space
38  	 * @param width the rectangles width
39  	 * @param height the rectangles height
40  	 * @param hightlightValue the index of the value to highlight 
41  	 * @param values array of int values, each must be 0-15
42  	 * @param maxValue the global maximum value across all elements in the Graph
43  	 * @param config the configuration for the Graph
44  	 */
45  	public ValueGraphElements(int x, int y, int width, int height, 
46  			int hightlightValue, int values[], int maxValue,
47  			GraphConfiguration config) {
48  
49  		super(x, y, width, height);
50  //		System.err.format("Created VGEs (%d,%d)-(%d,%d)\n",x,y,width,height);
51  
52  		this.highlightValue = hightlightValue;
53  		this.values = values;
54  		this.config = config;
55  		this.maxValue = maxValue;
56  	}
57  
58  	/**
59  	 * return the i'th ValueGraphElement
60  	 * @param i the index of the element to return
61  	 * @return the ValueGraphElement at index i
62  	 */
63  	public ValueGraphElement getElement(int i) {
64  		if((i < 0) || (i >= values.length)) {
65  			throw new NoSuchElementException();
66  		}
67  		int minHeight = config.valueMinHeight;
68  		
69  		// normalize height to value between 0 and 1:
70  		float value = ((float) values[i]) / ((float) maxValue);
71  		float usableHeight = height - minHeight;
72  		int valueHeight = (int) (usableHeight * value) + minHeight;
73  		
74  		int elX = Graph.xlateX(width, values.length, i);
75  		int elW = Graph.xlateX(width, values.length, i+1) - elX;
76  		int elY = height - valueHeight;
77  		boolean hot = i == highlightValue;
78  		return new ValueGraphElement(x + elX, y + elY, elW, valueHeight, hot, config);
79  	}
80  	
81  	public void draw(Graphics2D g2d) {
82  		for(int i = 0; i < values.length; i++) {
83  			if(values[i] > 0) {
84  				getElement(i).draw(g2d);
85  			}
86  		}
87  	}
88  
89  	/**
90  	 * @return the index of the highlighted value
91  	 */
92  	public int getHighlightValue() {
93  		return highlightValue;
94  	}
95  
96  	/**
97  	 * @return the raw int values for the graph
98  	 */
99  	public int[] getRawValues() {
100 		return values;
101 	}
102 	
103 }
