123     private static final Pattern AUTHORITY_REGEX =
124         Pattern.compile("(([0-9a-z_.-]+)\\.(" + ALL_TLDS + "))|" +
125         		"(" + IP_PATTERN + ")");

132         Pattern.compile("^([0-9a-z_.:-]+)(?:@[0-9a-z_.-]++)");

134     /**
135      * Tests if the String argument looks like it could be a legitimate 
136      * authority fragment of a URL, that is, is it an IP address, or, are the
137      * characters legal in an authority, and does the string end with a legal
138      * TLD.
139      * 
140 	 * @param authString String representation of a fragment of a URL 
141 	 * @return boolean indicating whether urlPart might be an Authority.
142 	 */
143 	public static boolean isAuthority(String authString) {
144 		Matcher m = AUTHORITY_REGEX.matcher(authString);
145 		
146 		return (m != null) && m.matches();
147 	}

149 	/** Resolve URL, but return a minimally escaped version in case of
150 	 *  error
151 	 * @param baseUrl
152 	 * @param url
153 	 * @return
154 	 */
155 	
156 	public static String resolveUrl(String baseUrl, String url) {
157 		String resolvedUrl = resolveUrl(baseUrl, url, null);
158 		if (resolvedUrl == null) {
159 			resolvedUrl = url.replace(" ", "%20");
160 			resolvedUrl = resolvedUrl.replace("\r", "%0D");
161 		}
162 		return resolvedUrl;
163 	}

165 	/**
166 	 * Resolve a possibly relative url argument against a base URL.
167 	 * @param baseUrl the base URL against which the url should be resolved
168 	 * @param url the URL, possibly relative, to make absolute.
169 	 * @return url resolved against baseUrl, unless it is absolute already, and
170 	 * further transformed by whatever escaping normally takes place with a 
171 	 * UsableURI.
172 	 * In case of error, return the defaultValue
173 	 */
174 	public static String resolveUrl(String baseUrl, String url, String defaultValue) {
175 				
176 		for(final String scheme : ALL_SCHEMES) {
177 			if(url.startsWith(scheme)) {
178 				try {
179 					return UsableURIFactory.getInstance(url).getEscapedURI();
180 				} catch (URIException e) {
181 					LOGGER.warning(e.getLocalizedMessage() + ": " + url);
182 					// can't let a space exist... send back close to whatever came
183 					// in...
184 					return defaultValue;
185 				}
186 			}
187 		}
188 		UsableURI absBaseURI;
189 		UsableURI resolvedURI = null;
190 		try {
191 			absBaseURI = UsableURIFactory.getInstance(baseUrl);
192 			resolvedURI = UsableURIFactory.getInstance(absBaseURI, url);
193 		} catch (URIException e) {
194 			LOGGER.warning(e.getLocalizedMessage() + ": " + url);
195 			return defaultValue;
196 		}
197 		
198 		return resolvedURI.getEscapedURI();
199 	}

201 	/**
202 	 * Attempt to find the scheme (http://, https://, etc) from a given URL.
203 	 * @param url URL String to parse for a scheme.
204 	 * @return the scheme, including trailing "://" if known, null otherwise.
205 	 */
206 	public static String urlToScheme(final String url) {
207 		for(final String scheme : ALL_SCHEMES) {
208 			if(url.startsWith(scheme)) {
209 				return scheme;
210 			}
211 		}
212 		return null;
213 	}

215 	/**
216 	 * Return the default port for the scheme String argument, if known.
217 	 * @param scheme String scheme, including '://', as in, "http://", "ftp://"
218 	 * @return the default port for the scheme, or -1 if the scheme isn't known.
219 	 */
220 	public static int schemeToDefaultPort(final String scheme) {
221 		if(scheme.equals(HTTP_SCHEME)) {
222 			return 80;
223 		}
224 		if(scheme.equals(HTTPS_SCHEME)) {
225 			return 443;
226 		}
227 		if(scheme.equals(FTP_SCHEME)) {
228 			return 21;
229 		}
230 		if(scheme.equals(RTSP_SCHEME)) {
231 			return 554;
232 		}
233 		if(scheme.equals(MMS_SCHEME)) {
234 			return 1755;
235 		}
236 		return -1;
237 	}

240 	 * Attempt to extract the path component of a url String argument.
241 	 * @param url the URL which may contain a path, sans scheme.
242 	 * @return the path component of the URL, or "" if it contains no path.
243 	 */
244 	public static String getURLPath(String url) {
245 		url = stripURLScheme(url);
246 		int pathIdx = url.indexOf(UrlOperations.PATH_START);
247 		if(pathIdx == -1) {
248 			return "/";
249 		}
250 		return url.substring(pathIdx);
251 	}
252 	/**
253 	 * Attempt to extract the path component of a url String argument.
254 	 * @param url the URL which may contain a path, sans scheme.
255 	 * @return the path component of the URL, or "" if it contains no path.
256 	 */
257 	public static String stripURLScheme(String url) {
258 		String lcUrl = url.toLowerCase();
259 		for(String scheme : ALL_SCHEMES) {
260 			if(lcUrl.startsWith(scheme)) {
261 				return url.substring(scheme.length());
262 			}
263 		}
264 		return url;
265 	}	
266 	/**
267 	 * Attempt to strip default ports out of URL strings.
268 	 * @param url the original URL possibly including a port
269 	 * @return the URL sans port, if the scheme was recognized and the default
270 	 * port was supplied, otherwise, the original URL.
271 	 */
272 	public static String stripDefaultPortFromUrl(String url) {
273 		String scheme = urlToScheme(url);
274 		if(scheme == null) {
275 			return url;
276 		}
277 		int defaultPort = schemeToDefaultPort(scheme);
278 		if(defaultPort == -1) {
279 			return url;
280 		}
281 
282 		String portStr = null;
283 		// is there a slash after the scheme?
284 		int slashIdx = url.indexOf('/', scheme.length());
285 		if(slashIdx == -1) {
286 			portStr = String.format(":%d", defaultPort);
287 			if(url.endsWith(portStr)) {
288 				return url.substring(0,url.length() - portStr.length());
289 			}
290 		}
291 		portStr = String.format(":%d/", defaultPort);
292 		int idx = url.indexOf(portStr);
293 		if(idx == -1) {
294 			return url;
295 		}
296 		// if that occurred before the first / (after the scheme) then strip it:
297 		if(slashIdx < idx) {
298 			return url;
299 		}
300 		// we want to strip out the portStr:
301 		StringBuilder sb = new StringBuilder(url.length());
302 		sb.append(url.substring(0,idx));
303 		sb.append(url.substring(idx + (portStr.length()-1)));
304 		return sb.toString();
305 	}

308 	 * @param orig String containing a URL, possibly beginning with "http:/".
309 	 * @return original string if orig begins with "http://", or a new String
310 	 * with the extra slash, if orig only had one slash.
311 	 * @see #fixupScheme
312 	 */
313 	public static String fixupHTTPUrlWithOneSlash(String orig) {
314 		if(orig.startsWith("http:/") && ! orig.startsWith(HTTP_SCHEME)) {
315 			// very likely the IE "you must have meant 1 slash, not 2 bug:
316 			StringBuilder sb = new StringBuilder(orig.length()+1);
317 			sb.append(HTTP_SCHEME);
318 			return sb.append(orig.substring(6)).toString();
319 		}
320 		return orig;
321 	}

323 	/**
324 	 * fixes up malformed scheme part.
325 	 * <p>currently supports fixing missing second slash for protocols
326 	 * {@code http}, {@code https}, {@code ftp}, {@code rtsp} and
327 	 * {@code mms}. For example fixing {@code http:/} to {@code https://}</p>
328 	 * @param url URL to be checked and fixed
329 	 * @return new String, or {@code url} if not fix is required.
330 	 * @version 1.8.1
331 	 */
332 	public static String fixupScheme(String url) {
333 		final String[] SCHEMES = {
334 			"http:/", "https:/", "ftp:/", "rtsp:/", "mms:/"
335 		};
336 		int ul = url.length();
337 		for (String scheme : SCHEMES) {
338 			int sl = scheme.length();
339 			if (url.startsWith(scheme) && (ul == sl || url.charAt(sl) != '/')) {
340 				return scheme + "/" + url.substring(sl);
341 			}
342 		}
343 		return url;
344 	}

346 	/**
347 	 * Attempt to extract the hostname component of an absolute URL argument.
348 	 * @param url the url String from which to extract the hostname
349 	 * @return the hostname within the URL, or the url argument if the host 
350 	 * cannot be found.
351 	 */
352 	public static String urlToHost(String url) {
353 		String lcUrl = url.toLowerCase();
354 		if(lcUrl.startsWith(DNS_SCHEME)) {
355 			return lcUrl.substring(DNS_SCHEME.length());
356 		}
357 		for(String scheme : ALL_SCHEMES) {
358 			if(lcUrl.startsWith(scheme)) {
359 				int authorityIdx = scheme.length();
360 
361 				Matcher m = 
362 					HOST_REGEX_SIMPLE.matcher(lcUrl.substring(authorityIdx));
363 				if(m.find()) {
364 					return m.group(1);
365 				}
366 			}
367 		}
368 		return url;
369 	}

371 	/**
372 	 * Extract userinfo from the absolute URL argument, that is, "username@", or
373 	 * "username:password@" if present.
374 	 * @param url the URL from which to extract the userinfo
375 	 * @return the userinfo found, not including the "@", or null if no userinfo
376 	 * is found
377 	 */
378 	public static String urlToUserInfo(String url) {
379 		String lcUrl = url.toLowerCase();
380 		if(lcUrl.startsWith(DNS_SCHEME)) {
381 			return null;
382 		}
383 		for(String scheme : ALL_SCHEMES) {
384 			if(lcUrl.startsWith(scheme)) {
385 				int authorityIdx = scheme.length();
386 
387 				Matcher m = 
388 					USERINFO_REGEX_SIMPLE.matcher(lcUrl.substring(authorityIdx));
389 				if(m.find()) {
390 					return m.group(1);
391 				}
392 			}
393 		}
394 		return null;
395 	}

397 	/**
398 	 * Find and return the parent directory of the URL argument
399 	 * @param url to find the parent directory of
400 	 * @return parent directory of URL, or null, if either the url argument is
401 	 * invalid, or if the url is the root of the authority.
402 	 */
403 	public static String getUrlParentDir(String url) {
404 		
405 		try {
406 			UsableURI uri = UsableURIFactory.getInstance(url);
407 			String path = uri.getPath();
408 			if(path.length() > 1) {
409 				int startIdx = path.length()-1;
410 				if(path.charAt(path.length()-1) == '/') {
411 					startIdx--;
412 				}
413 				int idx = path.lastIndexOf('/',startIdx);
414 				if(idx >= 0) {
415 					uri.setPath(path.substring(0,idx+1));
416 					uri.setQuery(null);
417 					return uri.toString();
418 				}
419 			}
420 		} catch (URIException e) {
421 			LOGGER.warning(e.getLocalizedMessage() + ": " + url);
422 		}
423 		return null;
424 	}

426 	public static String computeIdentityUrl(WaybackRequest wbRequest)
427 	{
428 		AccessPoint accessPoint = wbRequest.getAccessPoint();
429 
430 		boolean origIdentity = wbRequest.isIdentityContext();
431 		wbRequest.setIdentityContext(true);
432 		
433 		ArchivalUrl aUrl = new ArchivalUrl(wbRequest);
434 		String bestPath = aUrl.toString();
435 		String betterURI = accessPoint.getReplayPrefix() + bestPath;
436 
437 		//reset the isIdentity flag just in case
438 		wbRequest.setIdentityContext(origIdentity);
439 		
440 		return betterURI;
441 	}
