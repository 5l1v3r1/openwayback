9   public class CDXFlexFormat extends CDXFormat {
10  	private final static String SCHEME_STRING = "://";
11  	private final static String DEFAULT_SCHEME = "http://";
12  	private final static String EMPTY_VALUE = "-";
13  
14  	private static final Logger LOGGER = 
15  		Logger.getLogger(CDXFlexFormat.class.getName());
16  	public CDXFlexFormat(String cdxSpec) throws CDXFormatException {
17  		super(cdxSpec);

19  	private static int getEndOfHostIndex(String url) {
20  		int portIdx = url.indexOf(UrlOperations.PORT_SEPARATOR);
21  		int pathIdx = url.indexOf(UrlOperations.PATH_START);
22  		if(portIdx == -1 && pathIdx == -1) {
23  			return url.length();
24  		}
25  		if(portIdx == -1) {
26  			return pathIdx;
27  		}
28  		if(pathIdx == -1) {
29  			return portIdx;
30  		}
31  		if(pathIdx > portIdx) {
32  			return portIdx;
33  		} else {
34  			return pathIdx;
35  		}
36  	}

50  	public static CaptureSearchResult parseCDXLineFlex(String line, CaptureSearchResult result) {		
51  		String[] tokens = line.split(" ");
52  		boolean hasRobotFlags = false;
53  		boolean hasLengthFlag = false;
54  		if (tokens.length != 9) {
55  			hasRobotFlags = true;
56  			if(tokens.length == 10) {
57  			} else if(tokens.length == 11) {
58  				hasLengthFlag = true;
59  			} else {
60  				return null;
61  			}
62  			//throw new IllegalArgumentException("Need 9 columns("+line+")");
63  		}
64  		String urlKey = tokens[0];
65  		String captureTS = tokens[1];
66  		String originalUrl = tokens[2];
67  		
68  		// convert from ORIG_HOST to ORIG_URL here:
69  		if(!originalUrl.contains(SCHEME_STRING)) {
70  			StringBuilder sb = new StringBuilder(urlKey.length());
71  			sb.append(DEFAULT_SCHEME);
72  			sb.append(originalUrl);
73  			sb.append(urlKey.substring(getEndOfHostIndex(urlKey)));
74  			originalUrl = sb.toString();
75  		}
76  		String mimeType = tokens[3];
77  		String httpCode = tokens[4];
78  		String digest = tokens[5];
79  		String redirectUrl = tokens[6];
80  		long compressedOffset = -1;
81  		int nextToken = 7;
82  		if(hasRobotFlags) {
83  			result.setRobotFlags(tokens[nextToken]);
84  			nextToken++;
85  		}
86  		String length = EMPTY_VALUE;
87  		if(hasLengthFlag) {
88  			length = tokens[nextToken];
89  			nextToken++;
90  		}
91  
92  		if(!tokens[nextToken].equals(EMPTY_VALUE)) {
93  			try {
94  				compressedOffset = Long.parseLong(tokens[nextToken]);
95  				if(!length.equals(EMPTY_VALUE)) {
96  					// try to set the endOffset:
97  					result.setCompressedLength(Long.parseLong(length));
98  				}
99  			} catch (NumberFormatException e) {
100 				LOGGER.warning("Bad compressed Offset field("+nextToken+") in (" +
101 						line +")");
102 				return null;
103 			}
104 		}
105 		nextToken++;
106 		String fileName = tokens[nextToken];
107 		result.setUrlKey(urlKey);
108 		result.setCaptureTimestamp(captureTS);
109 		result.setOriginalUrl(originalUrl);
110 		result.setMimeType(mimeType);
111 		result.setHttpCode(httpCode);
112 		result.setDigest(digest);
113 		result.setRedirectUrl(redirectUrl);
114 		result.setOffset(compressedOffset);
115 		result.setFile(fileName.trim());
116 
117 		return result;
118 	}
