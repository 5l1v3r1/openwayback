95  	public void init() throws ConfigurationException {
96  		LOGGER.info("initializing NutchResourceIndex...");
97  		LOGGER.info("Using base search url " + this.searchUrlBase);
98  
99  		this.factory.setNamespaceAware(true);
100 		try {
101 			this.builder = this.factory.newDocumentBuilder();
102 		} catch (ParserConfigurationException e) {
103 			// TODO: quiet extra stacktrace..
104 			e.printStackTrace();
105 			throw new ConfigurationException(e.getMessage());
106 		}

108 	/* (non-Javadoc)
109 	 * @see org.archive.wayback.ResourceIndex#query(org.archive.wayback.core.WaybackRequest)
110 	 */
111 	public SearchResults query(WaybackRequest wbRequest) 
112 		throws ResourceIndexNotAvailableException,
113 		ResourceNotInArchiveException, BadQueryException, 
114 		AccessControlException {
115 
116 		// Get the URL for the request:
117 		String requestUrl = getRequestUrl(wbRequest);
118 		Document document = null;
119 		try {
120 			// HTTP Request + parse
121 			LOGGER.info("Requesting OpenSearch: " + requestUrl);
122 			document =  getHttpDocument(requestUrl);
123 		} catch (IOException e) {
124 			// TODO: better error for user:
125 			e.printStackTrace();
126 			throw new ResourceIndexNotAvailableException(e.getMessage());
127 		} catch (SAXException e) {
128 			e.printStackTrace();
129 			throw new ResourceIndexNotAvailableException("Unexpected SAX: " + 
130 					e.getMessage());
131 		}
132 
133 		CaptureSearchResults results;
134 		if(wbRequest.isReplayRequest() || wbRequest.isCaptureQueryRequest()) {
135 			results = new CaptureSearchResults();			
136 		} else {
137 			// TODO: this is wrong, but needs exploration into what NutchWax 
138 			//       can actually do.
139 			throw new BadQueryException("Unable to perform path " +
140 					"prefix requests with this index type");
141 		}
142 		NodeList channel = getSearchChannel(document);
143 		NodeList nodes = getSearchItems(document);
144 
145 		if (channel == null || channel.getLength() != 1) {
146 			// TODO: better error for user:
147        	throw new ResourceNotInArchiveException("No results for " +
148        			requestUrl);
149        }
150 
151        if (nodes == null) {
152 			// TODO: better error for user:
153        	throw new ResourceNotInArchiveException("No results for " +
154        			requestUrl);
155        }
156 
157        for (int i = 0; i < nodes.getLength(); i++) {
158        	
159            Element e = (Element) nodes.item(i);
160 
161            List<CaptureSearchResult> resultsList = itemToSearchResults(e);
162            if(resultsList != null) {
163         	   for(CaptureSearchResult result : resultsList) {
164         		   results.addSearchResult(result);
165         	   }
166            }
167        }
168        Element channelElement = (Element) channel.item(0);
169        
170        results.putFilter(SearchResults.RESULTS_FIRST_RETURNED,
171        		getNodeContent(channelElement,NUTCH_FIRST_RESULT));
172        
173        results.putFilter(SearchResults.RESULTS_NUM_RESULTS,
174        		getNodeContent(channelElement,NUTCH_NUM_RESULTS));
175        
176        results.putFilter(SearchResults.RESULTS_NUM_RETURNED,
177        		getNodeContent(channelElement,NUTCH_NUM_RETURNED));
178        
179        results.putFilter(SearchResults.RESULTS_REQUESTED,
180        		String.valueOf(wbRequest.getResultsPerPage()));
181        
182 		results.putFilter(WaybackRequest.REQUEST_START_DATE,
183 				Timestamp.earliestTimestamp().getDateStr());
184 		
185        results.putFilter(WaybackRequest.REQUEST_END_DATE,
186        		Timestamp.latestTimestamp().getDateStr());
187 		return results;
188 	}
189 
190 	private List<CaptureSearchResult> itemToSearchResults(Element e)
191 		throws ResourceIndexNotAvailableException {
192 
193 		String fileName = getNodeNutchContent(e,NUTCH_FILENAME);
194 		String httpCode = NUTCH_DEFAULT_HTTP_CODE;
195 		String digest = getNodeNutchContent(e,NUTCH_DIGEST);
196 		String mimeType = getNodeNutchContent(e,NUTCH_MIME_TYPE);
197 		String offsetStr = getNodeNutchContent(e,NUTCH_FILEOFFSET);
198 		long offset = 0;
199 		if(offsetStr != null && offsetStr.length() > 0) {
200 			offset = Long.parseLong(offsetStr);
201 		}
202 		String redirectUrl = NUTCH_DEFAULT_REDIRECT_URL;
203 		String originalUrl = getNodeContent(e,NUTCH_CAPTURE_URL);
204 		String urlKey = originalUrl;
205 		
206 		NodeList nodes = e.getElementsByTagNameNS(NUTCH_NS,NUTCH_ARCDATE);
207 		int numDates = nodes.getLength();
208 		ArrayList<CaptureSearchResult> results = null;
209 
210 		if(numDates > 0) {
211 			results = new ArrayList<CaptureSearchResult>();
212 		
213 			for(int i = 0; i < numDates; i++) {
214 				String captureDate = getNodeTextValue(nodes.item(i));
215 				CaptureSearchResult result = new CaptureSearchResult();
216 				result.setFile(fileName);
217 				result.setCaptureTimestamp(captureDate);
218 				result.setHttpCode(httpCode);
219 				result.setDigest(digest);
220 				result.setMimeType(mimeType);
221 				result.setOffset(offset);
222 				result.setRedirectUrl(redirectUrl);
223 				result.setOriginalUrl(originalUrl);
224 				result.setUrlKey(urlKey);
225 				results.add(result);
226 			}
227 		}
228 		return results;
229 	}

231    protected NodeList getSearchChannel(Document d) {
232        if (d ==  null) {
233            return null;
234        }
235        // Jump to the search item list.
236        NodeList nodes = d.getElementsByTagName(NUTCH_SEARCH_RESULTS_TAG);
237        return (nodes.getLength() <= 0)? null: nodes;
238    }
239    
240    protected NodeList getSearchItems(Document d) {
241        if (d ==  null) {
242            return null;
243        }
244        // Jump to the search item list.
245        NodeList nodes = d.getElementsByTagName(NUTCH_SEARCH_RESULT_TAG);
246        return (nodes.getLength() <= 0)? null: nodes;
247    }

250    protected String getRequestUrl(WaybackRequest wbRequest) 
251    throws BadQueryException {
252 
253 	   String urlStr = wbRequest.getRequestUrl();
254 	   String exactDateStr = wbRequest.getReplayTimestamp();
255 	    if (exactDateStr != null && exactDateStr.length() == 0) {
256 	        exactDateStr = null;
257 	    }
258 	   	String endDateStr = wbRequest.getEndTimestamp();
259 	   	if (endDateStr == null || endDateStr.length() == 0) {
260 	   	   	endDateStr = Timestamp.latestTimestamp().getDateStr();
261 	   	}
262 	   	String startDateStr = wbRequest.getStartTimestamp();
263 	   	if (startDateStr == null || startDateStr.length() == 0) {
264 	   		startDateStr = Timestamp.earliestTimestamp().getDateStr();
265 	   	}
266 	   	int hitsPerPage = wbRequest.getResultsPerPage();
267 	   	if(hitsPerPage < 1) {
268 	   		throw new BadQueryException("Hits per page must be positive");
269 	   	}
270 	   	if(hitsPerPage > maxRecords) {
271 	   		throw new BadQueryException("Hits per page must be less than " +
272 	   				maxRecords);
273 	   	}
274 	   	int start = (wbRequest.getPageNum()-1) * hitsPerPage;
275        if (urlStr == null || urlStr.length() <= 0) {
276            throw new BadQueryException("Url is empty.");
277        }
278        // Construct the search url.
279        MutableString ms = new MutableString(this.searchUrlBase)
280            .append("?query=");
281        // Add 'date:...+' to query string.
282        ms.append("date%3A").append(startDateStr).append('-').append(endDateStr);
283        ms.append('+');
284        // Add 'url:URL'.
285        if(wbRequest.isUrlQueryRequest()) {
286            ms.append("url%3A");
287        } else {
288            ms.append("exacturl%3A");
289        }
290        try {
291             ms.append(java.net.URLEncoder.encode("\""+urlStr+"\"", "UTF-8"));
292        } catch (UnsupportedEncodingException e) {
293     	   throw new BadQueryException(e.toString());
294        }
295        ms.append("&hitsPerPage=").append(hitsPerPage);
296        ms.append("&start=").append(start);
297        ms.append("&dedupField=site");
298        // As we are always searching agains an url, a
299        // higher perDup/Site will return just more versions
300        ms.append("&hitsPerDup=").append(hitsPerPage);
301        ms.append("&hitsPerSite=").append(hitsPerPage);
302        
303        return ms.toString();
304    }

307 	// extract the text content of a single nutch: tag under a node
308    protected String getNodeNutchContent(Element e, String key) {
309        NodeList nodes = e.getElementsByTagNameNS(NUTCH_NS, key);
310        String result = null;
311        if (nodes != null && nodes.getLength() > 0) {
312            result = getNodeTextValue(nodes.item(0));
313        }
314        return (result == null || result.length() == 0)? null: result;
315    }
316 
317    // extract the text content of a single tag under a node
318    protected String getNodeContent(Element e, String key) {
319        NodeList nodes = e.getElementsByTagName(key);
320        String result = null;
321        if (nodes != null && nodes.getLength() > 0) {
322            result = getNodeTextValue(nodes.item(0));
323        }
324        return (result == null || result.length() == 0)? null: result;
325    }
326 	private String getNodeTextValue(Node n) {
327 		if(n.hasChildNodes()) {
328 			if(n.getFirstChild().getNodeName().equals("#text")) {
329 				return n.getFirstChild().getNodeValue();
330 			}
331 		}
332 		return "";
333 	}

335    // do an HTTP request, plus parse the result into an XML DOM
336    protected synchronized Document getHttpDocument(String url) 
337    	throws IOException, SAXException {
338    	
339        Document d = null;
340        d = this.builder.parse(url);
341        return d;
342    }
