3   import org.apache.commons.lang.math.NumberUtils;
4   import org.archive.format.cdx.CDXLine;
5   
6   
7   
8   public class DupeTimestampBestStatusFilter extends WrappedProcessor {
9   	final static int WORST_HTTP_CODE = 9999;
10  	
11  	protected String lastTimestamp;
12  
13  	protected int bestHttpCode = WORST_HTTP_CODE;
14  	protected int timestampDedupLength;
15  	
16  	protected String[] noCollapsePrefix;
17  	
18  	public DupeTimestampBestStatusFilter(BaseProcessor output, int timestampDedupLength, String[] noCollapsePrefix)
19  	{
20  		super(output);
21  		this.timestampDedupLength = timestampDedupLength;
22  		this.noCollapsePrefix = noCollapsePrefix;
23  	}
24  	
25  	@Override
26  	public int writeLine(CDXLine line)
27  	{
28  		if (include(line)) {
29  			return super.writeLine(line);
30  		} else {
31  			return 0;
32  		}
33  	}
34  	
35  	public boolean include(CDXLine line) {
36  		
37  		if (timestampDedupLength <= 0) {
38  			return true;
39  		}
40  		
41  		// If starts with special no collapse prefix, then always include
42  		if (noCollapsePrefix != null) {
43  			for (String prefix : noCollapsePrefix) {
44  				if (line.getFilename().startsWith(prefix)) {
45  					return true;
46  				}
47  			}
48  		}
49  		
50  		String timestamp = line.getTimestamp();
51  		timestamp = timestamp.substring(0, Math.min(timestampDedupLength, timestamp.length()));
52  		int httpCode = NumberUtils.toInt(line.getStatusCode(), WORST_HTTP_CODE);
53  		
54  		boolean isDupe = false;
55  		
56  		if ((lastTimestamp != null) && timestamp.equals(lastTimestamp)) {
57  			if (httpCode < bestHttpCode) {
58  				bestHttpCode = httpCode;
59  			} else {
60  				isDupe = true;
61  			}
62  		} else {
63  			bestHttpCode = httpCode;
64  		}
65  		
66  		lastTimestamp = timestamp;
67  		
68  		return !isDupe;
69  	}
70  }
