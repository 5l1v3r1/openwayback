1   package org.archive.wayback.proxy;

3   import org.archive.wayback.ResultURIConverter;
4   import org.archive.wayback.WaybackConstants;
5   
6   /**
7    * {@link ResultURIConverter} that replaces "{@code https://}" with "{@code http://}".
8    * <p>For use in proxy-mode, where all access need to be plain HTTP.</p>
9    *
10   */
11  public class ProxyHttpsResultURIConverter implements ResultURIConverter {
12  
13  	@Override
14  	public String makeReplayURI(String datespec, String url) {
15  		
16  		if (url.startsWith(WaybackConstants.HTTPS_URL_PREFIX)) {
17  			url = WaybackConstants.HTTP_URL_PREFIX + url.substring(WaybackConstants.HTTPS_URL_PREFIX.length());
18  		} else if (!url.startsWith(WaybackConstants.HTTP_URL_PREFIX)) {
19  			url = WaybackConstants.HTTP_URL_PREFIX + url;
20  		}
21  		
22  		return url;
23  	}
24  }
