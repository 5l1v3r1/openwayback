20  package org.archive.wayback.accesscontrol.robotstxt;
21  
22  import java.io.BufferedReader;
23  import java.io.IOException;
24  import java.io.InputStream;
25  import java.io.InputStreamReader;
26  import java.util.ArrayList;
27  import java.util.HashMap;
28  import java.util.LinkedList;
29  import java.util.List;
30  import java.util.logging.Logger;
31  import java.util.regex.Matcher;
32  import java.util.regex.Pattern;
33  
34  import org.archive.wayback.util.ByteOp;
35  
36  /**
37   * Class which parses a robots.txt file, storing the rules contained therein,
38   * and then allows for testing if path/userAgent tuples are blocked by those
39   * rules.
40   *
41   * @author brad
42   * @version $Date$, $Revision$
43   */
44  public class RobotRules {
45  	
46  	private static final long serialVersionUID = 2917420727021840982L;
47  	private static final Logger LOGGER = Logger.getLogger(RobotRules.class
48  			.getName());
49  	/**
50  	 * Special name for User-agent which matches all values
51  	 */
52  	public static final String GLOBAL_USER_AGENT = "*";
53  	
54  	protected static final Pattern USER_AGENT_PATTERN = Pattern.compile("(?i)^User-agent\\s*:(.*)");
55  	protected static final Pattern DISALLOW_PATTERN = Pattern.compile("(?i)Disallow\\s*:(.*)");
56  	protected static final Pattern ALLOW_PATTERN = Pattern.compile("(?i)Allow\\s*:(.*)");
57  	
58  	private boolean bSyntaxErrors = false;
59  	private HashMap<String, ArrayList<String>> rules = 
60  		new HashMap<String, ArrayList<String>>();
61  
62  	private LinkedList<String> userAgents = new LinkedList<String>();
63  
64  	/**
65  	 * @return true if the robots.txt file looked suspicious, currently meaning
66  	 * we found a Disallow rule that was not preceded by a "User-agent:" line
67  	 */
68  	public boolean hasSyntaxErrors() {
69  		return bSyntaxErrors;
70  	}
71  	
72  	/**
73  	 * @return a List of all UserAgents Found in the Robots.txt document
74  	 */
75  	public List<String> getUserAgentsFound() {
76  		return userAgents;
77  	}
78  	
79  	/**
80  	 * Read rules from InputStream argument into this RobotRules, as a 
81  	 * side-effect, sets the bSyntaxErrors property.
82  	 * 
83  	 * @param is InputStream containing the robots.txt document
84  	 * @throws IOException for usual reasons
85  	 */
86  	public void parse(InputStream is) throws IOException {
87  		
88  		BufferedReader br = new BufferedReader(new InputStreamReader(
89  				(InputStream) is,ByteOp.UTF8));
90          String read;
91          boolean allowRuleFound = false;
92          // true if curr or last line read was a User-agent line 
93          boolean currLineUA = false; 
94          boolean lastLineUA = false;
95          ArrayList<String> current = null;
96          while (br != null) {
97              lastLineUA = currLineUA;
98              do {
99                  read = br.readLine();
100                 // Skip comments & blanks
101             } while ((read != null) && ((read = read.trim()).startsWith("#") ||
102                 read.length() == 0));
103             if (read == null) {
104             	br.close();
105             	br = null;
106             } else {
107                 currLineUA = false;
108                 int commentIndex = read.indexOf("#");
109                 if (commentIndex > -1) {
110                     // Strip trailing comment
111                     read = read.substring(0, commentIndex);
112                 }
113                 read = read.trim();
114                 Matcher uaMatcher = USER_AGENT_PATTERN.matcher(read);
115                 if (uaMatcher.matches()) {
116                     String ua = uaMatcher.group(1).trim().toLowerCase();
117                     if (current == null || current.size() != 0 || allowRuleFound || !lastLineUA) {
118                         // only create new rules-list if necessary
119                         // otherwise share with previous user-agent
120                         current = new ArrayList<String>();
121                     }
122                     rules.put(ua, current);
123                     allowRuleFound = false;
124                     currLineUA = true;
125                     LOGGER.fine("Found User-agent(" + ua + ") rules...");
126                     continue;
127                 }
128                 Matcher disallowMatcher = DISALLOW_PATTERN.matcher(read);
129                 if (disallowMatcher.matches()) {
130                 	if (current == null) {
131                         // buggy robots.txt
132                     	bSyntaxErrors = true;
133                         continue;
134                     }
135                     String path = disallowMatcher.group(1).trim();
136 					// Disallow: without path is just ignored.
137 					if (!path.isEmpty())
138                     current.add(path);
139                     continue;
140                 }
141                 Matcher allowMatcher = ALLOW_PATTERN.matcher(read);
142                 if (allowMatcher.matches()) {
143                 	// Mark that there was an allow rule to clear the current list for next user-agent
144                 	allowRuleFound = true;
145                 }
146                 // unknown line; do nothing for now
147                 
148                 // TODO: check for "Allow" lines, and flag a syntax error if 
149                 //       we encounter any unknown lines?
150             }
151         }
152     }
153 	
154 	private boolean blocksPath(String path, String curUA, List<String> uaRules) {
155 		for (String disallowedPath : uaRules) {
156 			if (disallowedPath.isEmpty()) {
157 				// This is for extra caution. Empty path shouldn't be added
158 				// to uaRules in the first place.
159 				continue;
160 			}
161 			if (disallowedPath.equals("/") || path.startsWith(disallowedPath)) {
162 					return true;
163 			}
164 		}
165 		return false;
166 	}
167 	
168 	/**
169 	 * Checks first the specified ua UserAgent, if rules are present for it,
170 	 * and then falls back to using rules for the '*' UserAgent.
171 	 * 
172 	 * @param path String server relative path to check for access
173 	 * @param ua String user agent to check for access
174 	 * @return boolean value where true indicates the path is blocked for ua
175 	 */
176 	public boolean blocksPathForUA(String path, String ua) {
177 		final String lcua = ua.toLowerCase();
178 		if (rules.containsKey(lcua)) {
179 			return blocksPath(path, ua, rules.get(lcua));
180 		}
181 		if (rules.containsKey(GLOBAL_USER_AGENT)) {
182 			return blocksPath(path, GLOBAL_USER_AGENT,
183 					rules.get(GLOBAL_USER_AGENT));			
184 		}
185 		return false;
186 	}
187 }
