3   import org.archive.wayback.ResultURIConverter;
4   import org.archive.wayback.proxy.ProxyHttpsResultURIConverter;
5   
6   /**
7    * {@link ContextResultURIConverterFactory} that simply returns given {@link ResultURIConverter},
8    * regardless of usage context.
9    * <p>This class is better called <em>Context Independent</em> rather than <em>Identity</em>.</p>
10   *
11   */
12  public class IdentityResultURIConverterFactory implements
13  		ContextResultURIConverterFactory {
14  	
15  	protected ResultURIConverter converter;
16  	
17  	/**
18  	 * constructor for returning {@link ProxyHttpsResultURIConverter}.
19  	 */
20  	public IdentityResultURIConverterFactory()
21  	{
22  		this.converter = new ProxyHttpsResultURIConverter();
23  	}
24  	/**
25  	 * constructor for returning <code>converter</code>.
26  	 * @param converter pre-built ResultURIConverter to be returned.
27  	 */
28  	public IdentityResultURIConverterFactory(ResultURIConverter converter)
29  	{
30  		this.converter = converter;
31  	}
32  
33  	@Override
34  	public ResultURIConverter getContextConverter(String flags) {
35  		return converter;
36  	}
37  }
