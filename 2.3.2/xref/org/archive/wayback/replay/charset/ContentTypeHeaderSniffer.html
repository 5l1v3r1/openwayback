1   package org.archive.wayback.replay.charset;

3   import java.util.Map;
4   import java.util.Map.Entry;
5   
6   import org.archive.wayback.core.Resource;
7   
8   /**
9    * {@link EncodingSniffer} obtaining character encoding from
10   * {@code Content-Type} HTTP header.
11   * <p>Step 4 of WHAT-NG <em>character encoding sniffing</em> prescription.
12   */
13  public class ContentTypeHeaderSniffer extends BaseEncodingSniffer {
14  	@Override
15  	public String sniff(Resource resource) {
16  		Map<String, String> httpHeaders = resource.getHttpHeaders();
17  		for (Entry<String, String> e : httpHeaders.entrySet()) {
18  			String headerKey = e.getKey();
19  			if (headerKey.equalsIgnoreCase(HTTP_CONTENT_TYPE_HEADER)) {
20  				String ctype = e.getValue();
21  				String charsetName = contentTypeToCharset(ctype);
22  				return charsetName;
23  			}
24  		}
25  		return null;
26  	}
27  }
