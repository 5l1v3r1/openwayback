21  
22  import java.io.IOException;
23  import java.util.regex.Matcher;
24  import java.util.regex.Pattern;
25  
26  import javax.servlet.ServletException;
27  import javax.servlet.http.HttpServletRequest;
28  import javax.servlet.http.HttpServletResponse;
29  
30  import org.archive.wayback.ResultURIConverter;
31  import org.archive.wayback.core.CaptureSearchResult;
32  import org.archive.wayback.core.CaptureSearchResults;
33  import org.archive.wayback.core.Resource;
34  import org.archive.wayback.core.WaybackRequest;
35  import org.archive.wayback.replay.HttpHeaderProcessor;
36  import org.archive.wayback.replay.TextDocument;
37  import org.archive.wayback.replay.TextReplayRenderer;
38  import org.archive.wayback.util.Timestamp;
39  
40  /**
41   * {@link TextReplayRenderer} that rewrites URLs in JavaScript resource for
42   * replay in ArchivalURL mode, and inserts {@code jspInserts} at the top of the
43   * document.
44   * <p>This class looks up URLs in JavaScript by single regular expression set to
45   * its {@code regex} property, and rewrite them with {@link ResultURIConverter}
46   * passed to {@code updatePage} method (does not use the one set to {@link TextDocument}.)</p>
47   * <p>Regular expression shall match single URL. Be sure to enclose URL in capture group
48   * (i.e. {@code (}...{@code )}), or no rewrite will happen.  Regular expression can have
49   * optional capture group before URL capture group, which may be necessary for doing
50   * more complicated match.  Text captured in the first group will be copied to the
51   * output.</p>
52   * <p>There's an alternative implementation
53   * {@link ArchivalURLJSStringTransformerReplayRenderer}, which supports
54   * multiple rewrite patterns through <code>StringTransformer</code>.</p>
55   *
56   * @see ResultURIConverter
57   * @see ArchivalURLJSStringTransformerReplayRenderer
58   * @author brad
59   */
60  
61  public class ArchivalUrlJSReplayRenderer extends TextReplayRenderer {
62  	/**
63  	 * @param httpHeaderProcessor which should process HTTP headers
64  	 */
65  	public ArchivalUrlJSReplayRenderer(
66  			HttpHeaderProcessor httpHeaderProcessor) {
67  		super(httpHeaderProcessor);
68  	}
69  
70  	private final static Pattern defaultHttpPattern = Pattern
71  			.compile("(https?://[A-Za-z0-9:_@.-]+)");
72  	
73  	private Pattern pattern = defaultHttpPattern;
74  	
75  	/**
76  	 * regular expression for matching URLs.
77  	 * @param regex
78  	 */
79  	public void setRegex(String regex)
80  	{
81  		pattern = Pattern.compile(regex);
82  	}
83  	
84  	public String getRegex()
85  	{
86  		return pattern.pattern();
87  	}
88  
89  	protected void updatePage(TextDocument page,
90  			HttpServletRequest httpRequest, HttpServletResponse httpResponse,
91  			WaybackRequest wbRequest, CaptureSearchResult result,
92  			Resource resource, ResultURIConverter uriConverter,
93  			CaptureSearchResults results) throws ServletException, IOException {
94  		String resourceTS = result.getCaptureTimestamp();
95  		String captureTS = Timestamp.parseBefore(resourceTS).getDateStr();
96  
97  		StringBuilder sb = page.sb;
98  		StringBuffer replaced = new StringBuffer(sb.length());
99  		Matcher m = pattern.matcher(sb);
100 		
101 		// If at least 2 groups, prepend before 2nd group and include 1st group. Allows for more sophisticated matching.
102 		// Otherwise, insert before 1st group
103 		if (m.groupCount() > 1) {
104 			while (m.find()) {
105 				String beforeHost = m.group(1);
106 				String host = m.group(2);
107 				String replacement = uriConverter.makeReplayURI(captureTS, host);
108 				m.appendReplacement(replaced, beforeHost + replacement);
109 			}	
110 		} else {
111 			while (m.find()) {
112 				String host = m.group(1);
113 				String replacement = uriConverter.makeReplayURI(captureTS, host);
114 				m.appendReplacement(replaced, replacement);
115 			}			
116 		}
117 		
118 		m.appendTail(replaced);
119 		// blasted StringBuilder/StringBuffer... gotta convert again...
120 		page.sb.setLength(0);
121 		page.sb.ensureCapacity(replaced.length());
122 		page.sb.append(replaced);
123 
124 		// if any JS-specific jsp inserts are configured, run and insert...
125 		page.insertAtStartOfDocument(buildInsertText(page, httpRequest,
126 				httpResponse, wbRequest, results, result, resource));
127 	}
128 }
