37  /**
38   * Classic ReplayRenderer which uses a combination of server-side modification 
39   * and embedded javascript to rewrite URLs within an HTML page to make embedded
40   * URLs point back to a specific ArchivalURL AccessPoint.
41   *
42   * @author brad
43   */
44  public class ClientSideHTMLReplayRenderer extends TextReplayRenderer {
45  	/**
46  	 * @param httpHeaderProcessor which should process HTTP headers
47  	 */
48  	public ClientSideHTMLReplayRenderer(HttpHeaderProcessor httpHeaderProcessor) {
49  		super(httpHeaderProcessor);
50  	}
51  
52  	/* (non-Javadoc)
53  	 * @see org.archive.wayback.replay.HTMLReplayRenderer#updatePage(org.archive.wayback.replay.HTMLPage, javax.servlet.http.HttpServletRequest, javax.servlet.http.HttpServletResponse, org.archive.wayback.core.WaybackRequest, org.archive.wayback.core.CaptureSearchResult, org.archive.wayback.core.Resource, org.archive.wayback.ResultURIConverter, org.archive.wayback.core.CaptureSearchResults)
54  	 */
55  	@Override
56  	protected void updatePage(TextDocument page, HttpServletRequest httpRequest,
57  			HttpServletResponse httpResponse, WaybackRequest wbRequest,
58  			CaptureSearchResult result, Resource resource,
59  			ResultURIConverter uriConverter, CaptureSearchResults results) 
60  		throws ServletException, IOException {
61  
62  		page.resolvePageUrls();
63  		
64  		// hope we're configured to use ClientSideJSInsert.jsp!
65  		// insert the new content:
66  		page.insertAtEndOfBody(buildInsertText(page, httpRequest, httpResponse,
67  				wbRequest, results, result, resource));
68  	}
69  }
