1   package org.archive.cdxserver.processor;
2   
3   import java.util.Date;
4   import java.util.TreeMap;
5   
6   import org.archive.format.cdx.CDXLine;
7   import org.archive.util.ArchiveUtils;
8   
9   public class ClosestTimestampSorted extends WrappedProcessor {
10      
11      enum Dir
12      {
13          ANY,
14          PREV,
15          NEXT,
16      };
17      
18      Dir dir;
19      TreeMap<Long, CDXLine> closestLines;
20      Long target;
21      int limit;
22      boolean done = false;
23      
24      protected Long convTimestamp(String timestamp)
25      {        
26          return ArchiveUtils.getDate(timestamp, new Date()).getTime();
27      }
28      
29      public ClosestTimestampSorted(BaseProcessor output, String target, int limit) {
30          super(output);
31          
32          if (target.startsWith("-")) {
33              target = target.substring(1);
34              dir = Dir.PREV;
35          } else if (target.startsWith("^")) {
36              target = target.substring(1);
37              dir = Dir.NEXT;
38          } else {
39              dir = Dir.ANY;
40          }
41          
42          this.target = convTimestamp(target);
43          this.limit = (limit > 0 ? limit : Integer.MAX_VALUE);
44          closestLines = new TreeMap<Long, CDXLine>();
45      }
46  
47      @Override
48      public int writeLine(CDXLine line) {
49          if (done) {
50              return Integer.MAX_VALUE;
51          }
52          
53          Long curr = convTimestamp(line.getTimestamp());
54          Long diff;
55          
56          switch (dir) {            
57          case PREV:
58              if (curr < target) {
59                  diff = target - curr;
60              } else {
61                  diff = target + curr;
62              }
63              break;
64              
65          case NEXT:
66              if (curr > target) {
67                  diff = curr - target;
68              } else {
69                  diff = (target - curr) + target * 2;
70              }
71              break;
72              
73          case ANY:
74          default:
75              diff = Math.abs(curr - target);
76              break;
77          }
78          
79          if (closestLines.size() == limit) {
80              // Assumes ascending timestamp input 
81              if (diff > closestLines.lastKey()) {
82                  done = true;
83                  return writeSorted();
84              }
85          }
86          
87          closestLines.put(diff, line);
88        
89          if (closestLines.size() > limit) {
90              closestLines.remove(closestLines.lastKey());
91          }
92          
93          return 0;
94      }
95  
96      protected int writeSorted() {
97          int count = 0;
98          for (CDXLine line : closestLines.values()) {
99              super.writeLine(line);
100             ++count;
101         }
102         closestLines.clear();
103         return count;
104     }
105 
106     @Override
107     public void writeResumeKey(String resumeKey) {
108         writeSorted();
109         super.writeResumeKey(resumeKey);
110     }
111 
112     @Override
113     public void end() {
114         writeSorted();
115         super.end();
116     }
117 }
