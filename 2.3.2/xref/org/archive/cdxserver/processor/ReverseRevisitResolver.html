3   import java.util.LinkedList;

5   import org.archive.format.cdx.CDXLine;

7   public class ReverseRevisitResolver extends RevisitResolver {
8   
9   	public ReverseRevisitResolver(
10  			BaseProcessor output, 
11  			boolean showDupeCount) {
12  		
13  		super(output, showDupeCount);
14  	}
15  	
16  	class RevisitTrack extends DupeTrack {
17  		LinkedList<CDXLine> lines;
18  		
19  		void add(CDXLine line)
20  		{
21  			if (lines == null) {
22  				lines = new LinkedList<CDXLine>();
23  			}
24  			lines.add(line);
25  		}
26  		
27  		void fillRevisits(CDXLine origLine)
28  		{
29  			if (lines != null) {
30  				for (CDXLine revisitLine : lines) {
31  					ReverseRevisitResolver.this.fillRevisit(revisitLine, origLine);
32  				}
33  				lines.clear();
34  			}
35  		}
36  	}
37  	
38  	@Override
39      protected DupeTrack createDupeTrack()
40      {
41      	return new RevisitTrack();
42      }
43      
44  	@Override
45      protected void handleLine(DupeTrack counter, CDXLine line, boolean isDupe) {
46      	RevisitTrack revisits = (RevisitTrack)counter;
47  		
48  		super.fillBlankOrig(line);
49  		
50  		if (isRevisit(line)) {
51  			revisits.add(line);
52  		} else {
53  			revisits.fillRevisits(line);
54  		}
55      }
56  }
