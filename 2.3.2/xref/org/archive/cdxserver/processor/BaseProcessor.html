1   package org.archive.cdxserver.processor;

3   import org.archive.cdxserver.writer.CDXWriter;
4   import org.archive.format.cdx.CDXLine;
5   import org.archive.format.cdx.FieldSplitFormat;
6   
7   /**
8    * {@code BaseProcessor} is an interface for a receiver
9    * of {@link CDXLine}s.
10   * <p>Implementation may be final output formatter ({@link CDXWriter}
11   * subclasses for example), or an intermediary processor that performs
12   * transformation and/or filtering on the sequence of CDXLines.</p>
13   * <p>{@code CDXServer} starts from {@code CDXWriter} and builds up
14   * nested pipeline of {@code BaseProcessor}s, and then calls following
15   * methods in sequence on the {@code BaseProcessor} at the top:
16   * <ol>
17   * <li>modifyOutputFormat(FieldSplitFormat)</li>
18   * <li>begin()</li>
19   * <li>for each CDXLine:</li>
20   * <ol>
21   * <li>trackLine(CDXLine)</li>
22   * <li>writeLine(CDXLine)</li>
23   * </ol>
24   * <li>writeResumeKey(String) (if {@code showResumeKey})</li>
25   * <li>end()</li>
26   * </ol>
27   */
28  public interface BaseProcessor {
29  	/**
30  	 * This method will be called just before looping over
31  	 * the sequence of CDX lines.
32  	 * Intermediary processor must call {@code begin()}
33  	 * on nested processor.
34  	 */
35  	public void begin();
36  
37  	/**
38  	 * Called on each CDX line, just before timestamp range filtering
39  	 * (to and from parameters), regexp filtering, and {@code collapser}
40  	 * processing. Typically used for counting the number of CDX lines
41  	 * collapsed / grouped.
42  	 * @param line CDX line
43  	 */
44  	public void trackLine(CDXLine line);
45  
46  	/**
47  	 * Process {@code line}.
48  	 * @param line {@code CDXLine}
49  	 * @return 1 if {@code line} is sent to output, 0 otherwise.
50  	 */
51  	public int writeLine(CDXLine line);
52  
53  	/**
54  	 * Write resumption key.
55  	 * Only the final {@code CDXWriter} should do
56  	 * actual work. All intermediaries shall simply call
57  	 * {@code writeResumeKey(resumeKey)} on nested processor.
58  	 * @param resumeKey
59  	 */
60  	public void writeResumeKey(String resumeKey);
61  
62  	/**
63  	 * Called at the end.
64  	 * Perform any clean ups / finalizations here.
65  	 * Intermediaries processor should call {@code end()}
66  	 * on nested processor.
67  	 */
68  	public void end();
69  
70  	/**
71  	 * Return output format (list of fields), given input format {@code format}.
72  	 * Intermediaries should call {@code modifyOutputFormat(format)} on nested
73  	 * processor first, then make appropriate changes to it if they add/remove
74  	 * fields.
75  	 * @param format input format
76  	 * @return output format
77  	 */
78  	public FieldSplitFormat modifyOutputFormat(FieldSplitFormat format);
79  }
