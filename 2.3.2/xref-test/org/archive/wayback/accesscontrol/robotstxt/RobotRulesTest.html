51  	protected RobotRules load(String txt) throws IOException {
52  		rr.parse(new ByteArrayInputStream(txt.getBytes("UTF-8")));
53  		assertFalse(rr.hasSyntaxErrors());
54  		return rr;
55  	}

69  	/**
70  	 * Disallow: with empty path has no effect.
71  	 */
72  	public void testEmptyDisallowHasNoEffect() throws Exception {
73  		final String testString = 
74  				"User-agent: *\n" +
75  				"Disallow:\n" +
76  				"Disallow: /wp-admin\n";
77  		load(testString);
78  
79  		assertFalse(rr.blocksPathForUA("/", WB_UA));
80  		// Path-less Disallow: should just be ignored. It shall not
81  		// disable other non-empty Disallow's. ARI-4212.
82  		assertTrue(rr.blocksPathForUA("/wp-admin/index.php", WB_UA));
83  	}

151 	/**
152 	 * multiple records for different User-agent's.
153 	 * <p>while RFC states "the format logically consists of a non-empty set
154 	 * or records, separated by blank lines", Google's documentation has no
155 	 * mention to blank lines as group separator - instead, it recognizes a sequence 
156 	 * of User-agent: as the start of "group". So this sample is syntax error per
157 	 * RFC, but okay according to Google/Yahoo/Bing/Ask convention.
158 	 * @throws Exception
159 	 */
160 	public void testNonBlocksPathForUA() throws Exception {
161 		final String testString =
162 				"User-agent: *\n" +
163 				"Allow: /\n" +
164 				"User-agent: Google-bot\n" +
165 				"Disallow: /\n";
166 		load(testString);
167 
168 		assertFalse(rr.blocksPathForUA("/", WB_UA));
169 	}

173 	 * next user-agent should start a new group.
174 	 * ia_archiver ALLOWED as disallow rule only applies to B.
175 	 */
176 	public void testMultiUAWithOtherLinesLine() throws Exception {

206 	/**
207 	 * this is a syntax error per RFC, but ok with Google/Yahoo/Bing/Ask convention.
208 	 * blank lines are permitted within a group to improve readability. they are simply
209 	 * ignored, don't end group.
210 	 */
211 	public void testBlankLineInGroup() throws Exception {
212 		final String testString =
213 				"User-agent: *\n" +
214 				"Disallow: /media/\n" +
215 				"\n" +
216 				"Disallow: /cgi-bin/\n" +
217 				"# images\n" + 
218 				"Disallow: /img/\n" +
219 				"Disallow: /icon/\n" +
220 				"\n" +
221 				"\n" +
222 				"Disallow: /actions/\n" +
223 				"\n" +
224 				"# sensitive stuff\n" +
225 				"Disallow: /etc\n";
226 		load(testString);
227 		
228 		assertTrue(rr.blocksPathForUA("/cgi-bin/", WB_UA));

252 	/**
253 	 * path matching basics. substring-based, <code>/</code> is no special,
254 	 * and case-sensitive.
255 	 */
256 	public void testSubpath() throws Exception {
257 		String testString =
258 				"User-agent: *\n" +
259 				"Disallow: /media\n" +
260 				"Disallow: /hidden/\n";
261 		load(testString);
262 		
263 		assertFalse(rr.blocksPathForUA("/", WB_UA));
264 		assertTrue(rr.blocksPathForUA("/media/theme.mp3", WB_UA));
265 		assertTrue(rr.blocksPathForUA("/media?order=2", WB_UA));
266 		assertFalse(rr.blocksPathForUA("/images/logo.png", WB_UA));
267 		assertFalse(rr.blocksPathForUA("/hidden", WB_UA));
268 		assertTrue(rr.blocksPathForUA("/hidden/index.html", WB_UA));
269 		
270 		// path match is case-sensitive
271 		assertFalse(rr.blocksPathForUA("/Hidden/notreally.mp3", WB_UA));
272 	}

299 	/**
300 	 * <p>By GYBA convention, if multiple
301 	 * disallow and allow directives matches the URL, the most specific
302 	 * rule based on the length of the path will win over less specific
303 	 * (shorter) ones.</p> 
304 	 * <p>RFC says differently: "a robot must attempt to match
305 	 * the paths in Allow and Disallow lines against the URL, in the
306 	 * order they occur in the record. The first match found is used."</p>
307 	 * <p>we follow GYBA convention here.</p>
308 	 * @throws Exception
309 	 */
310 	public void testMostSpecificPathPrevails() throws Exception {
311 		final String testString =
312 				"User-agent: *\n" +
313 				"Disallow: /\n" +
314 				"Allow: /media\n" +
315 				"Disallow: /media/private\n";
316 		load(testString);
317 		
318 		// (some assertions are disabled until we decide to fix it)
319 		assertTrue(rr.blocksPathForUA("/index.html", WB_UA));
320 //		assertFalse(rr.blocksPathForUA("/media/toc.html", WB_UA));
321 //		assertFalse(rr.blocksPathForUA("/media/priv", WB_UA));
322 		assertTrue(rr.blocksPathForUA("/media/private.html", WB_UA));
323 		assertTrue(rr.blocksPathForUA("/media/private/voice.mp3", WB_UA));
324 	}
